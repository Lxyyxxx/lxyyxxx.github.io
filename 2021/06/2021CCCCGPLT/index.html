#include<bits/stdc++.h>

bool judge(string s){
    if (s.find("qiandao")!=s.npos)
        return 0;
    if (s.find("easy")!=s.npos)
        return 0;
    return 1;
}

int main(){
    int N, M;
    cin>>N>>M;
    string s;
    bool f=0;
    getchar();
    int cnt=0;
    for (int i=0; i<N; i++){
        getline(cin, s);
        if (judge(s)){
            cnt++;
            if (cnt>M){
                cout<<s;

        }

}

# L1-080 乘法口诀数列

#include<bits/stdc++.h>
using namespace std;
int main(){
    vector<int>  v;
    int a1, a2, cnt;
    cin>>a1>>a2>>cnt;
    v.push_back(a1);
    v.push_back(a2);
    int s = v.size();
    int a3;
    int cur = 0;
    string ss;
    while(s<cnt){
        a3 = v[cur]*v[cur + 1];
        ss = to_string(a3);
        for (auto &i:ss)
            v.push_back(i-'0');
        s = v.size();
        cur ++;
    }
    cout<<v[0];
    for (int i=1; i<cnt; i++){
        cout<<' '<<v[i];
    }
    return 0;
}

# L2-037 包装机

#include<bits/stdc++.h>
using namespace std;

const int n = 110, m = 1010;

int main(){
    int N, M, Smax;
    cin>>N>>M>>Smax;
    char q[n][m];
    int cur_qf[n]={0}, cur_qr[n]={0};
    char s[100010];
    int cur_s=0;
    char t;
    for (int i=1; i<=N; i++){
        for (int j=0; j<M; j++){
            cin>>t;
            q[i][cur_qr[i]] = t;
            cur_qr[i]++;
        }
    }
    int x;
    while(cin>>x&&x!=-1){
        if (x==0){
            // 如果筐是空的，按 0 号按钮也不会发生任何事
            if(cur_s > 0) {
                cout<<s[cur_s-1];
                cur_s--;
            }
        }else{
            // 如果轨道已经空了，再按对应的按钮不会发生任何事
            if (cur_qf[x] == cur_qr[x])
                continue;
            // 当筐已经满了，但仍然有某条轨道的按钮被按下时，系统应强制启动 0 号键，先从筐里抓出一件物品
            if (cur_s>=Smax){
                cout<<s[cur_s-1];
                cur_s--;
            }
            // 再将对应轨道的物品推落
            s[cur_s] = q[x][cur_qf[x]];
            cur_s++;
            cur_qf[x]++;
        }
    }
    return 0;
}

# L2-038 病毒溯源

#include<bits/stdc++.h>
using namespace std;

const int n=1e4+10;
set<int>g[n];
int ans = -1;
vector <int>path;
typedef pair<int, int> pii;

void dfs(int cur, vector<int> &v){
    if (g[cur].empty()){
        int t = v.size();
        if (t>ans){
            ans = t;
            path = v;
        }
        return;
    }
    for(auto &i:g[cur]){
        v.push_back(i);
        dfs(i, v);
        v.pop_back();
    }
}

int main(){
    int N;
    cin>>N;
    int k;
    bool vis[N] = {false};
    int b;
    for (int i=0; i<N; i++){
        cin>>k;
        for (int j=0; j<k; j++){
            cin>>b;
            g[i].insert(b);
            vis[b]=1;
        }
    }
    int s;
    for (int i=0; i<N; i++){
        if (!vis[i]){
            s = i;
            break;
        }   
    }
    vector<int> v;
    v.push_back(s);
    dfs(s, v);
    cout<<path.size()<<'\n';
    cout<<path[0];
    for (int i=1; i<path.size(); i++){
        cout<< ' '<< path[i];
    } 
    return 0;
}

# L2-040 哲哲打游戏

#include<bits/stdc++.h>
using namespace std;

const int n = 1e5+10, m = 1e5+10;
int cd[m];
vector<int>g[n];

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0), cout.tie(0);
    int N, M;
    cin>>N>>M;
    int K;
    int nx;
    for (int i=1;i<=N; i++){
        cin>>K;
        g[i].push_back(-1);
        while(K--){
            cin>>nx;
            g[i].push_back(nx);
        }
    }
    int op, j;
    int jq = 1;
    while(M--){
        cin>>op>>j;
        if (op == 0){
            int t = g[jq][j];
            jq = t;
        }else if(op==1){
            cd[j] = jq;
            cout<<jq<<'\n';
        }else if(op==2){
            jq = cd[j];
        }
    }
    cout<<jq;
    return 0;
}
