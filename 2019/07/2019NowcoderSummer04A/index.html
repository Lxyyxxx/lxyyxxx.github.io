「2019牛客暑期多校 - 第四场」A - meeting
2019-07-30 22:59   folder 2019牛客暑期多校训练营 - 第四场 label 图论, 树的直径

题目传送门

_Say

就是求最远两个人之间的距离d，答案为 ⌈d2⌉\lceil \dfrac{d}{2} \rceil⌈​2​​d​​⌉
求d又相当于求树的直径，运用两次BFS即可

  1. 任取一人，找出与他最远的那个人x
  2. 找出离x最远的人y，x与y之间的路径即为d

_Code

#include<bits/stdc++.h>
using namespace std;

int n,k,a,b;
const int N=1e5+5;
bool vis[N]; // visited?
int dis[N]; // distance
vector<int>Graph[N]; // graph
bool per[N]={false}; // is person?
int last=0; // farest person

void bfs(int s,bool mark){ // mark bfs1 or bfs2
    memset(vis,~mark,sizeof vis);
    queue<int>q;
    q.push(s);
    vis[s]=mark;
    dis[s]=0;
    while(!q.empty()){
        int u=q.front();q.pop();
        if(per[u])last=u; // recorde last
        for(auto v:Graph[u]){
            if(vis[v]!=mark){
                q.push(v);
                vis[v]=mark;
                dis[v]=dis[u]+1; // recorde distance
            }
        }
    }
}
int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    while(cin>>n>>k){
        for(int i=0;i<n-1;i++){
            int a,b;
            cin>>a>>b;
            Graph[a].push_back(b);
            Graph[b].push_back(a);
        }
        int xi;
        for(int i=0;i<k;i++){
            cin>>xi;
            per[xi]=true;
        }
        bfs(xi,0); // bfs1
        bfs(last,1); // bfs2
        cout<<(dis[last]+1)/2<<endl; // farest distance = dis[last]
    }
    return 0;
}
