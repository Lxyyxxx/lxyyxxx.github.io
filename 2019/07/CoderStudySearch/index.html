# DFS | Depth First Search | 深度优先搜索

  • 递归 or stack
  • 时间复杂度：O(∣V∣+∣E∣)O(|V|+|E|)O(∣V∣+∣E∣)（邻接链表）

Code（递归）

// 用于图
const int N=1e3+5;
bool visited[N]={false};
vector<int>Graph[N];
 
void dfs(int u){
    visited[u]=true;        // 将起点u设置为已访问
    for(auto &v:Graph[u]){  // 遍历每个与顶点u相连的顶点v 
        if(!visited[v]){    // 如果顶点v未被访问过    
            dfs(v);         // 递归访问顶点v
        }
    }
}

# BFS | Breadth First Search | 广度优先搜索

  • queue
  • 边数最少，路径最短
  • 时间复杂度：O(∣V∣+∣E∣)O(|V|+|E|)O(∣V∣+∣E∣)（邻接链表）

Code

const int N=1e3+5;
bool visited[N]={false};
vector<int>Graph[N];
 
void bfs(int s){
    queue<int>q;                  // 新建队列q
    q.push(s);                    // 起点s放进队列q
    visited[s]=true;              // 起点s设置为已访问
    while(!q.empty()){            // 当队列q非空时重复
        int u=q.front();q.pop();  // 获得队列q的第一个元素u
        for(auto &v:Graph[u]){    // 遍历每个与顶点u相连的顶点v
            if(!visited[v]){      // 如果顶点v未被访问过
                q.push(v);        // 顶点v放进队列q
                visited[v]=true;  // 顶点v设置为已访问
            }
        }
    }
}

# 双端队列 BFS | 0-1 BFS

  • deque
  • 边权值只有0或1
  • 最短路
  • 越靠近队首，权值越小 and 0与1间有分隔
  • 时间复杂度：O(n)O(n)O(n)

Code

typedef pair<int,int> pii;
const int N=1e3+5;
int dis[N]; // save distance begin to i
bool vis[N]={false}; // visited?
vector<pii>g[N]; // graph
 
void bfs(int S){
    memset(dis,0x3f,sizeof dis); // init dis[N]={∞}
    dis[S]=0;
    deque<int>q;
    q.push_back(s);
    while(q.size()){ // deque isn't empty
        int x=q.front();q.pop_front();
        if(vis[x])continue;
        vis[x]=true;
        for(auto e:g[x]){
            int y=e.first,w=e.second;
            if(!vis[y]&&dis[y]>dis[x]+w){ // update
                dis[y]=dis[x]+w;
            }
            if(w)q.push_back(y); // w=1，添加到队尾
            else q.push_front(y); // w=0，添加到队首
        }
    }
}

# 优先队列 BFS

  • priority_queue
  • 最短路
  • 相当于堆优化的Dijkstra
  • 时间复杂度：O((n+m)log⁡m)O((n+m)\log m)O((n+m)logm)
typedef pair<int,int> pii;
const int N=1e3+5;
struct Node{
    int nx,nc; // current toPoint,cost
    bool operator<(const Node &node)const{return nc>node.nc}
};
vector<pii>g[N]; // graph
int dis[N]; // save distance begin to i
bool vis[N]={false}; // visited?
 
void dijkstra(int S){
    memset(dis,0x3f,sizeof dis); // init dis[N]={∞}
    dis[S]=0;
    priority_queue<Node>heap; // 优先队列
    heap.push({S,0});
    while(!heap.empty()){
        auto p=heap.top();heap.pop();
        int x=p.nx;
        if(vis[x])continue;
        vis[x]=true;
        for(auto e:g[x]){
            int y=e.first,w=e.second;
            if(!vis[y]&&dis[y]>dis[x]+w){ // update
                dis[y]=dis[x]+w;
                heap.push({y,dis[y]});
            }
        }
    }
}
