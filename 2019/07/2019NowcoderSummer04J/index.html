题目传送门

_Say

分层图最短路。
n个点m条边的无向图，有K次免费的机会，求S到T的最短路径。
从起点到i，免费j次的最短路径(i与x间有长度为e的边)：

dis[i][j]=min(dis[i][j−1],min(dis[x][j−1],dis[x][j]+e))\begin{aligned} dis[i][j]=min(dis[i][j-1],min(dis[x][j-1],dis[x][j]+e)) \end{aligned}​dis[i][j]=min(dis[i][j−1],min(dis[x][j−1],dis[x][j]+e))​​

_Code

#include<bits/stdc++.h>
using namespace std;

typedef pair<int,int> pii;
const int N=1e3+5;
struct Node{
    int nx,nc,nk; // current toPoint,cost,K
    bool operator<(const Node &node)const{return nc>node.nc||nc==node.nc&&nk>node.nk;}
};
vector<pii>g[N]; // graph
int dis[N][N];
bool vis[N][N]={false};
int n,m,S,T,K;

void dijkstra(){
    memset(dis,0x3f,sizeof dis); // init dis ∞
    dis[S][0]=0;
    priority_queue<Node>heap;
    heap.push({S,0,0});
    while(!heap.empty()){
        auto p=heap.top();heap.pop();
        int x=p.nx,k=p.nk;
        if(vis[x][k])continue;
        vis[x][k]=true;
        if(x==T)return;
        for(auto e:g[x]){ // second min
            int y=e.first,w=e.second;
            if(k<K&&dis[y][k+1]>dis[x][k]){ // can free
                dis[y][k+1]=dis[x][k];
                heap.push({y,dis[y][k+1],k+1});
            }
            if(dis[y][k]>dis[x][k]+w){ //can update
                dis[y][k]=dis[x][k]+w;
                heap.push({y,dis[y][k],k});
            }
        }
    }
}

int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    while(cin>>n>>m>>S>>T>>K){
        for(int i=0;i<m;i++){ // init graph
            int a,b,l;
            cin>>a>>b>>l;
            g[a].push_back({b,l});
            g[b].push_back({a,l});
        }
        dijkstra();
        int ans=numeric_limits<int>::max(); // ans = max int
        for(int i=0;i<=K;i++){
            if(ans>dis[T][i])ans=dis[T][i]; // first min
        }
        cout<<ans<<endl;
    }
    return 0;
}
