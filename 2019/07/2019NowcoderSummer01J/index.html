# _Say

签到题。
分数比大小，直接相乘用 unsigned long long 会超范围，可以选择用 __int128 或 Java大数。

# _Code

#include<iostream>
using namespace std;
long long x,a,y,b;
int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    while(cin>>x>>a>>y>>b){
        __int128 c=(__int128)x*b,d=(__int128)a*y;
        if(c==d)cout<<"="<<endl;
        else if(c<d)cout<<"<"<<endl;
        else cout<<">"<<endl;
    }
    return 0;
}
