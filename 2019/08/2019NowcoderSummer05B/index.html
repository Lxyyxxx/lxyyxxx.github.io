用矩陣表示

[xnxn−1]=[ab10][xn−1xn−2]\begin{bmatrix} x_n \\ x_{n-1} \end{bmatrix} = \begin{bmatrix} a & b \\ 1 & 0 \end{bmatrix} \begin{bmatrix} x_{n-1} \\ x_{n-2} \end{bmatrix} [xn​xn−1​​]=[a1​b0​][xn−1​xn−2​​]

可以推出

[xn+1xn]=[ab10]n[x1x0]\begin{bmatrix} x_{n+1} \\ x_n \end{bmatrix} = {\begin{bmatrix} a & b \\ 1 & 0 \end{bmatrix}}^n \begin{bmatrix} x_1 \\ x_0 \end{bmatrix} [xn+1​xn​​]=[a1​b0​]n[x1​x0​​]

然後用矩陣快速冪就能直接算，輸出xnx_nxn​即可。
nnn由於比較大是字符串，在算的時候需要簡單處理一下。
題解說和一般的快速幂差别在于：以 2 为单位倍增要改成以 10 为单位。不需要把十进位转成二进位。指数的每个位数还是要用二进制倍增计算，否则可能会TLE。

# _Code

#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
int x0,x1,a,b,MOD;
string n;
struct Mat{
    int mat[2][2];
    Mat(){memset(mat,0,sizeof(mat));}
    // 矩阵快速幂取模（兩個矩陣相乘，重載運算符）a*b
    friend Mat operator*(const Mat &a,const Mat &b){
        Mat c;
        /* 一般形式
        for(int i=0;i<2;i++)
            for(int j=0;j<2;j++)
                for(int k=0;k<2;k++)
                    c.mat[i][j]=(c.mat[i][j]+1ll*a.mat[i][k]*b.mat[k][j])%MOD;
        */
        // 由於是2*2矩陣，所以優化一下
        c.mat[0][0]=(1ll*a.mat[0][0]*b.mat[0][0]+1ll*a.mat[0][1]*b.mat[1][0])%MOD;
        c.mat[0][1]=(1ll*a.mat[0][0]*b.mat[0][1]+1ll*a.mat[0][1]*b.mat[1][1])%MOD;
        c.mat[1][0]=(1ll*a.mat[1][0]*b.mat[0][0]+1ll*a.mat[1][1]*b.mat[1][0])%MOD;
        c.mat[1][1]=(1ll*a.mat[1][0]*b.mat[0][1]+1ll*a.mat[1][1]*b.mat[1][1])%MOD;
        return c;
    }
};
// 矩陣快速幂取模（矩陣的指數，二進制倍增）x^y
Mat qpow(Mat x,int y){
    Mat tmp;tmp.mat[0][0]=tmp.mat[1][1]=1; // 單位矩陣 1 0 0 1
    while(y){
        if(y&1)tmp=tmp*x;
        x=x*x;
        y>>=1;
    }
    return tmp;
}
int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    while(cin>>x0>>x1>>a>>b){
        cin>>n>>MOD;
        Mat m1,ans;
        m1.mat[0][0]=a;m1.mat[0][1]=b;m1.mat[1][0]=1; // a b 1 0
        ans.mat[0][0]=ans.mat[1][1]=1; // 單位矩陣 1 0 0 1
        for(int i=n.size()-1;i>=0;i--){
            ans=ans*qpow(m1,n[i]-'0');
            m1=qpow(m1,10);
        }
        int xn=(1ll*ans.mat[1][0]*x1+1ll*ans.mat[1][1]*x0)%MOD;
        cout<<xn<<endl;
    }
    return 0;
}
