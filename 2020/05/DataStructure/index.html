class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>>ans;
        if(!root)return ans;
        queue<TreeNode*>q;
        q.push(root);
        while(!q.empty()){
            vector<int>tmp;
            int level=q.size();
            while(level--){
                TreeNode *t=q.front();
                q.pop();
                tmp.push_back(t->val);
                if(t->left)q.push(t->left);
                if(t->right)q.push(t->right);
            }
            ans.push_back(tmp);
        }
        return ans;
    }
};

class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if(!root||root==p||root==q)
            return root;
        // 判断两个结点在根的哪一边
        TreeNode *left=lowestCommonAncestor(root->left,p,q);
        TreeNode *right=lowestCommonAncestor(root->right,p,q);
        // 右边为空，代表都在左边，那么LCA在左边
        if(!right)
            return left;
        // 左边为空，代表都在右边，那么LCA都在右边
        if(!left)
            return right;
        // 左右都不为空，代表该结点就是LCA
        return root;
    }
};

class Codec {
public:
    vector<string> split(string data){
        vector<string>ans;
        if(data.empty())return ans;
        string tmp;
        for(auto s:data){
            if(s==','){
                ans.push_back(tmp);
                tmp.clear();
            }else{
                tmp+=s;
            }
        }
        ans.push_back(tmp);
        return ans;
    }

    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        string ans;
        if(!root)return ans;
        queue<TreeNode*>q;
        q.push(root);
        while(!q.empty()){
            TreeNode *t=q.front();
            q.pop();
            if(!ans.empty())ans+=",";
            if(!t){
                ans+="null";
                continue;
            }
            ans+=to_string(t->val);
            q.push(t->left);
            q.push(t->right);
        }
        return ans;
    }

    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        vector<string>l=split(data);
        if(l.empty())return NULL;
        queue<TreeNode*>q;
        TreeNode *t=new TreeNode(stoi(l[0])),*ans=t;
        q.push(t);
        for(int i=1;i<l.size();i++){
            if(l[i]!="null"){
                t=new TreeNode(stoi(l[i]));
                q.push(t);
                if(i%2)q.front()->left=t;
                else q.front()->right=t;
            }
            if(i%2==0)q.pop();
        }
        return ans;
    }
};

  2. 如果目标节点只有一个子节点，我们可以用其子节点作为替换

class Solution {
public:
    TreeNode* deleteNode(TreeNode* &root, int key) {
        if(!root)
            return NULL;
           // 找到了要删除的结点
        else if(key==root->val){
            TreeNode *q=root;
            // 2. 只有左孩子 or 1. 叶子
            if(!root->right){
                root=root->left;
                delete q;
            // 2. 只有右孩子
            }else if(!root->left){
                root=root->right;
                delete q;
            // 3. 左右都有，找中序遍历的直接前驱，即左子树的最右端，进行填充，然后删除这个前驱
            }else{
                TreeNode *s=root->left;
                while(s->right)
                    q=s,s=s->right;
                // 填充
                root->val=s->val;
                // 删除该前驱节点
                if(q!=root)
                    q->right=s->left;
                else
                    q->left=s->left;
                delete s;
            }
        }
        // 找要删除的结点
        else if(key<root->val)
            deleteNode(root->left,key);
        else if(key>root->val)
            deleteNode(root->right,key);
        return root;
    }
};

int Partition(int low,int high){
    // 取队首为枢纽
    int pivot=a[low];
    while(low<high){
        // 从右边找小等于pivot的和low（pivot）交换
        while(low<high&&a[high]>=pivot)
            high--;
        a[low]=a[high];
        // 从左边找大等于pivot，和high（pivot）交换
        while(low<high&&a[low]<=pivot)
            low++;
        a[high]=a[low];
    }
    a[low]=pivot;
    return low;
}
void quick(int low,int high){

    quick(low,pos-1);
    // 枢纽右边排序
    quick(pos+1,high);
}

    int tmp[3];
    // 将a[low,mid]和a[mid+1,high]合并成a[low,high]
    // 将a复制到辅助数组中
    for(int k=low;k<=high;k++)
        tmp[k]=a[k];
    // 比较辅助数组两段中元素，并将较小的复制到a中
    int i=low,j=mid+1,cur=i;
    for(;i<=mid&&j<=high;cur++){
        if(tmp[i]<tmp[j])
            a[cur]=tmp[i++];
        else
            a[cur]=tmp[j++];
    }
    // 若第一个表没有检测完
    while(i<=mid)
        a[cur++]=tmp[i++];
    // 第二个表没有检测完
    while(j<=high)
        a[cur++]=tmp[j++];
}

        while(j<n&&i<m){
            if(j==-1||haystack[i]==needle[j])
                i++,j++;
            else
                j=next[j];
        }
        int pos=i-n;
        return j==n?pos:-1;
    }
