        vector<int>ans;
        stack<TreeNode*>s;   
        TreeNode *t=root;
        while(!s.empty()||t){
            if(t){
                ans.push_back(t->val);

class Solution {
public:
    void dfs(TreeNode* root,int level,vector<vector<int>> &ans){
        if(!root)
	        return;
        if(ans.size()==level)
	        ans.resize(level+1);
        ans[level].push_back(root->val);
        dfs(root->left,level+1,ans);
        dfs(root->right,level+1,ans);
    }
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>>ans;
        if(!root)return ans;
        dfs(root,0,ans);
        return ans;
    }
};

迭代，借用队列

  1. 队列取一个元素
  2. 访问
  3. 左右孩子非空入队
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>>ans;
        if(!root)return ans;
        queue<TreeNode*>q;
        q.push(root);
        while(!q.empty()){
            vector<int>tmp;
            int level=q.size();
            while(level--){
                TreeNode *t=q.front();
                q.pop();
                tmp.push_back(t->val);
                if(t->left)q.push(t->left);
                if(t->right)q.push(t->right);
            }
            ans.push_back(tmp);
        }
        return ans;
    }
};

不唯一的原因：无法确定是左孩子+根还是根+右孩子，即最后剩余两个元素，若按下述方式建树的话，默认为左子树+根

# 应用：二叉树的最近公共祖先LCA

class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if(!root||root==p||root==q)
            return root;
        // 判断两个结点在根的哪一边
        TreeNode *left=lowestCommonAncestor(root->left,p,q);
        TreeNode *right=lowestCommonAncestor(root->right,p,q);
        // 右边为空，代表都在左边，那么LCA在左边
        if(!right)
            return left;
        // 左边为空，代表都在右边，那么LCA都在右边
        if(!left)
            return right;
        // 左右都不为空，代表该结点就是LCA
        return root;
    }
};

# 应用：序列化与反序列化

class Codec {
public:
    vector<string> split(string data){
        vector<string>ans;
        if(data.empty())return ans;
        string tmp;
        for(auto s:data){
            if(s==','){
                ans.push_back(tmp);
                tmp.clear();
            }else{
                tmp+=s;
            }
        }
        ans.push_back(tmp);
        return ans;
    }

    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        string ans;
        if(!root)return ans;
        queue<TreeNode*>q;
        q.push(root);
        while(!q.empty()){
            TreeNode *t=q.front();
            q.pop();
            if(!ans.empty())ans+=",";
            if(!t){
                ans+="null";
                continue;
            }
            ans+=to_string(t->val);
            q.push(t->left);
            q.push(t->right);
        }
        return ans;
    }

    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        vector<string>l=split(data);
        if(l.empty())return NULL;
        queue<TreeNode*>q;
        TreeNode *t=new TreeNode(stoi(l[0])),*ans=t;
        q.push(t);
        for(int i=1;i<l.size();i++){
            if(l[i]!="null"){
                t=new TreeNode(stoi(l[i]));
                q.push(t);
                if(i%2)q.front()->left=t;
                else q.front()->right=t;
            }
            if(i%2==0)q.pop();
        }
        return ans;
    }
};

class BSTIterator {
public:
    TreeNode *now;
    stack<TreeNode*>stk;
    BSTIterator(TreeNode* root) {
        now=NULL;
        while(!stk.empty())stk.pop();
        TreeNode *t=root;
        while(t){
            stk.push(t);
            t=t->left;
        }
    }
    
    /** @return the next smallest number */
    int next() {
        now=stk.top(),stk.pop();
        int ans=now->val;
        TreeNode *t=now->right;
        while(t){
            stk.push(t);
            t=t->left;
        }
        return ans;
    }
    
    /** @return whether we have a next smallest number */
    bool hasNext() {
        return !stk.empty();
    }
};

# 删除

  1. 如果目标节点没有子节点，我们可以直接移除该目标节点
  2. 如果目标节点只有一个子节点，我们可以用其子节点作为替换
  3. 如果目标节点有两个子节点，我们需要用其中序后继节点或者前驱节点来替换，再删除该目标节点
class Solution {
public:
    TreeNode* deleteNode(TreeNode* &root, int key) {
        if(!root)
            return NULL;
           // 找到了要删除的结点
        else if(key==root->val){
            TreeNode *q=root;
            // 2. 只有左孩子 or 1. 叶子
            if(!root->right){
                root=root->left;
                delete q;
            // 2. 只有右孩子
            }else if(!root->left){
                root=root->right;
                delete q;
            // 3. 左右都有，找中序遍历的直接前驱，即左子树的最右端，进行填充，然后删除这个前驱
            }else{
                TreeNode *s=root->left;
                while(s->right)
                    q=s,s=s->right;
                // 填充
                root->val=s->val;
                // 删除该前驱节点
                if(q!=root)
                    q->right=s->left;
                else
                    q->left=s->left;
                delete s;
            }
        }
        // 找要删除的结点
        else if(key<root->val)
            deleteNode(root->left,key);
        else if(key>root->val)
            deleteNode(root->right,key);
        return root;
    }
};

        }else if(val<t->val){

                if(getBF(t->left)==1)
                    rrotate(t);

                    lrotate(t->left);

        t->left=lc->right;
        lc->right=t;
        updateHeight(t);
        updateHeight(lc);
        t=lc;
    }
    void lrotate(AVLTreeNode *&t){
        AVLTreeNode *rc=t->right;
        t->right=rc->left;
        rc->left=t;
        updateHeight(t);
        updateHeight(rc);
        t=rc;

void HeapAdjust(int root,int len){
    // 和左右子树最大的进行交换并调整
    int left=2*root+1,right=2*root+2;
    if(left>len)
        return;
    if(right<=len){
        if(a[left]>a[right]&&a[root]<a[left]){
            swap(a[root],a[left]);
            HeapAdjust(left,len);
        }else if(a[right]>a[left]&&a[root]<a[right]){
            swap(a[root],a[right]);
            HeapAdjust(right,len);
        }
    }else{
        if(a[left]>a[root]){
            swap(a[root],a[left]);
            HeapAdjust(left,len);
        }
    }
}

void MidInsertSort(){

    for(int i=1;i<3;i++){
        guard=a[i];
        // 在前面折半查找插入的位置

        while(low<=high){

            if(a[mid]<guard)
                low=mid+1;
            else
                high=mid-1;
        }
        // 移动元素
        for(int j=i-1;j>=high+1;j--)
            a[j+1]=a[j];
        a[high+1]=guard;
    }
}

int Partition(int low,int high){
    // 取队首为枢纽
    int pivot=a[low];
    while(low<high){
        // 从右边找小于pivot的和low（pivot）交换
        while(low<high&&a[high]>=pivot)
            high--;
        a[low]=a[high];
        // 从左边找大于pivot，和high（pivot）交换
        while(low<high&&a[low]<=pivot)
            low++;
        a[high]=a[low];
    }
    a[low]=pivot;
    return low;
}
void quick(int low,int high){
    if(low>=high)
        return;
    int pos=Partition(low,high);
    // 枢纽左边排序
    quick(low,pos-1);
    // 枢纽右边排序
    quick(pos+1,high);
}
void QuickSort(){
    quick(0,2);
}

        tmp[k]=a[k];
    // 比较辅助数组两段中元素，并将较小的复制到a中
    int i=low,j=mid+1,cur=i;
    for(;i<=mid&&j<=high;cur++){
        if(tmp[i]<tmp[j])
            a[cur]=tmp[i++];
        else
            a[cur]=tmp[j++];
    }
    // 若第一个表没有检测完
    while(i<=mid)
        a[cur++]=tmp[i++];
    // 第二个表没有检测完
    while(j<=high)
        a[cur++]=tmp[j++];
}
void merge(int low,int high){
    if(low>=high)
        return;
    int mid=low+((high-low)>>1);
    // 左排序
    merge(low,mid);
    // 右排序
    merge(mid+1,high);
    // 归并
    MergeVec(low,mid,high);
}
void MergeSort(){
    merge(0,3-1);
}

void ShellSort(){
    int guard,j;
    // 步长变化，初始n/2，每次/2，直至=1
    for(int d=3/2;d>=1;d>>=1){
        // 分组进行直接插入排序
        // 从第二个下标开始进行
        for(int i=d;i<3;i++){
            // 比前面的小，需要插入到前面
            if(a[i]<a[i-d]){
                guard=a[i];
                for(j=i-1;j>=0&&guard<a[j];j-=d)
                    a[j+d]=a[j];
                a[j+d]=guard;
            }
        }
    }
}
