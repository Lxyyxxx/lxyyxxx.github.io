# 层序遍历

递归

class Solution {
public:
    void dfs(TreeNode* root,int level,vector<vector<int>> &ans){
        if(!root)return;
        if(ans.size()==level)ans.resize(level+1);
        ans[level].push_back(root->val);
        dfs(root->left,level+1,ans);
        dfs(root->right,level+1,ans);
    }
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>>ans;
        if(!root)return ans;
        dfs(root,0,ans);
        return ans;
    }
};

迭代，借用队列

class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>>ans;
        if(!root)return ans;
        queue<TreeNode*>q;
        q.push(root);
        while(!q.empty()){
            vector<int>tmp;
            int level=q.size();
            while(level--){
                TreeNode *t=q.front();
                q.pop();
                tmp.push_back(t->val);
                if(t->left)q.push(t->left);
                if(t->right)q.push(t->right);
            }
            ans.push_back(tmp);
        }
        return ans;
    }
};

class Solution {
public:
    TreeNode* build(vector<int> &inorder, vector<int>& postorder,int begin,int end,int root_id){
        if(begin>end)return NULL;
        TreeNode *root=new TreeNode(postorder[root_id]);
        int mid=begin;

        root->left=build(inorder,postorder,begin,mid-1,root_id-end+mid-1);

# 二叉树的最近公共祖先

class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if(!root||root==p||root==q)return root;
        TreeNode *left=lowestCommonAncestor(root->left,p,q);
        TreeNode *right=lowestCommonAncestor(root->right,p,q);
        if(!right)return left;
        if(!left)return right;
        return root;
    }
};

# 序列化与反序列化

class Codec {
public:
    vector<string> split(string data){
        vector<string>ans;
        if(data.empty())return ans;
        string tmp;
        for(auto s:data){
            if(s==','){
                ans.push_back(tmp);
                tmp.clear();
            }else{
                tmp+=s;
            }
        }
        ans.push_back(tmp);
        return ans;
    }

    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        string ans;
        if(!root)return ans;
        queue<TreeNode*>q;
        q.push(root);
        while(!q.empty()){
            TreeNode *t=q.front();
            q.pop();
            if(!ans.empty())ans+=",";
            if(!t){
                ans+="null";
                continue;
            }
            ans+=to_string(t->val);
            q.push(t->left);
            q.push(t->right);
        }
        return ans;
    }

    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        vector<string>l=split(data);
        if(l.empty())return NULL;
        queue<TreeNode*>q;
        TreeNode *t=new TreeNode(stoi(l[0])),*ans=t;
        q.push(t);
        for(int i=1;i<l.size();i++){
            if(l[i]!="null"){
                t=new TreeNode(stoi(l[i]));
                q.push(t);
                if(i%2)q.front()->left=t;
                else q.front()->right=t;
            }
            if(i%2==0)q.pop();
        }
        return ans;
    }
};

class Solution {
public: 
    bool isValidBST(TreeNode* root) {
        stack<TreeNode*>s;
        TreeNode *t=root,*pre=NULL;
        while(!s.empty()||t){
            if(t){
                s.push(t);
                t=t->left;
            }else{
                t=s.top(),s.pop();
                if(pre&&pre->val>=t->val)return false;
                pre=t;
                t=t->right;
            }
        }
        return true;
    }
};

# 迭代器

class BSTIterator {
public:
    TreeNode *now;
    stack<TreeNode*>stk;
    BSTIterator(TreeNode* root) {
        now=NULL;
        while(!stk.empty())stk.pop();
        TreeNode *t=root;
        while(t){
            stk.push(t);
            t=t->left;
        }
    }
    
    /** @return the next smallest number */
    int next() {
        now=stk.top(),stk.pop();
        int ans=now->val;
        TreeNode *t=now->right;
        while(t){
            stk.push(t);
            t=t->left;
        }
        return ans;
    }
    
    /** @return whether we have a next smallest number */
    bool hasNext() {
        return !stk.empty();
    }
};

# 查找

对于每个节点：

  1. 如果目标值等于节点的值，则返回节点
  2. 如果目标值小于节点的值，则继续在左子树中搜索
  3. 如果目标值大于节点的值，则继续在右子树中搜索
class Solution {
public:
    TreeNode* searchBST(TreeNode* root, int val) {
        if(!root||val==root->val)
            return root;
        else if(val<root->val)
            return searchBST(root->left,val);
        else
            return searchBST(root->right,val);
    }
};

# 插入

对于每个节点：

  1. 根据节点值与目标节点值的关系，搜索左子树或右子树
  2. 重复步骤 1 直到到达外部节点
  3. 根据节点的值与目标节点的值的关系，将新节点添加为其左侧或右侧的子节点
class Solution {
public:
    TreeNode* insertIntoBST(TreeNode* root, int val) {
        if(root==NULL)
            return new TreeNode(val);
        else if(val<root->val)
            root->left=insertIntoBST(root->left,val);
        else
            root->right=insertIntoBST(root->right,val);
        return root;
    }
};

# 删除

  1. 如果目标节点没有子节点，我们可以直接移除该目标节点
  2. 如果目标节点只有一个子节点，我们可以用其子节点作为替换
  3. 如果目标节点有两个子节点，我们需要用其中序后继节点或者前驱节点来替换，再删除该目标节点
class Solution {
public:
    TreeNode* deleteNode(TreeNode* &root, int key) {
        if(!root)
            return NULL;
        else if(key==root->val){
            TreeNode *q=root;
            if(!root->right){
                root=root->left;
                delete q;
            }else if(!root->left){
                root=root->right;
                delete q;
            }else{
                TreeNode *s=root->left;
                while(s->right)
                    q=s,s=s->right;
                root->val=s->val;
                if(q!=root)
                    q->right=s->left;
                else
                    q->left=s->left;
                delete s;
            }
        }
        else if(key<root->val)
            deleteNode(root->left,key);
        else if(key>root->val)
            deleteNode(root->right,key);
        return root;
    }
};

struct AVLTreeNode {
    int val,height;
    AVLTreeNode *left,*right;
    AVLTreeNode(int x) : val(x), height(1), left(NULL), right(NULL) {}
};
class AVLTree {
public:
    int getHeight(AVLTreeNode *t){
        if(t==NULL)return 0;
        return t->height;
    }
    int getBF(AVLTreeNode *t){
        return getHeight(t->left)-getHeight(t->right);
    }
    void updateHeight(AVLTreeNode *&t){
        t->height=max(getHeight(t->left),getHeight(t->right))+1;
    }
    void insert(AVLTreeNode *&t,int val){
        if(!t){
            t=new AVLTreeNode(val);
        }else if(val==t->val){
            
        }else if(val<t->val){
            insert(t->left,val);
            updateHeight(t);
            if(getBF(t)==2){
                if(getBF(t->left)==1)
                    rrotate(t);
                else if(getBF(t->left)==-1){
                    lrotate(t->left);
                    rrotate(t);
                }
            }
        }else{
            insert(t->right,val);
            updateHeight(t);
            if(getBF(t)==-2){
                if(getBF(t->right)==-1)
                    lrotate(t);
                else if(getBF(t->right)==1){
                    rrotate(t->right);
                    lrotate(t);
                }
            }
        }
    }
    void rrotate(AVLTreeNode *&t){
        AVLTreeNode *lc=t->left;
        t->left=lc->right;
        lc->right=t;
        updateHeight(t);
        updateHeight(lc);
        t=lc;
    }
    void lrotate(AVLTreeNode *&t){
        AVLTreeNode *rc=t->right;
        t->right=rc->left;
        rc->left=t;
        updateHeight(t);
        updateHeight(rc);
        t=rc;
    }
};
