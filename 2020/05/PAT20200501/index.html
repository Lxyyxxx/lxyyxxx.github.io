# 7-1 Knuth洗牌法 (20分)

初始化数组，模拟交换，输出最后结果

#include<bits/stdc++.h>
using namespace std;
const int N=1010;
int main(){
    int n,a[N],b;
    cin>>n;
    for(int i=1;i<=n;i++)a[i]=i;
    for(int i=1;i<=n;i++){
        cin>>b;
        swap(a[i],a[b]);
    }
    for(int i=1;i<=n;i++){
        if(i!=1)cout<<" ";
        cout<<a[i];
    }
    return 0;
}

# 7-2 三阶幸福数 (20分)

模拟，不断计算各位数字立方和，等于一时退出输出次数，重复出现时退出输出值

#include<bits/stdc++.h>
using namespace std;
const int N=1010;
int main(){
    int n,a,s,t,h;
    bool f;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a;
        unordered_map<int,bool>mp;
        h=0;
        f=true;
        while(a!=1){
            mp[a]=true;
            s=0;
            while(a){
                t=a%10;
                s+=t*t*t;
                a/=10;
            }
            a=s;
            h++;
            if(mp[a]){
                cout<<a<<endl;
                f=false;
                break;
            }
        }
        if(f)
            cout<<h<<endl;
    }
    return 0;
}

# 7-3 垃圾分类 (20分)

建立一个map存垃圾及对应类型id，再建一张表存类型id对应的类型名称，能查到输出类型名称，查不到输出问号
有被卡一次I/O输出

#include<bits/stdc++.h>
using namespace std;
int n,a;
string s,ss[5]={"?","Gan laji","Shi laji","Ke Hui Shou","You Hai laji"};
unordered_map<string,int>mp;
int main(){
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>s>>a;
        mp[s]=a;
    }
    while(cin>>s&&s!="#"){
        if(mp.count(s))cout<<ss[mp[s]]<<'\n';
        else cout<<ss[0]<<'\n';
    }
    return 0;
}
