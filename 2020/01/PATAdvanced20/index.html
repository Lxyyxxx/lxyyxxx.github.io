# 1015 Reversible Primes

传送门
1不是质数（第二个测试点），如果输入的十进制数不是质数的话要输出No（第四个测试点）

#include<bits/stdc++.h>
using namespace std;
bool prime(int x){
    if(x==1)return false;
    for(int i=2;i<=sqrt(x);i++){
        if(x%i==0)return false;
    }
    return true;
}
int main(){
    int n,d,res,b,index;
    while(cin>>n&&n>=0){
        cin>>d;
        if(!prime(n)){
            cout<<"No"<<endl;
            continue;
        }
        char r[15];
        index=0;
        while(n>0){
            r[index++]=n%d+'0';
            n/=d;
        }
        res=0;b=1;
        for(int i=index-1;i>=0;i--){
            res+=(r[i]-'0')*b;
            b*=d;
        }
        if(prime(res))cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
bool fun(string a,string b){
    if(a.length()!=b.length())return false;
    else{
        int len=b.length();
        bool ch[len]={false};
        for(int i=0;i<len;i++){
            for(int j=0;j<len;j++){
                if(a[i]==b[j]&&!ch[j])ch[j]=true;
            }
        }
        for(int i=0;i<len;i++){
            if(!ch[i])return false;
        }
        return true;
    }
}
int main(){
    string s,sn;
    cin>>s;
    int len=s.length(),last=0,now;
    for(int i=len-1;i>=0;i--){
        now=(s[i]-'0')*2;
        now+=last;
        sn.insert(0,1,now%10+'0');
        last=now/10;
    }
    if(last!=0)sn.insert(0,1,last+'0');
    if(fun(sn,s))cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
    cout<<sn<<endl;
    return 0;
}

# 1046 Shortest Distance

传送门
在第三个测试点遇到了运行超时和段错误
比较一段和另一段的大小，小的就是答案。只需要用前缀和保存即可，两个和相减得到一段的距离，总和减这一段距离得到另一段距离，然后比较输出即可

#include <bits/stdc++.h>
using namespace std;
int main() {
    int n,m,a,b;
    cin>>n;
    int dis,d[n+10];
    d[0]=0;
    for(int i=1;i<=n;i++){
        cin>>dis;
        d[i]=d[i-1]+dis;
    }
    cin>>m;
    int s1,s2;
    for(int i=0;i<m;i++){
        cin>>a>>b;
        if(a>b){
            int t=b;b=a;a=t;
        }
        s1=d[b-1]-d[a-1];s2=d[n]-s1;
        cout<<(s1>s2?s2:s1)<<endl;
    }
    return 0;
}

# 1069 The Black Hole of Numbers

#include <bits/stdc++.h>

bool cmp(char a,char b){
    return a>b;
}
int main() {
    string num,ic,dc;
    cin>>num;
    num.insert(0,4-num.length(),'0');
    int pre=0;
    if(num=="0000")printf("0000 - 0000 = 0000\n");
    else if (num=="6174")printf("7641 - 1467 = 6174\n");
    while(num!="6174"&&num!="0000"){
       ic=num;dc=num;
       sort(dc.begin(),dc.end());
       sort(ic.begin(),ic.end(),cmp);
        for(int i=3;i>=0;i--){
            if(ic[i]-dc[i]<pre){
                num[i]=(10-pre+ic[i]-dc[i])+'0';
                pre=1;
            }else{
                num[i]=(ic[i]-dc[i]-pre)+'0';
                pre=0;
            }
        }
       printf("%s - %s = %s\n",ic.c_str(),dc.c_str(),num.c_str());

# 1081 Rational Sum

传送门
如果答案为0，需要输出0（最后一个测试点）

#include <bits/stdc++.h>
using namespace std;
long long gcd(long long a,long long b){
    return b==0?abs(a):gcd(b,a%b);
}
int main() {
    long int n,f,nr,dr;
    long long smnr=0,smdr=1,ir,g,up;
    cin>>n;
    string s;
    for(int i=0;i<n;i++){
        cin>>s;
        f=s.find('/');
        nr=stoi(s.substr(0,f));
        dr=stoi(s.substr(f+1,s.length()));
        g=gcd(smdr,dr);
        smnr=(smnr*dr+smdr*nr)/g;
        smdr*=dr/g;
    }
    ir=smnr/smdr;
    up=smnr-ir*smdr;
    g=gcd(smdr,up);
    if(ir!=0)cout<<ir;
    if(ir!=0&&up!=0)cout<<" ";
    if(up!=0)cout<<up/g<<"/"<<smdr/g;
    if(ir==0&&up==0)cout<<0;
    return 0;
}
