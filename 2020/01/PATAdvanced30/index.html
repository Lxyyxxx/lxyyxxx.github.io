# 1004 Counting Leaves

传送门
一次过
遍历树，计算每一层的叶子结点个数

#include <bits/stdc++.h>
using namespace std;
vector<int>children[100];
int result[100]={0};
int n,m,maxdp=0;
void dfs(int start,int depth){
    if(children[start].empty()){
        result[depth]++;
        maxdp=max(depth,maxdp);
        return;
    }
    for(int j:children[start]){
        dfs(j,depth+1);
    }
}
int main() {
    cin>>n>>m;
    int id,k,child;
    for(int i=0;i<m;i++){
        cin>>id>>k;
        for(int j=0;j<k;j++){
            cin>>child;
            children[id].push_back(child);
        }
    }
    dfs(1,0);
    for(int i=0;i<=maxdp;i++){
        if(i!=0)cout<<" ";
        cout<<result[i];
    }
    return 0;
}

# 1014 Waiting in Line

传送门
如果17点前服务了，但服务结束时间在17点后，则需要输出时间而不是Sorry，满足这个条件的测试点共11分。模拟队列即可

#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,m,k,q,time[1010],id,que[25][15],tail[25]={0},top[25]={0},wait[1010],w=0,fast,fastq,sorry[1010]={false};
    cin>>n>>m>>k>>q;
    for(int i=0;i<k;i++)cin>>time[i];
    for(int i=0;i<n*m;i++){
        w=w%n;
        que[w][tail[w]]=i;
        if(!tail[w])wait[i]=time[i];
        else{
            wait[i]=wait[que[w][tail[w]-1]]+time[i];
            if(wait[que[w][tail[w]-1]]>=540)sorry[i]=true;
        }
        tail[w]++;w++;
    }
    for(int i=n*m;i<k;i++){
        fast=wait[que[0][top[0]]];fastq=0;
        for(int j=1;j<n;j++){
            if(wait[que[j][top[j]]]<fast){
                fast=wait[que[j][top[j]]];
                fastq=j;
            }
        }
        top[fastq]++;
        que[fastq][tail[fastq]]=i;
        wait[i]=wait[que[fastq][tail[fastq]-1]]+time[i];
        if(wait[que[fastq][tail[fastq]-1]]>=540)sorry[i]=true;
        tail[fastq]++;
    }
    while(q--){
        cin>>id;
        if(sorry[id-1])printf("Sorry\n");
        else printf("%02d:%02d\n",wait[id-1]/60+8,wait[id-1]%60);
    }
    return 0;
}

# 1107 Social Clusters

传送门
共同爱好的为一个群体，找出群体的个数及相应的人数，按降序排列
很明显可以用并查集，合并的是同一类的爱好，统计的时候，统计相应爱好的人数，加到父节点上，合并的时候，采用路径压缩，将爱好合并到父节点上，并删除本结点
这样，到最后，统计的map中一对映射就是一个群体，按人数排序后输出即可

#include<bits/stdc++.h>
using namespace std;
const int N=1010;
int father[N];
unordered_map<int,int>mp;
int Find(int x){
    if(x!=father[x])
        father[x]=Find(father[x]);
    return father[x];
}
void Union(int x,int y){
    int fx=Find(x),fy=Find(y);
    if(fx<fy){
        father[fy]=fx;
        mp[fx]+=mp[fy];
        mp.erase(fy);
    }
    else if(fx>fy){
        father[fx]=fy;
        mp[fy]+=mp[fx];
        mp.erase(fx);
    }
}
int main(){
    for(int i=1;i<=1000;i++)
        father[i]=i;
    int n,k,hobby,pre;
    cin>>n;
    for(int i=0;i<n;i++){
        scanf("%d:",&k);
        pre=-1;
        while(k--){
            cin>>hobby;
            if(pre!=-1)
                Union(pre,hobby);
            else
                mp[Find(hobby)]++;
            pre=hobby;
        }
    }
    vector<int>ans;
    for(auto i:mp){
        if(i.second>0)
            ans.push_back(i.second);
    }
    sort(ans.begin(),ans.end());
    cout<<ans.size()<<'\n';
    for(int i=ans.size()-1;i>=0;i--){
        cout<<ans[i];
        if(i!=0)cout<<" ";
    }
    return 0;
}

# 1111 Online Map

传送门
给出两条路，一条是距离最短，一条是最快的
如果距离最短的不唯一，输出最快的；如果最快的不唯一，输出最少结点的（而不是最短路径的）
分别以距离和时间为权，用两次Dijkstra得到带权图的最短路，因为要求路径，所以中途有记录pre数组

#include<bits/stdc++.h>
using namespace std;
const int N=520;
int n,m;
struct Street{
    int next,length,time;
};
vector<Street>g[N];
vector<int>shortest,fastest;
int anslen,anstime;
struct cmpLength{
    bool operator()(const Street &a,const Street &b){
        if(a.length==b.length)
            return a.time>b.time;
        return a.length>b.length;
    }
};
struct cmpTime{
    bool operator()(const Street &a,const Street &b){
        if(a.time==b.time)
            return a.length>b.length;
        return a.time>b.time;
    }
};
void DijkstraLength(int begin,int end){
    priority_queue<Street,vector<Street>,cmpLength>que;
    int d[N],pre[N],t[N];
    fill(d,d+n,1e9);
    fill(t,t+n,1e9);
    d[begin]=t[begin]=0;
    pre[begin]=-1;
    que.push({begin,0,0});
    while(!que.empty()){
        Street cur=que.top();
        que.pop();
        if(cur.next==end){
            anslen=cur.length;
            break;
        }
        int v=cur.next;
        if(d[v]<cur.length)
            continue;
        for(auto i:g[v]){
            if((d[i.next]>d[v]+i.length)||(d[i.next]==d[v]+i.length&&t[i.next]>t[v]+i.time)){
                d[i.next]=d[v]+i.length;
                t[i.next]=t[v]+i.time;
                que.push({i.next,d[i.next],t[i.next]});
                pre[i.next]=v;
            }
        }
    }
    int pos=end;
    while(pos!=-1){
        shortest.insert(shortest.begin(),pos);
        pos=pre[pos];
    }
}
void DijkstraTime(int begin,int end){
    priority_queue<Street,vector<Street>,cmpTime>que;
    int d[N],pre[N],t[N];
    fill(t,t+n,1e9);
    fill(d,d+n,1e9);
    d[begin]=t[begin]=0;
    pre[begin]=-1;
    que.push({begin,0,0});
    while(!que.empty()){
        Street cur=que.top();
        que.pop();
        if(cur.next==end){
            anstime=cur.time;
            break;
        }
        int v=cur.next;
        if(t[v]<cur.time)
            continue;
        for(auto i:g[v]){
            if(t[i.next]>t[v]+i.time){
                t[i.next]=t[v]+i.time;
                d[i.next]=d[v]+1;
                que.push({i.next,d[i.next],t[i.next]});
                pre[i.next]=v;
            }
        }
    }
    int cur=end;
    while(cur!=-1){
        fastest.insert(fastest.begin(),cur);
        cur=pre[cur];
    }
}
void printVec(vector<int> v){
    for(int i=0;i<v.size();i++){
        if(i!=0)cout<<" -> ";
        cout<<v[i];
    }
    cout<<endl;
}
int main(){
    cin>>n>>m;
    bool oneway;
    int start,end,length,time;
    for(int i=0;i<m;i++){
        cin>>start>>end>>oneway>>length>>time;
        g[start].push_back({end,length,time});
        if(!oneway)
            g[end].push_back({start,length,time});
    }
    cin>>start>>end;
    DijkstraLength(start,end);
    DijkstraTime(start,end);
    if(shortest==fastest){
        printf("Distance = %d; Time = %d: ",anslen,anstime);
        printVec(shortest);
    }else{
        printf("Distance = %d: ",anslen);
        printVec(shortest);
        printf("Time = %d: ",anstime);
        printVec(fastest);
    }
    return 0;
}

# 1115 Counting Nodes in a BST

传送门
一次过
求二叉搜索树中最底下两层的结点个数及其总和，题中二叉搜索树的定义中，左子树包括小等于根节点的结点
建立二叉搜索树+层序遍历，时刻记录本层和上一层的结点个数

#include<bits/stdc++.h>
using namespace std;
struct TreeNode{
    int val;
    TreeNode *left,*right;
    TreeNode(int x):val(x),left(NULL),right(NULL){}
};
TreeNode *insert(TreeNode *root,int val){
    if(!root)
        return new TreeNode(val);
    else if(val<=root->val)
        root->left=insert(root->left,val);
    else 
        root->right=insert(root->right,val);
    return root;
}
int main(){
    int n,node;
    TreeNode *root=NULL;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>node;
        root=insert(root,node);
    }
    vector<int>v(2,0);
    queue<TreeNode*>q;
    q.push(root);
    int level;
    TreeNode *cur;
    while(!q.empty()){
        level=q.size();
        if(v.size()==2)
            swap(v[0],v[1]);
        v[1]=level;
        while(level--){
            cur=q.front();
            q.pop();
            if(cur->left)
                q.push(cur->left);
            if(cur->right)
                q.push(cur->right);
        }
    }
    printf("%d + %d = %d",v[1],v[0],v[0]+v[1]);
    return 0;
}

# 1119 Pre- and Post-order Traversals

传送门
46m22s
给定一棵树的前序和后序遍历，输出中序遍历。前序和后序可能不能确定唯一的一棵树，所以需要输出是否能唯一确定一颗二叉树，如果不能唯一确定的话，随便输出一种即可
根据前序和后序遍历建树然后输出即可，问题在于如何建树以及如何判断是唯一的树

  1. 建树
    建树是一个递归过程，先建根节点，再建左子树，最后建右子树。所以我们要找到哪里是根，哪一部分是左子树，哪一部分是右子树
    前序是根左右，后序是左右根，很明显我们可以找到根在前序的开头且在后序的结尾，利用这一性质，我们可以确定左右子树
    先确定右子树，那么除去根和右子树的部分就是左子树。在后序中，右子树的根在当前根的前一个，在前序中找到这个结点，就相当于确定了右子树的开头且且后面都属于右子树部分，那么左右子树根据长度和开始位置就确定完了
  2. 判断唯一
    在一个两层的子树中，分三种情况：111个结点、222个结点、333个结点
    111个结点的很容易知道它是根，且没有左右孩子
    333个结点的同样也很容易判断出是根和左右孩子
    所以以上两种情况是可以唯一确定一颗子树的，因此它可以唯一确定整棵树
    问题在222个结点上，会发现不管是只有左孩子还是只有右孩子，前序和后序的遍历都符合要求，所以是不能确定唯一一颗子树的，因此我们只要判断当前子树的结点数是不是222就可以判断出该二叉树是否唯一

最后输出中序遍历后一定要多输出一个回车换行，不然会报格式错误

#include<bits/stdc++.h>
using namespace std;
const int N=36;
int n,pre[N],post[N];
vector<int>in;
bool uni=true;
struct TreeNode{
    int val;
    TreeNode *left,*right;
    TreeNode(int x):val(x),left(NULL),right(NULL){}
};
TreeNode *build(int pre_left,int pre_right,int post_left,int post_right){
    if(pre_left>pre_right||post_left>post_right)
        return NULL;
    if(post_right-post_left==1)
        uni=false;
    TreeNode *root=new TreeNode(pre[pre_left]);
    if(post_right-1<post_left)
        return root;
    // 在pre中找post_right-1，即右子树的根，分为左右子树
    int mid=pre_left+1;
    while(mid<=pre_right&&pre[mid]!=post[post_right-1])
        mid++;
    root->left=build(pre_left+1,mid-1,post_left,post_left+mid-pre_left-2);
    root->right=build(mid,pre_right,post_left+mid-pre_left-1,post_right-1);
    return root;
}
void intra(TreeNode *root){
    if(!root)
        return;
    intra(root->left);
    in.push_back(root->val);
    intra(root->right);
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>pre[i];
    for(int i=0;i<n;i++)
        cin>>post[i];
    TreeNode *root=build(0,n-1,0,n-1);
    if(uni)
        cout<<"Yes\n";
    else
        cout<<"No\n";
    intra(root);
    for(int i=0;i<in.size();i++){
        if(i!=0)cout<<" ";
        cout<<in[i];
    }
    cout<<'\n';
    return 0;
}

# 1155 Heap Paths

传送门
要注意dfs中堆的遍历结束条件及输出

#include <bits/stdc++.h>
using namespace std;
vector<int>path;
int n,heap[1010],state=-1;
void dfs(int root){
    if(2*root+1>=n&&2*root+2>=n){
        if(root<n){
            for(int i=0;i<path.size();i++){
                if(i==0)cout<<path[i];
                else{
                    cout<<" "<<path[i];
                    if(path[i]>path[i-1]){
                        if(state==-1)state=1;
                        else if(state==2)state=0;
                    }
                    else if(path[i]<path[i-1]){
                        if(state==-1)state=2;
                        else if(state==1)state=0;
                    }
                }
            }
            cout<<'\n';
        }
    }else{
        path.push_back(heap[2*root+2]);
        dfs(2*root+2);
        path.pop_back();
        path.push_back(heap[2*root+1]);
        dfs(2*root+1);
        path.pop_back();
    }
}
int main(){
    string h[3]={"Not Heap","Min Heap","Max Heap"};
    cin>>n;
    for(int i=0;i<n;i++)cin>>heap[i];
    path.push_back(heap[0]);
    dfs(0);
    cout<<h[state];
    return 0;
}
