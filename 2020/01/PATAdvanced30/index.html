# 1018 Public Bike Management

传送门
杭州公共自行车容量分配

  1. 最短路
  2. 从PBMC运出的车最少
  3. 运回PBMC的车最少

带第二标尺的最短路，但是运出运回的车含有负权，所以在Dijkstra时不能直接更新，需要Dijkstra+DFS
send和back计算

  1. 如果c/2−bikei>0c/2-bike_i>0c/2−bikei​>0，则表示车太多了要运回，back+=c/2-bike[i]
  2. 否则，根据back的数量判断是否需要send，若back>bikei−c/2back>bike_i-c/2back>bikei​−c/2，则代表back足够可以填补send，不用从PBMC运车，此时back-=bike[i]-c/2，否则需要从PBMC运车，send=bike[i]-c/2-back,back=0

注意：DFS是从终点往起点回溯，如果期间是push_back，则计算bike是从size-2到0

#include<bits/stdc++.h>
using namespace std;
const int N=510;
typedef pair<int,int> pii;
int c,n,s,m,bike[N];
vector<int>pre[N];
vector<pii>g[N];
void dijkstra() {
    priority_queue<pii,vector<pii>,greater<pii>>q;
    vector<bool>vis(n+1,0);
    vector<int>d(n+1,1e9);
    q.push({0,0});
    d[0]=0;
    while(!q.empty()) {
        pii cur=q.top();
        q.pop();
        if(vis[cur.second])
            continue;
        vis[cur.second]=1;
        for(auto &i:g[cur.second]) {
            if(vis[i.first])
                continue;
            if(d[i.first]>i.second+d[cur.second]) {
                d[i.first]=i.second+d[cur.second];
                pre[i.first].clear();
                pre[i.first].push_back(cur.second);
                q.push({d[i.first],i.first});
            } else if(d[i.first]==i.second+d[cur.second]) {
                pre[i.first].push_back(cur.second);
                q.push({d[i.first],i.first});
            }
        }
    }
}
vector<int>tmppath;
struct node {
    int send;
    vector<int>path;
    int back;
} ans;
void dfs(int now) {
    if(now==0) {
        tmppath.push_back(now);
        int tmpsd=0,tmpbc=0;
        for(int i=tmppath.size()-1; i>=0; i--) {
            int res=bike[tmppath[i]];
            if(res>0) {
                tmpbc+=res;
            } else {
                res=-res;
                if(tmpbc>res)
                    tmpbc-=res;
                else
                    tmpsd+=res-tmpbc,tmpbc=0;
            }
        }
        if((ans.send>tmpsd)||(ans.send==tmpsd&&ans.back>tmpbc))
            ans= {tmpsd,tmppath,tmpbc};
        tmppath.pop_back();
        return;
    }
    tmppath.push_back(now);
    for(int i:pre[now])
        dfs(i);
    tmppath.pop_back();
}
int main() {

    c/=2;

        cin>>s1>>s2>>t;

        g[s2].push_back({s1,t});

按人枚举：

  1. 如果是vip
    1. 如果有vip桌，分配最小的
    2. 如果有普通桌，分配最小的
    3. 等待
  2. 如果不是vip
    1. 如果有桌子，分配最小的
    2. 等待
  3. 等待时间
    1. 同一时间空出的桌子全部记录，然后分配这些桌子
    2. 先将vip桌子分配给vip用户
    3. 将剩余未分配的桌子分配给普通用户

注意：

  1. 如果刚好21点的不能玩
  2. 时间边界上，好像不存在到达时间等于上一桌离开时间的，小等和相等的都能过
  3. 前k个人不一定是分配到k张桌子上，可能前面有人已经使用完了，那么这张桌子是参与分配的，且优先级比后面编号大的的桌子高
#include<bits/stdc++.h>
using namespace std;
const int N=10010,K=110;
const int SERVING=2*3600,END=21*3600;
typedef pair<int,int> pii;
struct Player {
    int arr,ser;
    bool vip,vis;
} player[N];
struct Table {
    bool vip,vis;
    int cnt;
} table[K];
bool cmp(const Player &a,const Player &b) {
    return a.arr<b.arr;
}
void print_time(int x) {
    int h=x/3600;
    int s=x%60;
    int m=x%3600/60;
    printf("%02d:%02d:%02d",h,m,s);
}
void output(int t1,int t2) {
    print_time(t1);
    printf(" ");
    print_time(t2);
    printf(" %d\n",(t2-t1+30)/60);
}
int n,k,m;
set<int>at,av;
bool f;
void solve() {
    priority_queue<pii,vector<pii>,greater<pii>>q;
    int pos=0,curtable;
    f=false;
    for(; pos<n; pos++) {
        if(f)
            break;
        if(player[pos].vis)
            continue;
        pii cur;
        // 在pos来之前已经被清空的桌子
        while(!q.empty()) {
            cur=q.top();
            if(cur.first>player[pos].arr)
                break;
            if(table[cur.second].vip)
                av.insert(cur.second);
            at.insert(cur.second);
            table[cur.second].vis=false;
            q.pop();
        }
        curtable=-1;
        // 队首是vip
        if(player[pos].vip) {
            // 优先分配vip
            if(!av.empty()) {
                curtable=*av.begin();
                av.erase(curtable);
                at.erase(curtable);
                // 没有vip分配普通桌子
            } else if(!at.empty()) {
                curtable=*at.begin();
                at.erase(curtable);
            }
            // 都没有需要等待
        } else {
            // 分配编号最小的
            if(!at.empty()) {
                curtable=*at.begin();
                at.erase(curtable);
                if(table[curtable].vip)
                    av.erase(curtable);
            }
        }
        // 来的时候就有空桌子，已分配
        if(curtable!=-1) {
            output(player[pos].arr,player[pos].arr);
            player[pos].vis=1;
            table[curtable].vis=1;
            table[curtable].cnt++;
            q.push({player[pos].arr+player[pos].ser,curtable});
        } else {
            cur=q.top();
            // 同一时间空出的桌子
            int curtime=cur.first;
            // 关门不让玩
            if(curtime>=END) {
                f=true;
                break;
            }
            do {
                q.pop();
                if(table[cur.second].vip)
                    av.insert(cur.second);
                at.insert(cur.second);
                table[cur.second].vis=false;
                cur=q.top();
            } while(curtime==cur.first);
            // 先分配vip桌子
            int v;
            for(v=pos; v<n; v++) {
                if(av.empty())
                    break;
                if(player[v].vip&&!player[v].vis) {
                    curtable=*av.begin();
                    output(player[v].arr,curtime);
                    player[v].vis=1;
                    table[curtable].vis=1;
                    table[curtable].cnt++;
                    q.push({curtime+player[v].ser,curtable});
                    av.erase(curtable);
                    at.erase(curtable);
                }
            }
            // 当前是vip，上面被分配过，不必重复分配
            if(player[pos].vis)
                continue;
            // 在桌子释放的时间内分配可分配的人
            for(auto &i:at) {
                while(pos<n&&player[pos].vis)
                    pos++;
                if(player[pos].arr>curtime)
                    break;
                // 分配编号最小的
                at.erase(i);
                if(table[i].vip)
                    av.erase(i);
                output(player[pos].arr,curtime);
                player[pos].vis=1;
                table[i].vis=1;
                table[i].cnt++;
                q.push({curtime+player[pos].ser,i});
            }
        }
        if(!player[pos].vis)
            pos--;
    }
    for(int i=1; i<=k; i++) {
        if(i!=1)
            cout<<" ";
        cout<<table[i].cnt;
    }
}
int main() {
    cin>>n;
    int hh,mm,ss,p,tag,tt,vt;
    for(int i=0; i<n; i++) {
        scanf("%d:%d:%d %d %d",&hh,&mm,&ss,&p,&tag);
        tt=hh*3600+mm*60+ss;

            player[i]= {tt,min(SERVING,p*60),tag==1,false};
        else
            player[i]= {tt,min(SERVING,p*60),tag==1,1};
    }
    sort(player,player+n,cmp);
    cin>>k>>m;
    for(int i=0; i<m; i++) {
        cin>>vt;
        table[vt].vip=1;
    }
    for(int i=1; i<=k; i++) {
        if(table[i].vip)
            av.insert(i);
        at.insert(i);
    }
    solve();
    return 0;
}

# 1030 Travel Plan

传送门
一次过
有第二标尺要求的最短路
Dijkstra+DFS，由Dijkstra求最短路并保存所有路径，再通过DFS根据代价得到最小的解

#include<bits/stdc++.h>
using namespace std;
const int N=510;
typedef pair<int,int> pii;
int c[N][N];
vector<pii>g[N];
vector<int>ans;
vector<vector<int>>path;
int sdis=1e9,scost=1e9;
void dijkstra(int s,int e) {
    priority_queue<pii,vector<pii>,greater<pii>>q;
    vector<int>d(N,1e9),pre(N,-1);
    q.push({0,s});
    d[s]=0;
    while(!q.empty()) {
        pii cur=q.top();
        q.pop();
        if(cur.first>d[cur.second])
            continue;
        for(auto i:g[cur.second]) {
            if(cur.first+i.second<d[i.first]) {
                d[i.first]=cur.first+i.second;
                pre[i.first]=cur.second;
                if(i.first==e) {
                    path.clear();
                    vector<int>tmp;
                    int pos=i.first;
                    while(pos!=-1) {
                        tmp.insert(tmp.begin(),pos);
                        pos=pre[pos];
                    }
                    path.push_back(tmp);
                    sdis=min(sdis,d[e]);
                } else
                    q.push({d[i.first],i.first});
            } else if(cur.first+i.second==d[i.first]) {
                pre[i.first]=cur.second;
                if(i.first==e) {
                    vector<int>tmp;
                    int pos=i.first;
                    while(pos!=-1) {
                        tmp.insert(tmp.begin(),pos);
                        pos=pre[pos];
                    }
                    path.push_back(tmp);
                    sdis=min(sdis,d[e]);
                } else
                    q.push({d[i.first],i.first});
            }
        }
    }
}
void dfs() {
    int s,cnt;
    for(auto p:path) {
        s=p[0];
        cnt=0;
        for(int i=1; i<p.size(); i++) {
            cnt+=c[s][p[i]];
            s=p[i];
        }
        if(cnt<scost) {
            scost=cnt;
            ans=p;
        }
    }
}
int main() {
    int n,m,s,d,c1,c2,dis,cost;
    cin>>n>>m>>s>>d;
    for(int i=0; i<m; i++) {
        cin>>c1>>c2>>dis>>cost;
        g[c1].push_back({c2,dis});
        g[c2].push_back({c1,dis});
        c[c1][c2]=c[c2][c1]=cost;
    }
    dijkstra(s,d);
    dfs();
    for(int i:ans)
        cout<<i<<" ";
    cout<<sdis<<" "<<scost;
    return 0;
}

# 1034 Head of a Gang

传送门
一次过
求一个集合的头，满足

  1. 该集合人数大于2人
  2. 该集合的所有weight相加大于k
  3. head是weight最大的人

BFS遍历无向图的连通分量，每个连通分量即为一个集合，求每个集合的head，按head的字母序升序输出

#include<bits/stdc++.h>
using namespace std;
typedef pair<string,int> psi;
unordered_map<string,vector<psi>>g;
unordered_set<string>st;
unordered_map<string,bool>vis;
set<psi>ans;
int n,k;
void bfs(string s){
    queue<string>q;
    q.push(s);
    unordered_map<string,int>w;
    int sum=0;
    while(!q.empty()){
        string cur=q.front();
        q.pop();
        if(vis[cur])
            continue;
        vis[cur]=true;
        for(auto &i:g[cur]){
            w[cur]+=i.second;
            if(!vis[i.first])
                q.push(i.first);
            sum+=i.second;
        }
    }
    const int SZ=w.size();
    if(sum<=k*2||SZ<=2)
        return;
    int mwt=0;
    string head;
    for(auto &i:w){
        if(i.second>mwt){
            mwt=i.second;
            head=i.first;
        }
    }
    ans.insert({head,SZ});
}
int main(){
    cin>>n>>k;
    string n1,n2;
    int time;
    for(int i=0;i<n;i++){
        cin>>n1>>n2>>time;
        g[n1].push_back({n2,time});
        g[n2].push_back({n1,time});
        st.insert(n1);
        st.insert(n2);
        vis[n1]=false;
        vis[n2]=false;
    }
    for(auto &i:st){
        if(!vis[i])
            bfs(i);
    }
    cout<<ans.size()<<'\n';
    for(auto &i:ans)
        cout<<i.first<<" "<<i.second<<'\n';
    return 0;
}

# 1038 Recover the Smallest Number

找拼接起来最小的数
贪心

# 1045 Favorite Color Stripe

传送门
DP，求按Eva喜欢顺序的最长子序列长度
dp[i][j]表示eva[i]和stripe[j]之前产生的最长长度

# 1103 Integer Factorization

传送门
DFS+剪枝，与TLE斗智斗勇
枚举可能的因子，在符合条件时保留最优的解
剪枝为了避免重复计算，浪费不必要的时间
当解为空时，代表没有合法的因子

#include<bits/stdc++.h>
using namespace std;
vector<int>v,fac,ans;
int mxfac=0,n,k,p;
void dfs(int id,int sum,int cnt,int facsum){
    if(cnt>k||cnt==k&&sum!=n)
        return;
    if(sum+k-cnt==n){
        if(facsum+k-cnt>mxfac){
            mxfac=facsum+k-cnt;
            ans=fac;
            for(int i=0;i<k-cnt;i++)
                ans.push_back(1);
        }
        return;
    }
    if(sum==n&&cnt==k){
        if(mxfac<facsum){
            mxfac=facsum;
            ans=fac;
        }
        return;
    }
    for(int i=id;i>=2;i--){
        if(sum+v[i]>n)
            continue;
        fac.push_back(i);
        dfs(i,sum+v[i],cnt+1,facsum+i);
        fac.pop_back();
    }
}

    ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    cin>>n>>k>>p;
    for(int i=0;;i++){
        v.push_back(pow(i,p));
        if(v.back()>=n-k+1)
            break;
    }
    dfs(v.size()-1,0,0,0);
    if(ans.empty())
        cout<<"Impossible\n";
    else{
        sort(ans.begin(),ans.end(),greater<>());
        cout<<n<<" = ";
        bool print=false;
        for(int i:ans){
            if(print)
                cout<<" + ";
            cout<<i<<"^"<<p;

        }
    }
    return 0;
}

# 1107 Social Clusters

传送门
共同爱好的为一个群体，找出群体的个数及相应的人数，按降序排列
很明显可以用并查集，合并的是同一类的爱好，统计的时候，统计相应爱好的人数，加到父节点上，合并的时候，采用路径压缩，将爱好合并到父节点上，并删除本结点
这样，到最后，统计的map中一对映射就是一个群体，按人数排序后输出即可

#include<bits/stdc++.h>
using namespace std;
const int N=1010;
int father[N];
unordered_map<int,int>mp;
int Find(int x){
    if(x!=father[x])
        father[x]=Find(father[x]);
    return father[x];
}
void Union(int x,int y){
    int fx=Find(x),fy=Find(y);
    if(fx<fy){
        father[fy]=fx;
        mp[fx]+=mp[fy];
        mp.erase(fy);
    }
    else if(fx>fy){
        father[fx]=fy;
        mp[fy]+=mp[fx];
        mp.erase(fx);
    }
}
int main(){
    for(int i=1;i<=1000;i++)
        father[i]=i;
    int n,k,hobby,pre;
    cin>>n;
    for(int i=0;i<n;i++){
        scanf("%d:",&k);
        pre=-1;
        while(k--){
            cin>>hobby;
            if(pre!=-1)
                Union(pre,hobby);
            else
                mp[Find(hobby)]++;
            pre=hobby;
        }
    }
    vector<int>ans;
    for(auto i:mp){
        if(i.second>0)
            ans.push_back(i.second);
    }
    sort(ans.begin(),ans.end());
    cout<<ans.size()<<'\n';
    for(int i=ans.size()-1;i>=0;i--){
        cout<<ans[i];
        if(i!=0)cout<<" ";
    }

}

# 1111 Online Map

传送门
给出两条路，一条是距离最短，一条是最快的
如果距离最短的不唯一，输出最快的；如果最快的不唯一，输出最少结点的（而不是最短路径的）
分别以距离和时间为权，用两次Dijkstra得到带权图的最短路，因为要求路径，所以中途有记录pre数组

#include<bits/stdc++.h>
using namespace std;
const int N=520;
int n,m;
struct Street{
    int next,length,time;
};
vector<Street>g[N];
vector<int>shortest,fastest;
int anslen,anstime;
struct cmpLength{
    bool operator()(const Street &a,const Street &b){
        if(a.length==b.length)
            return a.time>b.time;
        return a.length>b.length;
    }
};
struct cmpTime{
    bool operator()(const Street &a,const Street &b){
        if(a.time==b.time)
            return a.length>b.length;
        return a.time>b.time;
    }
};
void DijkstraLength(int begin,int end){
    priority_queue<Street,vector<Street>,cmpLength>que;
    int d[N],pre[N],t[N];
    fill(d,d+n,1e9);
    fill(t,t+n,1e9);
    d[begin]=t[begin]=0;
    pre[begin]=-1;
    que.push({begin,0,0});
    while(!que.empty()){
        Street cur=que.top();
        que.pop();
        if(cur.next==end){
            anslen=cur.length;
            break;
        }
        int v=cur.next;
        if(d[v]<cur.length)
            continue;
        for(auto i:g[v]){
            if((d[i.next]>d[v]+i.length)||(d[i.next]==d[v]+i.length&&t[i.next]>t[v]+i.time)){
                d[i.next]=d[v]+i.length;
                t[i.next]=t[v]+i.time;
                que.push({i.next,d[i.next],t[i.next]});
                pre[i.next]=v;
            }
        }
    }
    int pos=end;
    while(pos!=-1){
        shortest.insert(shortest.begin(),pos);
        pos=pre[pos];
    }
}
void DijkstraTime(int begin,int end){
    priority_queue<Street,vector<Street>,cmpTime>que;
    int d[N],pre[N],t[N];
    fill(t,t+n,1e9);
    fill(d,d+n,1e9);
    d[begin]=t[begin]=0;
    pre[begin]=-1;
    que.push({begin,0,0});
    while(!que.empty()){
        Street cur=que.top();
        que.pop();
        if(cur.next==end){
            anstime=cur.time;
            break;
        }
        int v=cur.next;
        if(t[v]<cur.time)
            continue;
        for(auto i:g[v]){
            if(t[i.next]>t[v]+i.time){
                t[i.next]=t[v]+i.time;
                d[i.next]=d[v]+1;
                que.push({i.next,d[i.next],t[i.next]});
                pre[i.next]=v;
            }
        }
    }
    int cur=end;
    while(cur!=-1){
        fastest.insert(fastest.begin(),cur);
        cur=pre[cur];
    }
}
void printVec(vector<int> v){
    for(int i=0;i<v.size();i++){
        if(i!=0)cout<<" -> ";
        cout<<v[i];
    }
    cout<<endl;
}
int main(){
    cin>>n>>m;
    bool oneway;
    int start,end,length,time;
    for(int i=0;i<m;i++){
        cin>>start>>end>>oneway>>length>>time;
        g[start].push_back({end,length,time});
        if(!oneway)
            g[end].push_back({start,length,time});
    }
    cin>>start>>end;
    DijkstraLength(start,end);
    DijkstraTime(start,end);
    if(shortest==fastest){
        printf("Distance = %d; Time = %d: ",anslen,anstime);
        printVec(shortest);
    }else{
        printf("Distance = %d: ",anslen);
        printVec(shortest);
        printf("Time = %d: ",anstime);
        printVec(fastest);
    }
    return 0;
}

# 1119 Pre- and Post-order Traversals

传送门
46m22s
给定一棵树的前序和后序遍历，输出中序遍历。前序和后序可能不能确定唯一的一棵树，所以需要输出是否能唯一确定一颗二叉树，如果不能唯一确定的话，随便输出一种即可
根据前序和后序遍历建树然后输出即可，问题在于如何建树以及如何判断是唯一的树

  1. 建树
    建树是一个递归过程，先建根节点，再建左子树，最后建右子树。所以我们要找到哪里是根，哪一部分是左子树，哪一部分是右子树
    前序是根左右，后序是左右根，很明显我们可以找到根在前序的开头且在后序的结尾，利用这一性质，我们可以确定左右子树
    先确定右子树，那么除去根和右子树的部分就是左子树。在后序中，右子树的根在当前根的前一个，在前序中找到这个结点，就相当于确定了右子树的开头且且后面都属于右子树部分，那么左右子树根据长度和开始位置就确定完了
  2. 判断唯一
    在一个两层的子树中，分三种情况：111个结点、222个结点、333个结点
    111个结点的很容易知道它是根，且没有左右孩子
    333个结点的同样也很容易判断出是根和左右孩子
    所以以上两种情况是可以唯一确定一颗子树的，因此它可以唯一确定整棵树
    问题在222个结点上，会发现不管是只有左孩子还是只有右孩子，前序和后序的遍历都符合要求，所以是不能确定唯一一颗子树的，因此我们只要判断当前子树的结点数是不是222就可以判断出该二叉树是否唯一

最后输出中序遍历后一定要多输出一个回车换行，不然会报格式错误

#include<bits/stdc++.h>
using namespace std;
const int N=36;
int n,pre[N],post[N];
vector<int>in;
bool uni=true;
struct TreeNode{
    int val;
    TreeNode *left,*right;
    TreeNode(int x):val(x),left(NULL),right(NULL){}
};
TreeNode *build(int pre_left,int pre_right,int post_left,int post_right){
    if(pre_left>pre_right||post_left>post_right)
        return NULL;
    if(post_right-post_left==1)
        uni=false;
    TreeNode *root=new TreeNode(pre[pre_left]);
    if(post_right-1<post_left)
        return root;
    // 在pre中找post_right-1，即右子树的根，分为左右子树
    int mid=pre_left+1;
    while(mid<=pre_right&&pre[mid]!=post[post_right-1])
        mid++;
    root->left=build(pre_left+1,mid-1,post_left,post_left+mid-pre_left-2);
    root->right=build(mid,pre_right,post_left+mid-pre_left-1,post_right-1);
    return root;
}
void intra(TreeNode *root){
    if(!root)
        return;
    intra(root->left);
    in.push_back(root->val);
    intra(root->right);
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>pre[i];
    for(int i=0;i<n;i++)
        cin>>post[i];
    TreeNode *root=build(0,n-1,0,n-1);
    if(uni)
        cout<<"Yes\n";
    else
        cout<<"No\n";

        if(i!=0)cout<<" ";

    }

#include<bits/stdc++.h>
using namespace std;
struct TNode{
    int val,height;
    TNode *left,*right;
    TNode(int x):val(x),height(1),left(NULL),right(NULL){}
};
typedef TNode* Tree;
int getHeight(Tree T){
    if(!T)
        return 0;
    return T->height;
}
int getBF(Tree T){
    return getHeight(T->left)-getHeight(T->right);
}
void updateHeight(Tree &T){
    T->height=max(getHeight(T->left),getHeight(T->right))+1;
}
void r(Tree &T){
    Tree tmp=T->left;
    T->left=tmp->right;
    tmp->right=T;
    updateHeight(T);
    updateHeight(tmp);
    T=tmp;
}
void l(Tree &T){
    Tree tmp=T->right;
    T->right=tmp->left;
    tmp->left=T;
    updateHeight(T);
    updateHeight(tmp);
    T=tmp;
}

#include<bits/stdc++.h>

struct TreeNode{

Tree build(Tree T,int root,int l,int r){
    if(l>r)
        return NULL;

    while(mid<=r&&in[mid]!=post[root])
        mid++;
    T=new TreeNode(post[root]);
    T->left=build(T->left,root-r+mid-1,l,mid-1);
    T->right=build(T->right,root-1,mid+1,r);
    return T;
}

#include<bits/stdc++.h>

const int N=1e5+7;
typedef pair<int,int> pii;
vector<pii>g[N],path;
bool vis[N];
int min_transfer,min_path;
void dfs(int u,int v,vector<pii> tmp){
    if(u==v){
        int tmp_transfer=0,tmp_path=tmp.size();
        for(int i=1;i<tmp.size();i++){
            if(tmp[i-1].second!=tmp[i].second)
                tmp_transfer++;
        }
        if((tmp_path<min_path)||(tmp_transfer<min_transfer&&tmp_path<=min_path)){
            min_transfer=tmp_transfer;
            min_path=tmp_path;
            path=tmp;
        }
        return;
    }
    for(auto i:g[u]){
        if(!vis[i.first]){
            vis[i.first]=true;
            tmp.push_back(i);
            dfs(i.first,v,tmp);
            tmp.pop_back();
            vis[i.first]=false;
        }
    }
}

# 1135 Is It A Red-Black Tree

传送门
判断是否是红黑树

  1. 根节点必须是黑
  2. 红的孩子必须是黑
  3. 任意节点到叶子所含的黑个数相同

步骤是

  1. 根据先序序列建树
  2. 根据以上条件判断并输出
#include<bits/stdc++.h>
using namespace std;
const int N=36;
struct TNode{
    int val;
    bool black;
    TNode *left,*right;
    TNode(int v,bool b):val(v),black(b),left(nullptr),right(nullptr){}
};
typedef TNode* BRTree;
typedef pair<BRTree,int> pti;
BRTree build(BRTree T,int X){
    if(!T)
        T=new TNode(abs(X),X>0);
    else if(abs(X)<T->val)
        T->left=build(T->left,X);
    else if(abs(X)>T->val)
        T->right=build(T->right,X);
    return T;
}
int getBlack(BRTree T){
    if(!T)
        return 1;
    int left=getBlack(T->left);
    int right=getBlack(T->right);
    return T->black?max(left,right)+1:max(left,right);
}
bool isbrt(BRTree T){
    if(!T->black)
        return false;
    queue<BRTree>q;
    q.push(T);
    while(!q.empty()){
        BRTree cur=q.front();
        q.pop();
        if(cur->left){
            if(!cur->black&&!cur->left->black)
                return false;
            q.push(cur->left);
        }
        if(cur->right){
            if(!cur->black&&!cur->right->black)
                return false;
            q.push(cur->right);
        }
        if(getBlack(cur->left)!=getBlack(cur->right))
            return false;
    }
    return true;
}
int main(){
    int n,k,v;
    cin>>k;
    BRTree T;
    while(k--){
        cin>>n;
        T=nullptr;
        for(int i=0;i<n;i++){
            cin>>v;
            T=build(T,v);
        }
        if(isbrt(T))
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}

# 1151 LCA in a Binary Tree

传送门
最近公共祖先
与1143 Lowest Common Ancestor类似，但是此题不是BST

  1. 根据前序和中序建树
  2. 递归找LCA：如果两个结点分别在左右孩子，则该节点为LCA；全在右孩子，则在右孩子找LCA；全在左孩子，则在左孩子找LCA
#include<bits/stdc++.h>
using namespace std;
const int N=10010;
int pre[N],in[N],ans;
struct TNode{
    int val;
    TNode *left,*right;
    TNode(int v):val(v),left(nullptr),right(nullptr){}
};
typedef TNode* Tree;
Tree build(Tree T,int root,int l,int r){
    if(l>r)
        return nullptr;
    int mid=l;
    while(mid<=r&&in[mid]!=pre[root])
        mid++;
    T=new TNode(pre[root]);
    T->left=build(T->left,root+1,l,mid-1);
    T->right=build(T->right,root+mid-l+1,mid+1,r);
    return T;
}
Tree lca(Tree T,int u,int v){
    if(!T||T->val==u||T->val==v)
        return T;
    Tree l=lca(T->left,u,v);
    Tree r=lca(T->right,u,v);
    if(!l)
        return r;
    if(!r)
        return l;
    return T;
}

            cout<<"ERROR: "<<u<<" is not found.\n";
