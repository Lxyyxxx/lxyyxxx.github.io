#include <bits/stdc++.h>
using namespace std;
vector<int>children[100];
int result[100]={0};
int n,m,maxdp=0;
void dfs(int start,int depth){
    if(children[start].empty()){
        result[depth]++;
        maxdp=max(depth,maxdp);
        return;
    }
    for(int j:children[start]){
        dfs(j,depth+1);
    }
}
int main() {
    cin>>n>>m;
    int id,k,child;
    for(int i=0;i<m;i++){
        cin>>id>>k;
        for(int j=0;j<k;j++){
            cin>>child;
            children[id].push_back(child);
        }
    }
    dfs(1,0);
    for(int i=0;i<=maxdp;i++){
        if(i!=0)cout<<" ";
        cout<<result[i];
    }
    return 0;
}

}

# 1064 Complete Binary Search Tree

传送门
一次过
求给定序列的完全二叉查找树的层序遍历
二叉查找树的中序遍历一定是升序的，所以先排序即可得到二叉查找树的中序遍历序列，通过中序序列，找到根节点，建左右子树这样递归建树即可
如何确定根节点？这需要考虑完全二叉树
设总结点数为NNN，完全二叉树由一个高度为HHH的满二叉树和XXX个剩余元素组成，那么满足式子2H−1+X=N2^H-1+X=N2H−1+X=N，可以解得H=log⁡2(N+1−X)H=\log_2(N+1-X)H=log2​(N+1−X)，即H⩽log⁡2(N+1)H \leqslant \log_2(N+1)H⩽log2​(N+1)
对于剩下的XXX个元素，我们考虑其中属于左子树的元素个数resresres，这样左子树的总个数就能确定了，即根节点就能确定
剩下的XXX元素在H+1H+1H+1层，若该层满，则有2H2^H2H个元素，此时左边有2H−12^{H-1}2H−1个元素，因此res=min(X,2H−1)res=min(X,2^{H-1})res=min(X,2H−1)。此时，左子树的元素个数为H−1H-1H−1层满二叉树元素个数加上剩余元素个数resresres，即lcnt=2H−1−1+reslcnt=2^{H-1}-1+reslcnt=2H−1−1+res，这样整个区间就确定完毕，可以开始建树，建完树进行层序遍历输出即可

#include<bits/stdc++.h>
using namespace std;
const int N=1010;
struct TreeNode{
    int val;
    TreeNode *left,*right;
    TreeNode(int x):val(x),left(NULL),right(NULL){}
};
typedef TreeNode* Tree;
Tree build(Tree T,int a[],int l,int r){
    if(l>r)
        return NULL;
    int cnt=r-l+1;
    int h=log2(cnt+1);
    int res=cnt-(1<<h)+1;
    int mx=(1<<(h-1));
    res=min(res,mx);
    int lcnt=mx-1+res;
    int mid=l+lcnt;
    T=new TreeNode(a[mid]);
    T->left=build(T->left,a,l,mid-1);
    T->right=build(T->right,a,mid+1,r);
    return T;
}

            que.push(tmp->left);
        if(tmp->right)
            que.push(tmp->right);
    }
}

# 1107 Social Clusters

传送门
共同爱好的为一个群体，找出群体的个数及相应的人数，按降序排列
很明显可以用并查集，合并的是同一类的爱好，统计的时候，统计相应爱好的人数，加到父节点上，合并的时候，采用路径压缩，将爱好合并到父节点上，并删除本结点
这样，到最后，统计的map中一对映射就是一个群体，按人数排序后输出即可

#include<bits/stdc++.h>
using namespace std;
const int N=1010;
int father[N];
unordered_map<int,int>mp;
int Find(int x){
    if(x!=father[x])
        father[x]=Find(father[x]);
    return father[x];
}
void Union(int x,int y){
    int fx=Find(x),fy=Find(y);
    if(fx<fy){
        father[fy]=fx;
        mp[fx]+=mp[fy];
        mp.erase(fy);
    }
    else if(fx>fy){
        father[fx]=fy;
        mp[fy]+=mp[fx];
        mp.erase(fx);
    }
}
int main(){
    for(int i=1;i<=1000;i++)
        father[i]=i;
    int n,k,hobby,pre;
    cin>>n;
    for(int i=0;i<n;i++){
        scanf("%d:",&k);
        pre=-1;
        while(k--){
            cin>>hobby;
            if(pre!=-1)
                Union(pre,hobby);
            else
                mp[Find(hobby)]++;
            pre=hobby;
        }
    }
    vector<int>ans;
    for(auto i:mp){
        if(i.second>0)
            ans.push_back(i.second);
    }
    sort(ans.begin(),ans.end());
    cout<<ans.size()<<'\n';
    for(int i=ans.size()-1;i>=0;i--){
        cout<<ans[i];
        if(i!=0)cout<<" ";
    }
    return 0;
}

# 1111 Online Map

传送门
给出两条路，一条是距离最短，一条是最快的
如果距离最短的不唯一，输出最快的；如果最快的不唯一，输出最少结点的（而不是最短路径的）
分别以距离和时间为权，用两次Dijkstra得到带权图的最短路，因为要求路径，所以中途有记录pre数组

#include<bits/stdc++.h>
using namespace std;
const int N=520;
int n,m;
struct Street{
    int next,length,time;
};
vector<Street>g[N];
vector<int>shortest,fastest;
int anslen,anstime;
struct cmpLength{
    bool operator()(const Street &a,const Street &b){
        if(a.length==b.length)
            return a.time>b.time;
        return a.length>b.length;
    }
};
struct cmpTime{
    bool operator()(const Street &a,const Street &b){
        if(a.time==b.time)
            return a.length>b.length;
        return a.time>b.time;
    }
};
void DijkstraLength(int begin,int end){
    priority_queue<Street,vector<Street>,cmpLength>que;
    int d[N],pre[N],t[N];
    fill(d,d+n,1e9);
    fill(t,t+n,1e9);
    d[begin]=t[begin]=0;
    pre[begin]=-1;
    que.push({begin,0,0});
    while(!que.empty()){
        Street cur=que.top();
        que.pop();
        if(cur.next==end){
            anslen=cur.length;
            break;
        }
        int v=cur.next;
        if(d[v]<cur.length)
            continue;
        for(auto i:g[v]){
            if((d[i.next]>d[v]+i.length)||(d[i.next]==d[v]+i.length&&t[i.next]>t[v]+i.time)){
                d[i.next]=d[v]+i.length;
                t[i.next]=t[v]+i.time;
                que.push({i.next,d[i.next],t[i.next]});
                pre[i.next]=v;
            }
        }
    }
    int pos=end;
    while(pos!=-1){
        shortest.insert(shortest.begin(),pos);
        pos=pre[pos];
    }
}
void DijkstraTime(int begin,int end){
    priority_queue<Street,vector<Street>,cmpTime>que;
    int d[N],pre[N],t[N];
    fill(t,t+n,1e9);
    fill(d,d+n,1e9);
    d[begin]=t[begin]=0;
    pre[begin]=-1;
    que.push({begin,0,0});
    while(!que.empty()){
        Street cur=que.top();
        que.pop();
        if(cur.next==end){
            anstime=cur.time;
            break;
        }
        int v=cur.next;
        if(t[v]<cur.time)
            continue;
        for(auto i:g[v]){
            if(t[i.next]>t[v]+i.time){
                t[i.next]=t[v]+i.time;
                d[i.next]=d[v]+1;
                que.push({i.next,d[i.next],t[i.next]});
                pre[i.next]=v;
            }
        }
    }
    int cur=end;
    while(cur!=-1){
        fastest.insert(fastest.begin(),cur);
        cur=pre[cur];
    }
}
void printVec(vector<int> v){
    for(int i=0;i<v.size();i++){
        if(i!=0)cout<<" -> ";
        cout<<v[i];
    }
    cout<<endl;
}
int main(){
    cin>>n>>m;
    bool oneway;
    int start,end,length,time;
    for(int i=0;i<m;i++){
        cin>>start>>end>>oneway>>length>>time;
        g[start].push_back({end,length,time});
        if(!oneway)
            g[end].push_back({start,length,time});
    }
    cin>>start>>end;
    DijkstraLength(start,end);
    DijkstraTime(start,end);
    if(shortest==fastest){
        printf("Distance = %d; Time = %d: ",anslen,anstime);
        printVec(shortest);
    }else{
        printf("Distance = %d: ",anslen);
        printVec(shortest);
        printf("Time = %d: ",anstime);
        printVec(fastest);
    }
    return 0;
}

# 1119 Pre- and Post-order Traversals

传送门
46m22s
给定一棵树的前序和后序遍历，输出中序遍历。前序和后序可能不能确定唯一的一棵树，所以需要输出是否能唯一确定一颗二叉树，如果不能唯一确定的话，随便输出一种即可
根据前序和后序遍历建树然后输出即可，问题在于如何建树以及如何判断是唯一的树

  1. 建树
    建树是一个递归过程，先建根节点，再建左子树，最后建右子树。所以我们要找到哪里是根，哪一部分是左子树，哪一部分是右子树
    前序是根左右，后序是左右根，很明显我们可以找到根在前序的开头且在后序的结尾，利用这一性质，我们可以确定左右子树
    先确定右子树，那么除去根和右子树的部分就是左子树。在后序中，右子树的根在当前根的前一个，在前序中找到这个结点，就相当于确定了右子树的开头且且后面都属于右子树部分，那么左右子树根据长度和开始位置就确定完了
  2. 判断唯一
    在一个两层的子树中，分三种情况：111个结点、222个结点、333个结点
    111个结点的很容易知道它是根，且没有左右孩子
    333个结点的同样也很容易判断出是根和左右孩子
    所以以上两种情况是可以唯一确定一颗子树的，因此它可以唯一确定整棵树
    问题在222个结点上，会发现不管是只有左孩子还是只有右孩子，前序和后序的遍历都符合要求，所以是不能确定唯一一颗子树的，因此我们只要判断当前子树的结点数是不是222就可以判断出该二叉树是否唯一

最后输出中序遍历后一定要多输出一个回车换行，不然会报格式错误

#include<bits/stdc++.h>
using namespace std;
const int N=36;
int n,pre[N],post[N];
vector<int>in;
bool uni=true;
struct TreeNode{
    int val;
    TreeNode *left,*right;
    TreeNode(int x):val(x),left(NULL),right(NULL){}
};
TreeNode *build(int pre_left,int pre_right,int post_left,int post_right){
    if(pre_left>pre_right||post_left>post_right)
        return NULL;
    if(post_right-post_left==1)
        uni=false;
    TreeNode *root=new TreeNode(pre[pre_left]);
    if(post_right-1<post_left)
        return root;
    // 在pre中找post_right-1，即右子树的根，分为左右子树
    int mid=pre_left+1;
    while(mid<=pre_right&&pre[mid]!=post[post_right-1])
        mid++;
    root->left=build(pre_left+1,mid-1,post_left,post_left+mid-pre_left-2);
    root->right=build(mid,pre_right,post_left+mid-pre_left-1,post_right-1);
    return root;
}
void intra(TreeNode *root){
    if(!root)
        return;
    intra(root->left);
    in.push_back(root->val);
    intra(root->right);
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>pre[i];
    for(int i=0;i<n;i++)
        cin>>post[i];
    TreeNode *root=build(0,n-1,0,n-1);
    if(uni)
        cout<<"Yes\n";
    else
        cout<<"No\n";
    intra(root);
    for(int i=0;i<in.size();i++){
        if(i!=0)cout<<" ";
        cout<<in[i];
    }
    cout<<'\n';
    return 0;
}

# 1123 Is It a Complete AVL Tree

传送门
一次过
是否是完全平衡二叉树，并输出层序遍历
建树，然后判断是否是完全二叉树
如何判断是完全二叉树？借用层序遍历，遇到空结点，判断是否经过全部的结点。如果全访问过了，则是完全二叉树，否则不是

#include<bits/stdc++.h>
using namespace std;
struct TNode{
    int val,height;
    TNode *left,*right;
    TNode(int x):val(x),height(1),left(NULL),right(NULL){}
};
typedef TNode* Tree;
int getHeight(Tree T){
    if(!T)
        return 0;
    return T->height;
}
int getBF(Tree T){
    return getHeight(T->left)-getHeight(T->right);
}
void updateHeight(Tree &T){
    T->height=max(getHeight(T->left),getHeight(T->right))+1;
}
void r(Tree &T){
    Tree tmp=T->left;
    T->left=tmp->right;
    tmp->right=T;
    updateHeight(T);
    updateHeight(tmp);
    T=tmp;
}
void l(Tree &T){
    Tree tmp=T->right;
    T->right=tmp->left;
    tmp->left=T;
    updateHeight(T);
    updateHeight(tmp);
    T=tmp;
}
Tree insert(Tree T,int X){
    if(!T)
        return new TNode(X);
    else if(X<T->val){
        T->left=insert(T->left,X);
        if(getBF(T)==2){
            if(getBF(T->left)==1)
                r(T);
            else if(getBF(T->left)==-1){
                l(T->left);
                r(T);
            }
        }
    }
    else if(X>T->val){
        T->right=insert(T->right,X);
        if(getBF(T)==-2){
            if(getBF(T->right)==-1)
                l(T); 
            else if(getBF(T->right)==1){
                r(T->right);
                l(T);
            }
        }
    }
    updateHeight(T);
    return T;
}
bool levelTra(Tree T,int n){
    queue<Tree>q;
    bool iscom=true;
    int cnt=0;
    Tree tmp;
    q.push(T);
    while(!q.empty()){
        tmp=q.front();
        q.pop();
        if(tmp){
            if(cnt!=0)
                cout<<" ";
            cout<<tmp->val;
            cnt++;
            q.push(tmp->left);
            q.push(tmp->right);
        }else if(cnt==n)
            break;   
        else
            iscom=false;
    }
    cout<<'\n';
    return iscom;
}
int main(){
    Tree root=NULL;
    int n,num;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>num;
        root=insert(root,num);
    }

}

# 1127 ZigZagging on a Tree

传送门
给定后序和中序遍历序列，输出这棵树’Z’字形的层序遍历序列，先从左往右再从右往左输出
在每一层判断是从左往右还是从右往左，按顺序加进数组，然后输出该数组即可

#include<bits/stdc++.h>
using namespace std;
struct TreeNode{
    int val;
    TreeNode *left,*right;
    TreeNode(int v):val(v),left(NULL),right(NULL){}
};
typedef TreeNode* Tree;
const int N=36;
int n,in[N],post[N];
Tree build(Tree T,int root,int l,int r){
    if(l>r)
        return NULL;
    int mid=l;
    while(mid<=r&&in[mid]!=post[root])
        mid++;
    T=new TreeNode(post[root]);
    T->left=build(T->left,root-r+mid-1,l,mid-1);
    T->right=build(T->right,root-1,mid+1,r);
    return T;
}
void levelTra(Tree T){
    queue<Tree>q;
    q.push(T);
    bool f=false,print=false;
    int size;
    Tree tmp;
    vector<int>v;
    while(!q.empty()){
        size=q.size();
        while(size--){
            tmp=q.front();
            q.pop();
            if(f)
                v.push_back(tmp->val);
            else
                v.insert(v.begin(),tmp->val);
            if(tmp->left)
                q.push(tmp->left);
            if(tmp->right)
                q.push(tmp->right);
        }
        for(int i:v){
            if(print)
                cout<<" ";
            cout<<i;
            print=true;
        }
        v.clear();
        f=!f;
    }
}
int main(){
    cin>>n;

        cin>>in[i];
    for(int i=0;i<n;i++)
        cin>>post[i];
    Tree T=NULL;

# 1131 Subway Map

传送门
一次过
换乘地铁，求站最少且换乘最少的方案
无权最短路，因为要考虑换乘最少，所以DFS更合适

#include<bits/stdc++.h>
using namespace std;
const int N=1e5+7;
typedef pair<int,int> pii;
vector<pii>g[N],path;
bool vis[N];
int min_transfer,min_path;
void dfs(int u,int v,vector<pii> tmp){
    if(u==v){
        int tmp_transfer=0,tmp_path=tmp.size();
        for(int i=1;i<tmp.size();i++){
            if(tmp[i-1].second!=tmp[i].second)
                tmp_transfer++;
        }
        if((tmp_path<min_path)||(tmp_transfer<min_transfer&&tmp_path<=min_path)){
            min_transfer=tmp_transfer;
            min_path=tmp_path;
            path=tmp;
        }
        return;
    }
    for(auto i:g[u]){
        if(!vis[i.first]){
            vis[i.first]=true;
            tmp.push_back(i);
            dfs(i.first,v,tmp);
            tmp.pop_back();
            vis[i.first]=false;
        }
    }
}
int main(){
    int n,m,u,v,k,line,begin;
    vector<pii>tmp;
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>m>>u;
        for(int j=1;j<m;j++){
            cin>>v;
            g[u].push_back({v,i});
            g[v].push_back({u,i});
            u=v;
        }
    }
    cin>>k;
    while(k--){
        cin>>u>>v;
        fill(vis,vis+N,false);
        vis[u]=true;
        min_transfer=min_path=1e9;
        path.clear();
        tmp.clear();
        dfs(u,v,tmp);
        cout<<min_path<<endl;
        begin=u,line=path[0].second;
        for(int i=0;i<min_path;i++){
            if(path[i].second!=line){
                printf("Take Line#%d from %04d to %04d.\n",line,begin,path[i-1].first);
                begin=path[i-1].first;
                line=path[i].second;
            }
        }
        printf("Take Line#%d from %04d to %04d.\n",path[min_path-1].second,begin,path[min_path-1].first);
    }
    return 0;
}

# 1135 Is It A Red-Black Tree

传送门
判断是否是红黑树

  1. 根节点必须是黑
  2. 红的孩子必须是黑
  3. 任意节点到叶子所含的黑个数相同

步骤是

  1. 根据先序序列建树
  2. 根据以上条件判断并输出
#include<bits/stdc++.h>
using namespace std;
const int N=36;
struct TNode{
    int val;
    bool black;
    TNode *left,*right;
    TNode(int v,bool b):val(v),black(b),left(nullptr),right(nullptr){}
};
typedef TNode* BRTree;
typedef pair<BRTree,int> pti;
BRTree build(BRTree T,int X){
    if(!T)
        T=new TNode(abs(X),X>0);
    else if(abs(X)<T->val)
        T->left=build(T->left,X);
    else if(abs(X)>T->val)
        T->right=build(T->right,X);
    return T;
}
int getBlack(BRTree T){
    if(!T)
        return 1;
    int left=getBlack(T->left);
    int right=getBlack(T->right);
    return T->black?max(left,right)+1:max(left,right);
}
bool isbrt(BRTree T){
    if(!T->black)
        return false;
    queue<BRTree>q;
    q.push(T);
    while(!q.empty()){
        BRTree cur=q.front();
        q.pop();
        if(cur->left){
            if(!cur->black&&!cur->left->black)
                return false;
            q.push(cur->left);
        }
        if(cur->right){
            if(!cur->black&&!cur->right->black)
                return false;
            q.push(cur->right);
        }
        if(getBlack(cur->left)!=getBlack(cur->right))
            return false;
    }
    return true;
}
int main(){
    int n,k,v;
    cin>>k;
    BRTree T;
    while(k--){
        cin>>n;
        T=nullptr;
        for(int i=0;i<n;i++){
            cin>>v;
            T=build(T,v);
        }
        if(isbrt(T))
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}

# 1143 Lowest Common Ancestor

传送门
二叉搜索树的最近公共祖先
用set判断是否在树中
因为是BST，所以可以用一个特殊的方法：遍历前序序列，第一个出现的在[min(u,v),max(u,v)][min(u,v),max(u,v)][min(u,v),max(u,v)]的点即为uuu和vvv的LCA

#include<bits/stdc++.h>
using namespace std;
const int N=10010;
int pre[N];
unordered_set<int>pos;
int main(){
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int m,n,u,v;
    cin>>m>>n;
    for(int i=0;i<n;i++){
        cin>>pre[i];
        pos.insert(pre[i]);
    }
    bool f1,f2;
    while(m--){
        cin>>u>>v;
        f1=pos.count(u)!=0;
        f2=pos.count(v)!=0;
        if(!f1&&!f2)
            cout<<"ERROR: "<<u<<" and "<<v<<" are not found.\n";
        else if(!f1)
            cout<<"ERROR: "<<u<<" is not found.\n";
        else if(!f2)
            cout<<"ERROR: "<<v<<" is not found.\n";
        else{
            int ancestor;
            for(int i=0;i<n;i++){
                if(pre[i]>=min(u,v)&&pre[i]<=max(u,v)){
                    ancestor=pre[i];
                    break;
                }
            }
            if(ancestor==u)
                cout<<u<<" is an ancestor of "<<v<<".\n";
            else if(ancestor==v)
                cout<<v<<" is an ancestor of "<<u<<".\n";
            else
                cout<<"LCA of "<<u<<" and "<<v<<" is "<<ancestor<<".\n";
        }
    }
    return 0;
}

# 1147 Heaps

传送门
一次过
判断一个完全二叉树是否是堆，是大顶堆还是小顶堆，输出后序遍历
STL中is_heap的结果是错的，递归判断每个二层子树是否是堆即可
由层序遍历输出后序遍历根据下标关系递归即可

#include<bits/stdc++.h>
using namespace std;
vector<int>heap;
bool print;
void post(int root){
    if(root>=heap.size())
        return;
    post(2*root+1);
    post(2*root+2);
    if(print)
        cout<<" ";
    cout<<heap[root];
    print=true;
}
bool is_max(int root){
    if(root>=heap.size())
        return true;
    if((2*root+1<heap.size()&&heap[root]<heap[2*root+1])||(2*root+2<heap.size()&&heap[root]<heap[2*root+2]))
        return false;
    return is_max(2*root+1)&&is_max(2*root+2);
}
bool is_min(int root){
    if(root>=heap.size())
        return true;
    if((2*root+1<heap.size()&&heap[root]>heap[2*root+1])||(2*root+2<heap.size()&&heap[root]>heap[2*root+2]))
        return false;
    return is_min(2*root+1)&&is_min(2*root+2);
}
int main(){
    int m,n,num;
    cin>>m>>n;
    while(m--){
        heap.clear();
        for(int i=0;i<n;i++){
            cin>>num;
            heap.push_back(num);
        }
        if(is_max(0))
            cout<<"Max Heap\n";
        else if(is_min(0))
            cout<<"Min Heap\n";
        else
            cout<<"Not Heap\n";
        print=false;
        post(0);
        cout<<'\n';
    }
    return 0;
}

# 1151 LCA in a Binary Tree

传送门
最近公共祖先
与1143 Lowest Common Ancestor类似，但是此题不是BST

  1. 根据前序和中序建树
  2. 递归找LCA：如果两个结点分别在左右孩子，则该节点为LCA；全在右孩子，则在右孩子找LCA；全在左孩子，则在左孩子找LCA
#include<bits/stdc++.h>
using namespace std;
const int N=10010;
int pre[N],in[N],ans;
struct TNode{
    int val;
    TNode *left,*right;
    TNode(int v):val(v),left(nullptr),right(nullptr){}
};
typedef TNode* Tree;
Tree build(Tree T,int root,int l,int r){
    if(l>r)
        return nullptr;
    int mid=l;
    while(mid<=r&&in[mid]!=pre[root])
        mid++;
    T=new TNode(pre[root]);
    T->left=build(T->left,root+1,l,mid-1);
    T->right=build(T->right,root+mid-l+1,mid+1,r);
    return T;
}
Tree lca(Tree T,int u,int v){
    if(!T||T->val==u||T->val==v)
        return T;
    Tree l=lca(T->left,u,v);
    Tree r=lca(T->right,u,v);
    if(!l)
        return r;
    if(!r)
        return l;
    return T;
}
int main(){
    int n,m,u,v;
    unordered_set<int>st;
    cin>>m>>n;
    for(int i=0;i<n;i++){
        cin>>in[i];
        st.insert(in[i]);
    }
    for(int i=0;i<n;i++)
        cin>>pre[i];
    Tree T=nullptr;
    T=build(T,0,0,n-1);
    bool f1,f2;
    while(m--){
        cin>>u>>v;
        f1=st.count(u)!=0;
        f2=st.count(v)!=0;
        if(!f1&&!f2)
            cout<<"ERROR: "<<u<<" and "<<v<<" are not found.\n";
        else if(!f1)
            cout<<"ERROR: "<<u<<" is not found.\n";
        else if(!f2)
            cout<<"ERROR: "<<v<<" is not found.\n";
        else{
            Tree tmp=lca(T,u,v);
            ans=tmp->val;
            if(ans==u)
                cout<<u<<" is an ancestor of "<<v<<".\n";
            else if(ans==v)
                cout<<v<<" is an ancestor of "<<u<<".\n";
            else
                cout<<"LCA of "<<u<<" and "<<v<<" is "<<ans<<".\n";
        }
    }
    return 0;
}
