# 1003 Emergency

传送门
求最短距离的个数及最短距离时最大的救援队数量，dfs即可

#include <bits/stdc++.h>
using namespace std;
int n,m,c1,c2;
int rescue[520],l[520][520],visited[520],maxres=0,minrou=INT_MAX,cnt=0;
void init(){
    for(int i=0;i<520;i++){
        visited[i]=false;
        for(int j=0;j<520;j++){
            l[i][j]=-1;
        }
    }
}
void dfs(int s,int nrou, int nres){
    if(nrou>minrou)return;
    if(s==c2){
        nres+=rescue[s];
        if(minrou>nrou){
            cnt=1;
            minrou=nrou;
            maxres=nres;
        }else{
            cnt++;
            if(nres>maxres)maxres=nres;
        }
        return;
    }
    for(int i=0;i<n;i++){
        visited[s]=true;
        if(!visited[i]&&l[s][i]!=-1)dfs(i,nrou+l[s][i],nres+rescue[s]);
        visited[s]=false;
    }
}
int main(){
    init();
    cin>>n>>m>>c1>>c2;
    for(int i=0;i<n;i++)cin>>rescue[i];
    int C1,C2,L;
    for(int _=0;_<m;_++){
        cin>>C1>>C2>>L;
        l[C1][C2]=l[C2][C1]=L;
    }
    dfs(c1,0,0);
    cout<<cnt<<" "<<maxres<<'\n';
    return 0;
}

# 1010 Radix

传送门
基数范围为2到正无穷，在限制条件下，最小基数为N2最大元素加一，最大基数为十进制下的N1和最小基数的最大值
第一个测试用例（一分）总是未通过，应该是N1和N2相等情况下的输出，原因是最小基数和最大基数没设置对
使用顺序查找的话，这时候会有一个一分的测试用例总是超时，需要使用二分查找降低时间复杂度，比大小时需要判断一下转十进制时导致的上溢出情况（结果小于0）

#include <bits/stdc++.h>
using namespace std;
long long ch2ll(char ch){
    return isdigit(ch)?ch-'0':ch-'a'+10;
}
long long radix2ten(string x,long long radix){
    long long tmp=1,sum=0;
    for(int i=x.length()-1;i>=0;i--){
        sum+=ch2ll(x[i])*tmp;
        tmp*=radix;
    }
    return sum;
}
long long run(string x,string y,long long radix){
    long long a=radix2ten(x,radix),b;
    long long low=ch2ll(*max_element(y.begin(),y.end()))+1,high=max(a,low),mid;
    while(low<=high){
        mid=(low+high)/2;
        b=radix2ten(y,mid);
        if(a==b)return mid;
        else if(b>a||b<0)high=mid-1;
        else low=mid+1;
    }
    return -1;
}
int main(){
    string n1,n2;
    long long t,r,res;
    cin>>n1>>n2>>t>>r;
    if(t==1)res=run(n1,n2,r);
    else if(t==2)res=run(n2,n1,r);
    if(res==-1)cout<<"Impossible";
    else cout<<res;
    return 0;
}

# 1020 Tree Traversals

传送门
给出后序遍历和中序遍历，输出层序遍历
在后序遍历中，最后一个结点一定是树的根。先在后序遍历中得到树的根，然后在中序遍历中找到这个根，即可得到左右子树，在左右子树中按此规则再递归寻找，即可按左右子树、从上到下的顺序输出结果
但我们需要的是层序遍历，所以需要保存结点的序号，按序号的顺序输出结果，设根结点的序号为indexindexindex（以0开始），则左结点的序号为2×index+12 \times index + 12×index+1，右结点的序号为2×index+22 \times index + 22×index+2
题目给出最大结点个数为30，如果这棵树没有左子树只有右子树的话，假设序号以0开始，那么序号的最大值为230−22^{30}-2230−2，显然以这个数字取开数组的话是不可能的，所以用map代替。而map恰好在存的时候是按key从小到大顺序排好序的，所以直接按顺序输出即可，注意一下空格的输出就好了
第一次提交时数组开得不够大，导致了最后一个测试点出现了段错误（3分）

#include <bits/stdc++.h>

void run(int root,int begin,int end,int index){

# 1021 Deepest Root

传送门
可以得到最大深度的树根，按从小到大排序
通过一次或多次dfs可以遍历完整张图，进行了几次dfs代表连通子图个数为几。如果个数为1，则是树，再一次dfs，即可得到最深的距离，遍历的起点及终点都可以作为题目所要求的根
需要注意多叉情况下，结果为两次dfs得到的并集

#include <bits/stdc++.h>
using namespace std;
int n,a,b,k=0,maxdepth=-1;
bool visited[10010]={false};
vector<int>graph[10010],deep;
set<int>ans;
void dfs(int start,int depth){
    visited[start]=true;
    if(depth==maxdepth)deep.push_back(start);
    else if(maxdepth<depth){
        maxdepth=depth;
        deep.clear();
        deep.push_back(start);
    }
    for(int i:graph[start]){
        if(!visited[i]){
            dfs(i,depth+1);
        }
    }
}
int main(){
    cin>>n;
    for(int i=0;i<n-1;i++){
        cin>>a>>b;
        graph[a].push_back(b);
        graph[b].push_back(a);
    }
    for(int i=1;i<=n;i++){
        if(!visited[i]){
            k++;
            dfs(i,0);
        }
    }
    if(k!=1)printf("Error: %d components",k);
    else{
        for(int i=1;i<=n;i++)visited[i]=false;
        for(auto i:deep)ans.insert(i);
        dfs(deep[0],0);
        for(auto i:deep)ans.insert(i);
        for(auto i:ans)cout<<i<<'\n';
    }
    return 0;
}

# 1024 Palindromic Number

#include <bits/stdc++.h>
using namespace std;
bool palindromic(string n){
    int i=0,j=n.size()-1;
    while(i<=j){
        if(n[i]!=n[j])return false;
        i++;j--;
    }
    return true;
}
string add(string x){
    string y=x,res;int cs=0,a;
    reverse(y.begin(),y.end());
    for(int i=x.length()-1;i>=0;i--){
        a=(x[i]-'0')+(y[i]-'0')+cs;
        res.push_back(a%10+'0');
        cs=a/10;
    }
    if(cs!=0)res.push_back(cs+'0');
    reverse(res.begin(),res.end());
    return res;
}
int main(){

# 1043 Is It a Binary Search Tree

传送门
给一串树的前序遍历，判断是否是二叉搜索树或其镜像，如果是，输出YES及其后序遍历
通过前序遍历，我们可以知道根、左子树和右子树。通过先遍历左子树，再遍历右子树，最后遍历根，可以得到后序遍历
如果后序遍历得不到相同数量的结点，表明不是bst。但是它有可能是mirror的，所以需要再遍历一次判断

#include <bits/stdc++.h>
using namespace std;
vector<int>pre,post;
int n,num;
bool mirror=false;
void postorder(int begin,int end){
    if(begin>end)return;
    int right=begin+1,left=end;
    if(mirror){
        while(right<=end&&pre[right]>=pre[begin])right++;
        while(left>begin&&pre[left]<pre[begin])left--;
    }else{
        while(right<=end&&pre[right]<pre[begin])right++;
        while(left>begin&&pre[left]>=pre[begin])left--;
    }
    if(left+1!=right)return;
    postorder(begin+1,left);
    postorder(right,end);
    post.push_back(pre[begin]);
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>num;
        pre.push_back(num);
    }
    postorder(0,n-1);
    if(post.size()!=n){
        mirror=true;
        post.clear();
        postorder(0,n-1);
    }
    if(post.size()!=n)cout<<"NO";
    else{
        cout<<"YES\n"<<post[0];
        for(int i=1;i<n;i++)cout<<" "<<post[i];
    }
    return 0;
}

    for (int i = 0; i < n; i++) {
        cin >> p[i].id >> p[i].vg >> p[i].tg;
        p[i].add = p[i].vg + p[i].tg;
        if (p[i].vg >= l && p[i].tg >= l) {
            p[i].show = true;
            cnt++;
        } 
        else p[i].show = false;
        if (p[i].vg >= h && p[i].tg >= h) p[i].type = 4;
        else if (p[i].vg >= h && p[i].tg < h) p[i].type = 3;
        else if (p[i].vg < h && p[i].tg < h && p[i].vg >= p[i].tg) p[i].type = 2;
        else p[i].type = 1;
    }
    sort(p, p + n, cmp);
    cout << cnt << '\n';
    for (int i = 0; i < n; i++) {
        if(p[i].show)printf("%08d %d %d\n", p[i].id, p[i].vg, p[i].tg);
    }
    return 0;
}

# 1063 Set Similarity

传送门
一次过
用set记录，输入查询的时候再计算，公式为相同数字的个数除以去重后的不同数字个数

#include <bits/stdc++.h>
using namespace std;
const int N = 55;
int main() {
    int n, m, k, _, a, b, nc, nt;
    set<int> s[N];
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> m;
        for (int j = 0; j < m; j++) {
            cin >> _;
            s[i].insert(_);
        }
    }
    cin >> k;
    while (k--) {
        cin >> a >> b;
        nc = 0;
        nt = s[a].size() + s[b].size();
        for (auto i : s[a]) {
            if (s[b].find(i) != s[b].end()) nc++;
        }
        printf("%.01lf%\n", nc * 1.0 / (nt - nc) * 100);
    }
    return 0;
}
