#include <bits/stdc++.h>

long long ch2ll(char ch){
    return isdigit(ch)?ch-'0':ch-'a'+10;
}
long long radix2ten(string x,long long radix){
    long long tmp=1,sum=0;
    for(int i=x.length()-1;i>=0;i--){
        sum+=ch2ll(x[i])*tmp;
        tmp*=radix;
    }
    return sum;
}
long long run(string x,string y,long long radix){
    long long a=radix2ten(x,radix),b;
    long long low=ch2ll(*max_element(y.begin(),y.end()))+1,high=max(a,low),mid;
    while(low<=high){
        mid=(low+high)/2;
        b=radix2ten(y,mid);
        if(a==b)return mid;
        else if(b>a||b<0)high=mid-1;
        else low=mid+1;

}

传送门

通过前序遍历，我们可以知道根、左子树和右子树。通过先遍历左子树，再遍历右子树，最后遍历根，可以得到后序遍历

#include <bits/stdc++.h>
using namespace std;
vector<int>pre,post;
int n,num;
bool mirror=false;
void postorder(int begin,int end){
    if(begin>end)return;
    int right=begin+1,left=end;
    if(mirror){
        while(right<=end&&pre[right]>=pre[begin])right++;
        while(left>begin&&pre[left]<pre[begin])left--;
    }else{
        while(right<=end&&pre[right]<pre[begin])right++;
        while(left>begin&&pre[left]>=pre[begin])left--;
    }
    if(left+1!=right)return;
    postorder(begin+1,left);
    postorder(right,end);
    post.push_back(pre[begin]);
}

    for(int i=0;i<n;i++){
        cin>>num;
        pre.push_back(num);
    }
    postorder(0,n-1);
    if(post.size()!=n){
        mirror=true;
        post.clear();
        postorder(0,n-1);
    }
    if(post.size()!=n)cout<<"NO";
