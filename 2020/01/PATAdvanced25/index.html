#include<bits/stdc++.h>
using namespace std;
int main(){
    double _a[1001]={0.0};
    int k,n;
    double a;
    cin>>k;
    for(int i=0;i<k;i++){
        cin>>n>>a;
        _a[n]+=a;
    }
    cin>>k;
    for(int i=0;i<k;i++){
        cin>>n>>a;
        _a[n]+=a;
    }
    k=0;
    for(int i=0;i<=1000;i++){
        if(_a[i]!=0.0)k++;
    }
    cout<<k;
    for(int i=1000;i>=0;i--){
        if(_a[i]!=0.0)printf(" %d %.1f",i,_a[i]);
    }
    return 0;
}

1028 List Sorting

传送门
一次过
结构体排序

#include <bits/stdc++.h>
using namespace std;
struct info{
    int id;
    string name;
    int grade;
}record[int(1e5)];
bool cmp1(info &a, info &b){
    return a.id<b.id;
}
bool cmp2(info &a, info &b){
    if(a.name==b.name)return a.id<b.id;
    return a.name<b.name;
}
bool cmp3(info &a, info &b){
    if(a.grade==b.grade)return a.id<b.id;
    return a.grade<b.grade;
}
int main() {
    int n,c;
    cin>>n>>c;
    for(int i=0;i<n;i++){
        cin>>record[i].id>>record[i].name>>record[i].grade;
    }
    if(c==1)sort(record,record+n,cmp1);
    else if(c==2)sort(record,record+n,cmp2);
    else if(c==3)sort(record,record+n,cmp3);
    for(int i=0;i<n;i++){
        printf("%06d %s %d\n",record[i].id,record[i].name.c_str(),record[i].grade);
    }
    return 0;
}
