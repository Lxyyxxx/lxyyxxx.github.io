1003 Emergency

传送门
求最短距离的个数及最短距离时最大的救援队数量，dfs即可

#include <bits/stdc++.h>
using namespace std;
int n,m,c1,c2;
int rescue[520],l[520][520],visited[520],maxres=0,minrou=INT_MAX,cnt=0;
void init(){
    for(int i=0;i<520;i++){
        visited[i]=false;
        for(int j=0;j<520;j++){
            l[i][j]=-1;
        }
    }
}
void dfs(int s,int nrou, int nres){
    if(nrou>minrou)return;
    if(s==c2){
        nres+=rescue[s];
        if(minrou>nrou){
            cnt=1;
            minrou=nrou;
            maxres=nres;
        }else{
            cnt++;
            if(nres>maxres)maxres=nres;
        }
        return;
    }
    for(int i=0;i<n;i++){
        visited[s]=true;
        if(!visited[i]&&l[s][i]!=-1)dfs(i,nrou+l[s][i],nres+rescue[s]);
        visited[s]=false;
    }
}
int main(){
    init();
    cin>>n>>m>>c1>>c2;
    for(int i=0;i<n;i++)cin>>rescue[i];
    int C1,C2,L;
    for(int _=0;_<m;_++){
        cin>>C1>>C2>>L;
        l[C1][C2]=l[C2][C1]=L;
    }
    dfs(c1,0,0);
    cout<<cnt<<" "<<maxres<<'\n';
    return 0;
}

1010 Radix

传送门
基数范围为2到正无穷，在限制条件下，最小基数为N2最大元素加一，最大基数为十进制下的N1和最小基数的最大值
第一个测试用例（一分）总是未通过，应该是N1和N2相等情况下的输出，原因是最小基数和最大基数没设置对
使用顺序查找的话，这时候会有一个一分的测试用例总是超时，需要使用二分查找降低时间复杂度，比大小时需要判断一下转十进制时导致的上溢出情况（结果小于0）

#include <bits/stdc++.h>
using namespace std;
long long ch2ll(char ch){
    return isdigit(ch)?ch-'0':ch-'a'+10;
}
long long radix2ten(string x,long long radix){
    long long tmp=1,sum=0;
    for(int i=x.length()-1;i>=0;i--){
        sum+=ch2ll(x[i])*tmp;
        tmp*=radix;
    }
    return sum;
}
long long run(string x,string y,long long radix){
    long long a=radix2ten(x,radix),b;
    long long low=ch2ll(*max_element(y.begin(),y.end()))+1,high=max(a,low),mid;
    while(low<=high){
        mid=(low+high)/2;
        b=radix2ten(y,mid);
        if(a==b)return mid;
        else if(b>a||b<0)high=mid-1;
        else low=mid+1;
    }
    return -1;
}
int main(){
    string n1,n2;
    long long t,r,res;
    cin>>n1>>n2>>t>>r;
    if(t==1)res=run(n1,n2,r);
    else if(t==2)res=run(n2,n1,r);
    if(res==-1)cout<<"Impossible";
    else cout<<res;
    return 0;
}

1012 The Best Rank

传送门
找一个id在三科和平均分中最高的排名及相应的科目或平均分，要注意并列时的排名（第二和第三个测试点，4分）

#include <bits/stdc++.h>
using namespace std;
struct score{
    string id;
    int c,m,e,a;
}s[2020];
bool cmpC(score &x,score &y){return x.c>y.c;}
bool cmpM(score &x,score &y){return x.m>y.m;}
bool cmpE(score &x,score &y){return x.e>y.e;}
bool cmpA(score &x,score &y){return x.a>y.a;}
map<string,vector<int>>r;
int main(){
    int N,M,c,m,e,a,bestrank,bestindex=0;
    string id;
    char p[4]={'A','C','M','E'};
    cin>>N>>M;
    for(int i=0;i<N;i++){
        cin>>id>>c>>m>>e;
        a=(c+m+e)/3;
        s[i].id=id;s[i].c=c;s[i].m=m;s[i].e=e;s[i].a=a;
    }
    sort(s,s+N,cmpA);
    for(int i=0;i<N;i++)r[s[i].id].push_back((i!=0&&s[i].a==s[i-1].a)?i:i+1);
    sort(s,s+N,cmpC);
    for(int i=0;i<N;i++)r[s[i].id].push_back((i!=0&&s[i].c==s[i-1].c)?i:i+1);
    sort(s,s+N,cmpM);
    for(int i=0;i<N;i++)r[s[i].id].push_back((i!=0&&s[i].m==s[i-1].m)?i:i+1);
    sort(s,s+N,cmpE);
    for(int i=0;i<N;i++)r[s[i].id].push_back((i!=0&&s[i].e==s[i-1].e)?i:i+1);
    while(M--){
        cin>>id;
        if(!r.count(id))cout<<"N/A\n";
        else{
            bestrank=2020;
            for(int i=0;i<4;i++){
                if(r[id][i]<bestrank){
                    bestrank=r[id][i];
                    bestindex=i;
                }
            }
            cout<<bestrank<<" "<<p[bestindex]<<"\n";
        }
    }
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
int t[24],n,st=0;
struct bill{

}
int cal(bill &x,bill &y){

}
double charge(bill &x,bill &y){
    double sum=0;
    if(y.hour<x.hour){
        sum+=(y.day-1-x.day)*st*0.6;

        for(int i=x.hour+1;i<y.hour;i++)sum+=t[i]*60*1.0/100;
    }
    if(x.hour==y.hour)sum+=(y.minute-x.minute)*t[x.hour]*1.0/100;
    else sum+=((60-x.minute)*t[x.hour]+t[y.hour]*y.minute)*1.0/100;
    return sum;
}
int main(){

1020 Tree Traversals

传送门
给出后序遍历和中序遍历，输出层序遍历
在后序遍历中，最后一个结点一定是树的根。先在后序遍历中得到树的根，然后在中序遍历中找到这个根，即可得到左右子树，在左右子树中按此规则再递归寻找，即可按左右子树、从上到下的顺序输出结果
但我们需要的是层序遍历，所以需要保存结点的序号，按序号的顺序输出结果，设根结点的序号为indexindexindex（以0开始），则左结点的序号为2×index+12 \times index + 12×index+1，右结点的序号为2×index+22 \times index + 22×index+2
题目给出最大结点个数为30，如果这棵树没有左子树只有右子树的话，假设序号以0开始，那么序号的最大值为230−22^{30}-2230−2，显然以这个数字取开数组的话是不可能的，所以用map代替。而map恰好在存的时候是按key从小到大顺序排好序的，所以直接按顺序输出即可，注意一下空格的输出就好了
第一次提交时数组开得不够大，导致了最后一个测试点出现了段错误（3分）

#include <bits/stdc++.h>
using namespace std;
int n,post[36],in[36];
map<int,int>level;
void run(int root,int begin,int end,int index){
    if(begin>end)return;
    level[index]=post[root];
    int mid=begin;
    while(mid<end&&in[mid]!=post[root])mid++;
    run(root-end+mid-1,begin,mid-1,2*index+1);
    run(root-1,mid+1,end,2*index+2);
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++)cin>>post[i];
    for(int i=0;i<n;i++)cin>>in[i];
    run(n-1,0,n-1,0);
    for(auto i:level){
        if(i.second!=post[n-1])cout<<" ";
        cout<<i.second;
    }
    return 0;
}

1021 Deepest Root

传送门
可以得到最大深度的树根，按从小到大排序
通过一次或多次dfs可以遍历完整张图，进行了几次dfs代表连通子图个数为几。如果个数为1，则是树，再一次dfs，即可得到最深的距离，遍历的起点及终点都可以作为题目所要求的根
需要注意多叉情况下，结果为两次dfs得到的并集

#include <bits/stdc++.h>
using namespace std;
int n,a,b,k=0,maxdepth=-1;
bool visited[10010]={false};
vector<int>graph[10010],deep;
set<int>ans;
void dfs(int start,int depth){
    visited[start]=true;
    if(depth==maxdepth)deep.push_back(start);
    else if(maxdepth<depth){
        maxdepth=depth;
        deep.clear();
        deep.push_back(start);
    }
    for(int i:graph[start]){
        if(!visited[i]){
            dfs(i,depth+1);
        }
    }
}
int main(){
    cin>>n;
    for(int i=0;i<n-1;i++){
        cin>>a>>b;
        graph[a].push_back(b);
        graph[b].push_back(a);
    }
    for(int i=1;i<=n;i++){
        if(!visited[i]){
            k++;
            dfs(i,0);
        }
    }
    if(k!=1)printf("Error: %d components",k);
    else{
        for(int i=1;i<=n;i++)visited[i]=false;
        for(auto i:deep)ans.insert(i);
        dfs(deep[0],0);
        for(auto i:deep)ans.insert(i);
        for(auto i:ans)cout<<i<<'\n';
    }
    return 0;
}

1024 Palindromic Number

传送门
要注意101010^{10}1010虽然没有超过unsigned long long的范围，但相加后的结果会超过，所以直接用string计算并判断

#include <bits/stdc++.h>
using namespace std;
bool palindromic(string n){
    int i=0,j=n.size()-1;
    while(i<=j){
        if(n[i]!=n[j])return false;
        i++;j--;
    }
    return true;
}
string add(string x){
    string y=x,res;int cs=0,a;
    reverse(y.begin(),y.end());
    for(int i=x.length()-1;i>=0;i--){
        a=(x[i]-'0')+(y[i]-'0')+cs;
        res.push_back(a%10+'0');
        cs=a/10;
    }
    if(cs!=0)res.push_back(cs+'0');
    reverse(res.begin(),res.end());
    return res;
}
int main(){
    string n;int k;
    cin>>n>>k;
    if(palindromic(n))cout<<n<<"\n0\n";
    else{
        for(int i=1;i<=k;i++){
            n=add(n);
            if(palindromic(n)){
                cout<<n<<"\n"<<i<<"\n";
                break;
            }else if(i==k){
                cout<<n<<"\n"<<k<<"\n";
            }
        }
    }
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
struct Rank{
    string number;
    int location,score,locationrank;
}r[30010];
int n,k,score,cnt=0,finalrank=0;
bool cmp(Rank x,Rank y){return (x.score==y.score&&x.number<y.number)||x.score>y.score;}
int main(){
    string number;
    number.resize(13);
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>k;
        for(int j=0;j<k;j++){
            cin>>number>>score;
            r[cnt++]={number,i,score};
        }
        sort(r+cnt-k,r+cnt,cmp);
        r[cnt-k].locationrank=1;
        for(int j=cnt-k+1;j<cnt;j++){
            if(r[j].score!=r[j-1].score)r[j].locationrank=j-cnt+k+1;
            else r[j].locationrank=r[j-1].locationrank;
        }
    }
    sort(r,r+cnt,cmp);
    cout<<cnt<<'\n';
    for(int i=0;i<cnt;i++){
        cout<<r[i].number<<" ";
        if(i==0||r[i].score!=r[i-1].score)finalrank=i+1;
        cout<<finalrank<<" "<<r[i].location<<" "<<r[i].locationrank<<'\n';
    }

1033 To Fill or Not to Fill

传送门
贪就完事了。需要判断一下当不到达最后一个加油站是否可以到终点（第5个测试点，2分）

#include <bits/stdc++.h>
using namespace std;
struct node{
    double price;
    int dis;
}station[510];
bool cmp(node a,node b){
    if(a.dis==b.dis)return a.price<b.price;
    return a.dis<b.dis;
}
int main(){
    double cmax,d,davg,cmaxd,nowdis=0,cheap=0,far=0;
    int n;
    cin>>cmax>>d>>davg>>n;
    cmaxd=cmax*davg;
    for(int i=0;i<n;i++){
        cin>>station[i].price>>station[i].dis;
    }
    sort(station,station+n,cmp);
    if(station[0].dis!=0)cout<<"The maximum travel distance = 0.00";
    else{
        bool flag;int j,i;
        for(i=0;i<n;){
            if(i==n-1){
                cheap+=station[i].price/davg*(d-far);
                nowdis+=cmaxd;
                break;
            }
            flag=false;
            for(j=i+1;nowdis+cmaxd>=station[j].dis&&j<n;j++){
                if(station[j].price<station[i].price){
                    flag=true;
                    break;
                }
            }
            if(nowdis+cmaxd>=d&&!flag&&j==n){
                cheap+=station[i].price/davg*(d-far);
                nowdis=d;
                break;
            }
            if(flag){
                cheap+=station[i].price/davg*(station[j].dis-far);
                far=station[j].dis;
                nowdis=station[j].dis;
                i=j;
            }else{
                cheap+=station[i].price/davg*(cmaxd+station[i].dis-far);
                far=cmaxd+station[i].dis;
                nowdis=station[j-1].dis;
                i=j-1;
            }
        }
        if(nowdis>=d)printf("%.02lf",cheap);
        else printf("The maximum travel distance = %.02lf",far+cmaxd);
    }
    return 0;
}

#include <bits/stdc++.h>

    string s;
    getline(cin,s);
    int ans=0,cnt,left,right;
    for(int i=1;i<s.size();i++){
        cnt=0;
        if(s[i]==s[i-1]){
            left=i-1;right=i;
            while(left>=0&&right<s.size()&&s[left]==s[right]){left--;right++;cnt+=2;}
            if(ans<cnt)ans=cnt;
        }else if(s[i-1]==s[i+1]){
            cnt++;
            left=i-1;right=i+1;
            while(left>=0&&right<s.size()&&s[left]==s[right]){left--;right++;cnt+=2;}
            if(ans<cnt)ans=cnt;
        }
    }
    if(ans==0)ans=1;

}

1043 Is It a Binary Search Tree

传送门
给一串树的前序遍历，判断是否是二叉搜索树或其镜像，如果是，输出YES及其后序遍历
通过前序遍历，我们可以知道根、左子树和右子树。通过先遍历左子树，再遍历右子树，最后遍历根，可以得到后序遍历
如果后序遍历得不到相同数量的结点，表明不是bst。但是它有可能是mirror的，所以需要再遍历一次判断

#include <bits/stdc++.h>
using namespace std;
vector<int>pre,post;
int n,num;
bool mirror=false;
void postorder(int begin,int end){
    if(begin>end)return;
    int right=begin+1,left=end;
    if(mirror){
        while(right<=end&&pre[right]>=pre[begin])right++;
        while(left>begin&&pre[left]<pre[begin])left--;
    }else{
        while(right<=end&&pre[right]<pre[begin])right++;
        while(left>begin&&pre[left]>=pre[begin])left--;
    }
    if(left+1!=right)return;
    postorder(begin+1,left);
    postorder(right,end);
    post.push_back(pre[begin]);
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>num;
        pre.push_back(num);
    }
    postorder(0,n-1);
    if(post.size()!=n){
        mirror=true;
        post.clear();
        postorder(0,n-1);
    }
    if(post.size()!=n)cout<<"NO";
    else{
        cout<<"YES\n"<<post[0];
        for(int i=1;i<n;i++)cout<<" "<<post[i];
    }
    return 0;
}

1044 Shopping in Mars

切割钻石，如果可以得到m块钱，输出所有解决方案。如果不能，输出大于m块钱且损失最小的解决方案

            }
            ans.push_back({start,i});
            sum-=d[start++];
            sum-=d[i--];

}
