#include<bits/stdc++.h>
using namespace std;
long long gcd(long long a, long long b){
    return a%b?gcd(b,a%b):b;
}
int main(){
    int n;long long a[110],b[110],tmp,lcm;char c;
    cin>>n;
    for(int i=0;i<n;i++)cin>>a[i]>>c>>b[i];
    for(int i=1;i<n;i++){
        tmp=gcd(b[i-1],b[i]);
        lcm=b[i-1]*b[i]/tmp;
        a[i]=a[i-1]*(lcm/b[i-1])+a[i]*(lcm/b[i]);
        b[i]=lcm;
        tmp=gcd(a[i],b[i]);
        a[i]/=tmp;
        b[i]/=tmp;
    }
    long long p=a[n-1]/b[n-1],q=a[n-1]-p*b[n-1];
    if(p&&q){
        tmp=gcd(q,b[n-1]);
        cout<<p<<" "<<q/tmp<<"/"<<b[n-1]/tmp;
    }else if(!p&&q){
        tmp=gcd(q,b[n-1]);
        cout<<q/tmp<<"/"<<b[n-1]/tmp;
    }else if(p)cout<<p;
    else cout<<"0";
    return 0;
}

7-10 树的遍历 (25分)

这题好像做过好几遍了……就后序遍历是左右根，中序遍历中间是左根右，然后可以造出一棵树，层序遍历下标和根结点关系就是两倍加一加二的关系，然后就可以存在一个数组里
但是这个数组太大开不出来，所以用map代替

#include<bits/stdc++.h>
using namespace std;
int mid[33],post[33];
map<int,int>level;
void go(int root,int start,int end,int index){
    if(start>end)return;
    int i=start;
    while(post[root]!=mid[i]&&i<end)i++;
    level[index]=post[root];
    go(root-end+i-1,start,i-1,2*index+1);
    go(root-1,i+1,end,2*index+2);
}
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++)cin>>post[i];
    for(int i=0;i<n;i++)cin>>mid[i];
    go(n-1,0,n-1,0);
    for(auto i:level){
        if(i.second!=post[n-1])cout<<" ";
        cout<<i.second;
    }
    return 0;
}

7-11 家庭房产 (25分)

很明显的并查集，然后再加个结构体排序，就是要区分一下是祖先还是未出现的，因为祖先里存的都是自己，然后孩子也要算进去

#include<bits/stdc++.h>
using namespace std;
const int MAX_ID = 10000;
int pre[MAX_ID];
bool visited[MAX_ID]={false};
struct person{
    int id,house,area;
}p[1010];
struct family{
    int id,num;
    double avghouse,avgarea;
    bool Min;
}f[MAX_ID];
bool cmp(const family &a, const family &b){
    if(a.avgarea==b.avgarea)return a.id<b.id;
    return a.avgarea>b.avgarea;
}
int Find(int x){
    while(x!=pre[x])x=pre[x];
    return x;
}
void Union(int x,int y){
    int fx=Find(x),fy=Find(y);
    if(fx<fy)pre[fy]=fx;
    else if(fx>fy)pre[fx]=fy;
    visited[x]=visited[y]=true;
}
int main(){
    int n,k,kid,fid,mid,cnt=0;
    cin>>n;
    for(int i=0;i<MAX_ID;i++)pre[i]=i;
    for(int i=0;i<n;i++){
        cin>>p[i].id>>fid>>mid>>k;
        visited[p[i].id]=true;
        if(fid!=-1)Union(fid,p[i].id);
        if(mid!=-1)Union(mid,p[i].id);
        for(int _=0;_<k;_++){
            cin>>kid;
            Union(p[i].id,kid);
        }
        cin>>p[i].house>>p[i].area;
    }
    for(int i=0;i<n;i++){
        int id=Find(p[i].id);
        f[id].id=id;
        f[id].avghouse+=p[i].house;
        f[id].avgarea+=p[i].area;
        f[id].Min=true;
    }
    for(int i=0;i<MAX_ID;i++){
        if(visited[i])f[Find(i)].num++;
        if(f[i].Min)cnt++;
    }
    for(auto &i:f){
        if(i.Min){
            i.avghouse/=(i.num*1.0);
            i.avgarea/=(i.num*1.0);
        }
    }
    cout<<cnt<<'\n';
    sort(f,f+MAX_ID,cmp);
    for(int i=0;i<cnt;i++){
        printf("%04d %d %.3lf %.3lf\n",f[i].id,f[i].num,f[i].avghouse,f[i].avgarea);
    }
    return 0;
}

    int ans=1,cnt,l,r;
    for(int i=1;i<s.size();i++){
        cnt=0;
        if(s[i]==s[i-1]){
            l=i-1;r=i;
            while(l>=0&&r<s.size()&&s[l]==s[r]){l--;r++;cnt+=2;}
            if(ans<cnt)ans=cnt;
        }else if(s[i-1]==s[i+1]){
            cnt++;
            l=i-1;r=i+1;
            while(l>=0&&r<s.size()&&s[l]==s[r]){l--;r++;cnt+=2;}
            if(ans<cnt)ans=cnt;
        }
    }
    cout<<ans;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
const int MAXSIZE=101;
int fr[MAXSIZE];
bool en[MAXSIZE][MAXSIZE];
int Find(int x){
    while(x!=fr[x])x=fr[x];
    return x;
}
void Union(int x,int y){
    int fx=Find(x),fy=Find(y);
    if(fx!=fy)fr[y]=x;
}
int main(){
    for(int i=0;i<MAXSIZE;i++)fr[i]=i;
    int n,m,k,a,b,c;
    cin>>n>>m>>k;
    for(int i=0;i<m;i++){
        cin>>a>>b>>c;
        if(c==1)Union(a,b);
        else en[a][b]=en[b][a]=true;
    }
    for(int i=0;i<k;i++){
        cin>>a>>b;
        if(Find(a)==Find(b)&&!en[a][b])cout<<"No problem\n";
