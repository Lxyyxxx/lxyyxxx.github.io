# 7-10 树的遍历 (25分)

这题好像做过好几遍了……就后序遍历是左右根，中序遍历中间是左根右，然后可以造出一棵树，层序遍历下标和根结点关系就是两倍加一加二的关系，然后就可以存在一个数组里
但是这个数组太大开不出来，所以用map代替

#include<bits/stdc++.h>
using namespace std;
int mid[33],post[33];
map<int,int>level;
void go(int root,int start,int end,int index){
    if(start>end)return;
    int i=start;
    while(post[root]!=mid[i]&&i<end)i++;
    level[index]=post[root];
    go(root-end+i-1,start,i-1,2*index+1);
    go(root-1,i+1,end,2*index+2);
}
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++)cin>>post[i];
    for(int i=0;i<n;i++)cin>>mid[i];
    go(n-1,0,n-1,0);
    for(auto i:level){
        if(i.second!=post[n-1])cout<<" ";
        cout<<i.second;
    }
    return 0;
}
