# B Light bulbs

开关灯泡，每次拉[l,r][l,r][l,r]区间内的灯泡，问最后亮着的有几颗
直接暴力会超时，时间复杂度应被控制在O(MlogM)O(MlogM)O(MlogM)
为了避免区间重复，将区间调整为[l,r+1)[l,r+1)[l,r+1)，然后进行排序，每两个为一组，表示这个区间内的灯是亮的，那么个数就是右端减左端，累加起来就可以得到答案
因为区间和括号一样是配对的，单数嵌套内的是亮，双数嵌套内的是暗，每两个为一组就可以保证选择的是单数个嵌套的区间，而没有被选择的则是双数嵌套的区间
用cin和cout会tle

#include<bits/stdc++.h>
using namespace std;
const int M=2e3+10;
int main(){
    int t,n,m,l,r,ans,id;
    scanf("%d",&t);
    int query[M];
    for(int _=1;_<=t;_++){
        scanf("%d%d",&n,&m);
        id=0;
        while(m--){
            scanf("%d%d",&l,&r);
            query[id++]=l;
            query[id++]=r+1;
        }
        sort(query,query+id);
        ans=0;
        for(int i=0;i<id;i+=2)
            ans+=query[i+1]-query[i];
        printf("Case #%d: %d\n",_,ans);
    }
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int mod=1e9+7;
const int N=3e3+10;
ll inv[N],fac[N],ans;
int n;
// 当前a的个数
unordered_map<int,int>cnt;
// 快速幂
ll _pow(ll a,ll b,ll m){
    ll ans=1;
    while(b){
        if(b&1)ans=ans*a%m;
        a=a*a%m;
        b>>=1;
    }
    return ans;
}
// 初始化阶乘和阶乘的逆元
void init(){
    fac[0]=fac[1]=inv[0]=1;
    inv[1]=_pow(fac[1],mod-2,mod);
    for(int i=2;i<=3000;i++){
        fac[i]=fac[i-1]*i%mod;
        inv[i]=_pow(fac[i],mod-2,mod);
    }
}
// dfs求解方案数
// num 为当前的a，pos 为当前是第几个元素，sum 为当前等式左边的结果，mul 为当前等式右边的结果
void dfs(ll num,ll pos,ll sum,ll mul){
    if(sum+n-pos==mul||sum==mul&&pos==n){
        ll t=fac[n];
        for(auto i:cnt){
            if(i.second!=0)
                t=t*inv[i.second]%mod;
        }
        if(n!=pos)t=t*inv[n-pos]%mod;
        ans=(ans+t)%mod;
        return;
    }
    for(int i=num;i<=n;i++){
        // 剪枝
        if(mul*i>sum+i+n-pos-1)break;
        cnt[i]++;
        dfs(i,pos+1,sum+i,mul*i);
        cnt[i]--;
    }
}
int main(){
    init();
    int t;
    cin>>t;
    while(t--){
        cin>>n;
        cnt.clear();
        ans=0;
        dfs(2,0,0,1);
        cout<<ans<<'\n';
    }
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
const int mod=1e9+7;
const int N=150010;
int main(){
    int t,n,ans,sum,dp[N],a[310];
    cin>>t;
    while(t--){
        cin>>n;
        fill(a,a+n,0);
        fill(dp,dp+N,0);
        ans=0;
        sum=0;
        dp[0]=1;
        for(int i=0;i<n;i++){
            cin>>a[i];
            sum+=a[i];
        }
        sort(a,a+n,[&](int x,int y){return x>y;});
        for(int i=0;i<n;i++){
            for(int j=sum;j>=a[i];j--){
                dp[j]=(dp[j]+dp[j-a[i]])%mod;
                if(j>=sum-j&&j-a[i]<=sum-j){
                    ans=(ans+dp[j-a[i]])%mod;
                }
            }
        }
        cout<<ans<<'\n';
    }
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int dp[15][N]={0};
int cal(int n,int b){
    int ans=0;
    while(n){
        ans+=(n%b);
        n/=b;
    }
    return ans;
}
void init(){
    for(int i=2;i<=10;i++){
        for(int j=1;j<=int(1e6);j++){
            dp[i][j]=dp[i][j-1]+cal(j,i);
        }
    }
}
int main(){
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    init();
    int t,n,b;
    cin>>t;
    for(int _=1;_<=t;_++){
        cin>>n>>b;
        cout<<"Case #"<<_<<": "<<dp[b][n]<<'\n';
    }
    return 0;
}
