#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=1e6+10;
// 快速幂
ll _pow(ll a,ll b,ll m){
    ll t=1;
    while(b){
        if(b&1)t=t*a%m;
        a=a*a%m;
        b>>=1;
    }
    return t;
}
// 欧拉函数
int _phi(int x){
    int ans=x;
    for(int i=2;i<=sqrt(x);i++){
        if(x%i==0){
            ans=ans/i*(i-1);
            while(x%i==0)x/=i;
        }
    }
    if(x>1)ans=ans/x*(x-1);
    return ans;
}
// 比较b个a的幂塔函数与m的大小
bool check(ll a,ll b,ll m){
    if(b==0)return 1<m;
    if(b==1)return a<m;
    if(b==2){
        if(a==2)return 4<m;
        if(a==3)return 27<m;
        if(a==4)return 256<m;
        if(a==5)return 3125<m;
        if(a==6)return 46656<m;
        if(a==7)return 823543<m;
        return false;
    }
    if(b==3){
        if(a==2)return 16<m;
        return false;
    }
    if(b==4){
        if(a==2)return 65535<m;
        return false;
    }
    return false;
}
// 计算b个a的幂塔函数
ll cal(ll a,ll b,ll m){
    if(m==1||b==0)return 1;
    int phi=_phi(m);
    ll t=cal(a,b-1,phi);
    // 互质时，使用欧拉定理
    if(__gcd(a,m)==1)return _pow(a,t%phi,m);
    // 不互质，使用扩展欧拉定理
    else if(check(a,b-1,phi))return _pow(a,t,m);
    else return _pow(a,t%phi+phi,m);
}
int main(){
    int t;
    ll a,b,m;
    cin>>t;
    while(t--){
        cin>>a>>b>>m;
        cout<<cal(a,b,m)%m<<'\n';
    }
    return 0;
}
