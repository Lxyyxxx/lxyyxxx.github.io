# A ultiplication

两分钟交了一份while的TLE了。然后解了一下不等式。题目需要求使

k2day−1⩾n\begin{aligned} k^{2^{day-1}}\geqslant n \end{aligned}k2day−1⩾n​

成立的最小daydayday，化简就可以得到答案

day=log⁡2(log⁡kn)+1\begin{aligned} day=\log_{2}(\log_{k}{n})+1 \end{aligned}day=log2​(logk​n)+1​

使用换底公式可以得到答案为

day=log⁡(log⁡nlog⁡k)log⁡2+1\begin{aligned} day=\dfrac{\log(\dfrac{\log n}{\log k})}{\log 2}+1 \end{aligned}day=log2log(logklogn​)​+1​

#include<bits/stdc++.h>

int main(){
    int t,n,k;
    cin>>t;
    for(int i=0;i<t;i++){

# B Spread

并查集，这里使用了路径压缩，需要把已感染的设为爹

#include<bits/stdc++.h>
using namespace std;
const int N=100010;
bool vis[N];
int fa[N];
int Find(int x){
    if(x!=fa[x])fa[x]=Find(fa[x]);
    return fa[x];
}
void Union(int x,int y){
    int fx=Find(x),fy=Find(y);
    if(fx!=fy){
        if(vis[fx])fa[fy]=fx;
        else if(vis[fy])fa[fx]=fy;
        else fa[fx]=fy;
    }
}
int main(){
    int T,t,n,m,id,a,b,cnt,father;
    cin>>T;
    while(T--){
        cin>>n>>m>>t;
        for(int i=0;i<=n;i++)fa[i]=i;
        fill(vis,vis+n+1,false);
        for(int i=0;i<m;i++){
            cin>>id;
            vis[id]=true;
        }
        for(int i=0;i<t;i++){
            cin>>a>>b;
            Union(a,b);
        }
        cnt=0;
        for(int i=1;i<=n;i++){
            father=Find(i);
            if(vis[father])cnt++;
        }
        cout<<cnt<<'\n';
    }
    return 0;
}

# C Transport

BFS求最短路，一开始是用DFS总是WA

#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int>pii;
const int MAXN=1010;
int T,n,m,go[4][2]={{-1,0},{1,0},{0,-1},{0,1}},t1,t2,p[6],l;
char c;
bool g[MAXN][MAXN];
int dis[MAXN][MAXN];
int bfs(int sx,int sy,int ex,int ey){
    queue<pii>q;int nx,ny;
    bool f=false;
    q.push({sx,sy});
    dis[sx][sy]=0;
    while(!q.empty()){
        auto u=q.front();q.pop();
        if(u.first==ex&&u.second==ey){
            f=true;
            break;
        }
        for(int i=0;i<4;i++){
            nx=u.first+go[i][0];ny=u.second+go[i][1];
            if(nx>=1&&nx<=n&&ny>=1&&ny<=m&&g[nx][ny]&&dis[nx][ny]==-1){
                q.push({nx,ny});
                dis[nx][ny]=dis[u.first][u.second]+1;
            }
        }
    }
    if(f)return dis[ex][ey];
    else return -1;
}
int main(){
    cin>>T;
    while(T--){
        cin>>n>>m;
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                g[i][j]=false;
                dis[i][j]=-1;
            }
        }
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                cin>>c;
                g[i][j]=(c == '.');
            }
        }
        for(int i=0;i<6;i++)cin>>p[i];
        t1=bfs(p[0],p[1],p[2],p[3]);
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                dis[i][j]=-1;
            }
        }
        t2=bfs(p[2],p[3],p[4],p[5]);
        if(t1==-1||t2==-1)cout<<"-1\n";
        else cout<<t1+t2<<'\n';
    }
    return 0;
}

算CNN/2C_N^{N/2}CNN/2​的时候TLE了，一开始是没记录直接逆元算的。后来改成记录阶乘就过了
化成阶乘的公式

CNN/2=N!(N/2)!(N−N/2)!=N!(N/2)!(N/2)!\begin{aligned} C_N^{N/2}=\dfrac{N!}{(N/2)!(N-N/2)!}=\dfrac{N!}{(N/2)!(N/2)!} \end{aligned}CNN/2​=(N/2)!(N−N/2)!N!​=(N/2)!(N/2)!N!​​

#include<bits/stdc++.h>

const int MAXN=1000010;
long long MOD=1000000007,y,x;

long long Pow(long long x,long long y,long long mod){
    long long t=1;
    while(y){
        if(y&1) t=t*x%mod;
        x=x*x%mod;
        y>>=1;

}
// 求逆元
long long ni(long long x){
    return Pow(x,MOD-2,MOD);
}

# F 非对称加密算法RSA

这题是补的
按照步骤就可以解出来。输出的就是ddd和memod  nm^e\mod nmemodn，其中在第二步已经说明n=p∗qn=p*qn=p∗q
ddd在第四步给出可以用扩展欧几里得算法解出
**扩展欧几里得算法(exgcd)**就是解方程组ax+by=gcd(a,b)ax+by=gcd(a,b)ax+by=gcd(a,b)中的xxx和yyy
第四步中给出的式子为e∗dmod  ϕ(n)=1e*d \mod \phi(n)=1e∗dmodϕ(n)=1，化为倍数关系就是e∗d=kϕ(n)+1e*d=k\phi(n)+1e∗d=kϕ(n)+1，移动一下就变为e∗d−kϕ(n)=1e*d-k\phi(n)=1e∗d−kϕ(n)=1
由于题中给定的e=65537e=65537e=65537是个质数，所以gcd(e,ϕ(n))=1gcd(e,\phi (n))=1gcd(e,ϕ(n))=1
那么式子就可以与方程式相对应ex+ϕ(n)y=gcd(e,ϕ(n))ex+\phi(n)y=gcd(e,\phi (n))ex+ϕ(n)y=gcd(e,ϕ(n))，其中xxx的解就是我们需要的ddd
关于扩展欧几里得算法的证明在这里

#include<bits/stdc++.h>
using namespace std;
const int e=65537;
int p,q,m,n,f,d,_;
// 扩展欧几里得，求ax+by=gcd(a,b)
int exgcd(int a,int b,int &x,int &y){
    if(!b){
        x=1;y=0;
        return a;
    }
    int d=exgcd(b,a%b,y,x);
    y-=a/b*x;
    return d;
}
long long Pow(long long x,long long y,long long mod){
    long long t=1;
    while(y){
        if(y&1)t=t*x%mod;
        x=x*x%mod;
        y>>=1;
    }
    return t;
}
int main(){
    cin>>p>>q>>m;
    n=p*q;
    f=(p-1)*(q-1);
    exgcd(e,f,d,_);
    cout<<(d%f+f)%f<<'\n'<<Pow(m,e,n);
    return 0;
}

公式就是全部的减去相隔的，也就是
mn−m∗(m−1)n−1\begin{aligned} m^n-m*(m-1)^{n-1} \end{aligned}mn−m∗(m−1)n−1​
但是因为nnn和ppp范围都是10510^5105级别的，所以在快速幂的时候会出现102010^{20}1020数量级的数，超过了long long，用__int128或快速乘可以解决

__int128

#include<bits/stdc++.h>
using namespace std;
long long MOD,ans,T,p,n,m,x,y;
// 快速幂
__int128 Pow(__int128 x,__int128 y,__int128 mod){
    __int128 t=1;
    while(y){
        if(y&1) t=t*x%mod;
        x=x*x%mod;
        y>>=1;
    }
    return t;
}
int main(){
    cin>>T;
    while(T--){
        cin>>n>>m>>p;
        MOD=n*p;

快速乘，要注意p、n、m也必须是long long而不是int

#include<bits/stdc++.h>
using namespace std;
long long MOD,ans,T,p,n,m;
// 快速乘
long long Mul(long long x,long long y,long long mod){
    long long t=0;
    while(y){
        if(y&1)t=(t+x)%mod;
        x=2*x%mod;
        y>>=1;
    }
    return t;
}
// 使用快速乘的快速幂
long long Pow(long long x,long long y,long long mod){
    long long t=1;
    while(y){
        if(y&1)t=Mul(t,x,mod);
        x=Mul(x,x,mod);
        y>>=1;
    }
    return t;
}
int main(){
    cin>>T;
    while(T--){
        cin>>n>>m>>p;
        MOD=n*p;
        ans=((Pow(m,n,MOD)-Pow(m-1,n-1,MOD)*m%MOD)%MOD+MOD)%MOD;
        cout<<ans<<'\n';
    }
    return 0;
}

# I 神奇的硬币I

这题是补的
找到最小的能付得起的val，计算这个val能付得起的w的个数。如果结果为0则输出-1
注意cin和cout会TLE

#include<bits/stdc++.h>
using namespace std;
const int N=1e6+10;
int main(){
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int n,m,val[N],w[N],cur=0,s=0;
    cin>>n>>m;
    for(int i=0;i<n;i++)cin>>val[i];
    for(int i=0;i<m;i++)cin>>w[i];
    sort(val,val+n);
    sort(w,w+m);
    while(s<n&&val[s]<w[0])s++;
    while(s<n&&cur<m&&w[cur]<=val[s])cur++;
    if(!cur)cur=-1;
    cout<<cur;
    return 0;
}
