% 线性灰度变换函数
function y = linear1(x,a,b,c,d)
y = d.*(x>b) + ((d-c)/(b-a)*(x-a)+c).*(x>=a & x<=b) + c.*(x<a);
y(y>255) = 255;
y = uint8(y);
end
% 分段线性灰度变换函数
function y = linear2(x,mf,mg,a,b,c,d)
y = ((mg-d)/(mf-b)*(x-b)+d).*(x>=b & x<mf) + ((d-c)/(b-a)*(x-a)+c).*(x>=a & x<b) + (c/a*x).*(x>=0 & x<a);
y(y>255) = 255;
y = uint8(y);
end
