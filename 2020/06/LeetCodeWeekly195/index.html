# 5448. 判断路径是否相交

记录已走过的点，查看是否已经访问过
别用unordered_set

class Solution {
public:
    bool isPathCrossing(string path) {
        set<pair<int,int>>st;
        int curx=0,cury=0;
        st.insert({curx,cury});
        for(auto s:path){
            if(s=='N')
                cury++;
            else if(s=='S')
                cury--;
            else if(s=='E')
                curx++;
            else if(s=='W')
                curx--;
            if(st.find({curx,cury})!=st.end())
                return true;
            st.insert({curx,cury});
        }
        return false;
    }
};

# 5449. 检查数组对是否可以被 k 整除

统计不同余数的个数，保证两个相加等于kkk的余数个数相等，这样才能被kkk整除
对于余数为000的，要找的是余数为kkk即为000的元素个数，所以余数为000的个数必须是偶数

class Solution {
public:
    bool canArrange(vector<int>& arr, int k) {
        vector<int>mod(k);
        for(int i:arr)
            mod[((i%k)+k)%k]++;
        if(mod[0]%2)
            return false;
        for(int i=1;i<=k/2;i++)
            if(mod[i]!=mod[k-i])
                return false;
        return true;
    }
};

# 5450. 满足条件的子序列数目

排序，然后枚举每一个num，二分找target-num，即可确定一个区间，然后就可以用公式算出该集合的子集个数，累加即可
因为222的次幂会非常大，不能直接进行位运算，所以可以记录以下对应的值，然后直接取
加减取模可能会导致负数，要注意一下

class Solution {
public:
    int numSubseq(vector<int>& nums, int target) {
        const int N=nums.size(),mod=1e9+7;
        sort(nums.begin(),nums.end());
        vector<int>pow(N+1,1);
        for(int i=1;i<=N;i++){
            pow[i]=(pow[i-1]<<1)%mod;
        }
        int ans=pow[N]-1,d,tmp,i,j;
        if(nums[N-1]+nums[N-1]<=target)
            return ans;
        for(i=N-1;i>=0;i--){
            int j=upper_bound(nums.begin(),nums.end(),target-nums[i])-nums.begin();
            j--;
            if(i==j)
                break;
            d=i-j-1;
            ans=(ans-pow[d]+mod)%mod;
            if(i==j||d==0)
                break;
        }
        return ans;
    }
};

# 5451. 满足不等式的最大值

给定一个坐标数组，按照横坐标 xxx 的值从小到大排序，且 xxx 均不相等。找出 yi+yj+∣xi−xj∣y_i + y_j + |x_i - x_j|yi​+yj​+∣xi​−xj​∣ 的最大值，其中 ∣xi−xj∣<=k|x_i - x_j| <= k∣xi​−xj​∣<=k
枚举每一个 (xi,yi)(x_i,y_i)(xi​,yi​)，则要求给定式子的最大值，只需要求 yj−xjy_j-x_jyj​−xj​ 的最大值即可，所以可以把 xix_ixi​ 之前的元素的 y−xy-xy−x 的值建一个堆，每次选择最大的即可，由于有坐标限制，所以不符合要求的坐标要及时从堆中删除

class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        int ans=-1e9;
        const int N=points.size();
        vector<pair<int,int>>v;
        for(auto i:points){
            while(!v.empty()&&i[0]-v[0].second>k){
                pop_heap(v.begin(),v.end());
                v.pop_back();
            }
            if(!v.empty()){
                ans=max(ans,i[0]+i[1]+v[0].first);
            }
            v.push_back({i[1]-i[0],i[0]});
            push_heap(v.begin(),v.end());
        }
        return ans;
    }
};
