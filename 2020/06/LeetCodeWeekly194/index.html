# 5440. 数组异或操作

class Solution {
public:
    int xorOperation(int n, int start) {
        int ans=0;
        for(int i=0;i<n;i++)
            ans^=(start+2*i);
        return ans;
    }
};

# 5441. 保证文件名唯一

如果冲突从k=1k=1k=1开始查找的话会TLE，所以在map里记录当前文件名的下一个可用的后缀，但不代表这个后缀就一定可用，如示例 2的gta(1)，所以要从这个后缀开始查找可用下标，并更新

  1. 当前文件名的下一个可用后缀
  2. 修改后文件名的下一个可用后缀
class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string,int>mp;
        vector<string>ans;
        string s;
        for(auto i:names){
            if(mp.count(i)==0){
                ans.push_back(i);
                mp[i]=1;
            }   
            else{
                for(int k=mp[i];;k++){
                    s=i;
                    s=s+"("+to_string(k)+")";
                    if(mp.count(s)==0){
                        ans.push_back(s);
                        mp[i]=k+1;
                        mp[s]=1;
                        break;
                    }
                }
            }
        }
        return ans;
    }
};

# 5442. 避免洪水泛滥

思想类似于操作系统中的内存页面置换的最佳置换算法，但不同的是，最佳置换是选取未来最长时间不会被访问的页面进行置换，而这里我们是选择未来最近会访问的湖泊将它置空
设当前访问的第iii天，
当rain[i]=0时，可以放空一个湖泊，步骤如下：

  1. 选取最小的jjj使得rain[j]∈{满的湖泊}andj>irain[j] \in \{满的湖泊\} \quad and \quad j>irain[j]∈{满的湖泊}andj>i
  2. 将rain[j]rain[j]rain[j]设为第iii天的答案，并更新rain[j]rain[j]rain[j]的下一次下雨的时刻

当rain[i]!=0时，表明会下雨，需要做以下操作：

  1. 判断rain[i]rain[i]rain[i]是否是满的，如果是满的，直接返回空
  2. 将rain[i]rain[i]rain[i]加入{满的湖泊}\{满的湖泊\}{满的湖泊}

可以看出，本算法最大的复杂度在选取最小的jjj上，如果顺序查找的话，复杂度是O(n)O(n)O(n)，成功TLE，所以我们需要降低复杂度
分析这句话可得，我们就是每次要在一个集合中选取一个最小的元素，这个操作就是优先队列干的事情，所以我们使用优先队列将复杂度降为O(log⁡n)O(\log n)O(logn)
如果可以避免洪水泛滥的话，在下一次下雨前肯定会将这个湖泊清空，所以这个集合就是各个满的湖泊下一次下雨的时刻
如果下一次下雨前没有把这个湖泊清空的话，就会导致洪水泛滥，结果就应该返回空，条件就体现为在rain[i]!=0时，集合中有i
但在集合中搜索还是需要至少O(log⁡n)O(\log n)O(logn)复杂度，可以改变判断条件优化到O(1)O(1)O(1)
对于求每个湖泊下雨的时刻，可以用O(n)O(n)O(n)扫一遍，按顺序插入对应的链表中
再扫一遍的时候，每次清空时将对应链表的头删去，代表这个时刻的湖泊被清空
如果湖泊为空，当前的时刻肯定是对应的链表的头，那么下一次下雨的时刻就是头的下一个结点；否则，就代表前面还有时刻的湖泊没有被清空，链表的头就不为当前时刻，此时返回空
有一个特殊情况，当优先队列为空时，可以随便置换，如果置换成rain[0]的话是不行的，因为有可能为000，题目给出湖泊是无穷多个的，所以直接置1或者其他正整数就可以了

class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        const int N=rains.size();
        vector<int>ans(N,-1);
        priority_queue<int,vector<int>,greater<int>>q;
        unordered_map<int,vector<int>>mp;
        for(int i=0;i<N;i++){
            if(rains[i]!=0)
                mp[rains[i]].push_back(i);
        }
        for(int i=0;i<N;i++){
            if(rains[i]!=0){
                if(mp[rains[i]][0]!=i)
                    return {};
                if(mp[rains[i]].size()>1)
                    q.push(mp[rains[i]][1]);
            }else{
                if(q.empty()){
                    ans[i]=1;
                    continue;
                }
                int pos=q.top();
                q.pop();
                ans[i]=rains[pos];
                mp[ans[i]].erase(mp[ans[i]].begin());
            }
        }
        return ans;
    }
