# 7-1 Good in C (20分)

传送门

# Idea

给定26个大写字母的7×57\times 57×5点阵图，输出给定字符串中单词的点阵图，每个单词间用一个换行区分，每个字符间用空格区分，不能有多余的空格或换行
给定的字符串包含空格，所以需要getline
两个单词间可能存在多个分隔符需要处理，开头结尾也会有分隔符
找这些情况的样例，去解决就好了
应该是这4题中最麻烦的题了吧2333

# Code

#include<bits/stdc++.h>
using namespace std;
char g[26][7][5];
string op[7];
void print(int x){
    for(int i=0;i<7;i++){
        for(int j=0;j<5;j++)
            op[i].push_back(g[x][i][j]);
        op[i]+=" ";
    }
}
int main(){
    for(int i=0;i<26;i++)
        for(int j=0;j<7;j++)
            for(int k=0;k<5;k++)
                cin>>g[i][j][k];
    getchar();
    string s;
    getline(cin,s);
    int l=0,r=s.size()-1;
    while(l<(int)s.size()&&!isupper(s[l]))
        l++;
    while(r>=0&&!isupper(s[r]))
        r--;
    for(int j=l;j<=r;j++){
        if(isupper(s[j]))
            print(s[j]-'A');
        else{
            if(op[0].empty())
                continue;
            for(int i=0;i<7;i++){
                op[i].pop_back();
                cout<<op[i]<<'\n';
                op[i].clear();
            }
            cout<<'\n';
        }
    }
    if(!op[0].empty()){
        for(int i=0;i<7;i++){
            op[i].pop_back();
            cout<<op[i]<<'\n';
        }
    }
    return 0;
}

# 7-2 Block Reversing (25分)

传送门

# Idea

重排链表，按块重排，每块倒序，块内还是正序
用栈存每块的开头，然后就可以按倒序输出每块了，块内直接正序遍历输出就好了
果然正确打开方式是从第二题开始做……

# Code

#include<bits/stdc++.h>
using namespace std;
const int N=1e5+7;
struct node{
    int val,next;
}l[N];
int first,n,k;
void solve(){
    stack<int>s;
    int cnt=0;
    for(int i=first;i!=-1;i=l[i].next){
        if(cnt%k==0)
            s.push(i);
        cnt++;
    }
    bool p=false;
    while(!s.empty()){
        int cur=s.top();
        s.pop();
        cnt=0;
        for(int i=cur;cnt<k&&i!=-1;i=l[i].next,cnt++){
            if(p)
                printf(" %05d\n",i);
            printf("%05d %d",i,l[i].val);
            p=true;
        }
    }
    printf(" -1\n");
}
int main(){
    cin>>first>>n>>k;
    int a,b,c;
    for(int i=0;i<n;i++){
        cin>>a>>b>>c;
        l[a]={b,c};
    }
    solve();
    return 0;
}

# 7-3 Summit (25分)

传送门

# Idea

安排峰会的座位区域，要求每个区域内两两都是直接认识的，并且是最大的区域
直接暴力就行，最大的就看有没有不在这个区域内的和这些人都直接认识

# Code

#include<bits/stdc++.h>
using namespace std;
const int N=210;
int n,m,k;
bool g[N][N];
int main(){
    cin>>n>>m;
    int u,v;
    for(int i=0;i<m;i++){
        cin>>u>>v;
        g[u][v]=g[v][u]=1;
    }
    cin>>k;
    int l,idea[N];
    bool f1,vis[N];
    for(int i=1;i<=k;i++){
        cin>>l;
        f1=true;
        fill(vis,vis+N,0);
        for(int j=0;j<l;j++){
            cin>>u;
            idea[j]=u;
            vis[j]=true;
        }
        for(int j=0;j<l;j++){
            for(int x=j+1;x<l;x++){
                if(!g[idea[j]][idea[x]]){
                    f1=false;
                    break;
                }
            }
        }
        if(f1){
            int id=-1;
            for(int j=1;j<=n;j++){
                if(vis[j])
                    continue;
                bool f=true;
                for(int x=0;x<l;x++){
                    if(!g[j][idea[x]]){
                        f=false;
                        break;
                    }
                }
                if(f){
                    id=j;
                    break;
                }
            }
            if(id==-1){
                cout<<"Area "<<i<<" is OK.\n";
            }else{
                cout<<"Area "<<i<<" may invite more people, such as "<<id<<".\n";
            }
        }else{
            cout<<"Area "<<i<<" needs help.\n";
        }
    }
    return 0;
}

# 7-4 Cartesian Tree (30分)

传送门

# Idea

笛卡尔树，结点的值满足堆结构，并且中序遍历出来还是原来的序列。给定一个序列，求最小堆笛卡尔树的层序遍历
找出最小的元素即可确定根节点及左右子树，然后就可以建树，再输出层序遍历就做完了

# Code

#include<bits/stdc++.h>
using namespace std;
const int N=36;
int n,a[N];
typedef struct TNode* Tree;
struct TNode{
    int val;
    Tree left,right;
    TNode(int v):val(v),left(nullptr),right(nullptr){}
};
Tree build(Tree t,int l,int r){
    if(l>r)
        return nullptr;
    int mid=min_element(a+l,a+r+1)-a;
    t=new TNode(a[mid]);
    t->left=build(t->left,l,mid-1);
    t->right=build(t->right,mid+1,r);
    return t;
}
void lev(Tree t){
    queue<Tree>q;
    q.push(t);
    bool p=false;
    while(!q.empty()){
        Tree cur=q.front();
        q.pop();
        if(p)
            cout<<" ";
        cout<<cur->val;
        p=true;
        if(cur->left)
            q.push(cur->left);
        if(cur->right)
            q.push(cur->right);
    }
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    Tree t=nullptr;
    t=build(t,0,n-1);
    lev(t);
    return 0;
}
