# 7-1 Rational Arithmetic

2014 冬

  1. 数据范围long int
  2. long int直接乘法会溢出long long，需要约分
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
void print(ll a,ll b){
    if(b==0){
        cout<<"Inf";
        return;
    }
    if(a==0){
        cout<<"0";
        return;
    }
    if(b<0){
        a=-a;
        b=-b;
    }
    bool f=false,f1=false;
    if(a*b<0){
        f=true;
        cout<<'(';
    }
    if(a/b!=0){
        f1=true;
        cout<<a/b;
        a%=b;
    }
    if(f1&&a!=0)
        cout<<' ';
    ll gcd=abs(__gcd(a,b));
    if(a!=0){
        if(a<0&&f1)
            cout<<(-a/gcd);
        else
            cout<<a/gcd;
        cout<<'/'<<b/gcd;
    }
    if(f)
        cout<<')';
}

# 7-3 Highest Price in Supply Chain

2014 冬

#include<bits/stdc++.h>
using namespace std;
const int N=1e5+1;
int n,root;
double p,r;
vector<int>nxt[N];
int deep=0,cnt=0;
void dfs(int s,int l){
    if(nxt[s].empty()){
        if(l>deep){
            cnt=1;
            deep=l;
        }else if(l==deep)
            cnt++;
        return;
    }
    for(int i:nxt[s])
        dfs(i,l+1);
}
int main(){
    cin>>n>>p>>r;
    r/=100;
    r+=1;
    int u;
    for(int i=0;i<n;i++){
        cin>>u;
        if(u==-1) 
            root=i;
        else
            nxt[u].push_back(i);
    }
    dfs(root,0);
    double d=pow(r,deep);
    printf("%.02lf %d",p*d,cnt);
    return 0;
}

# 7-4 All Roads Lead to Rome

2014 秋

#include<bits/stdc++.h>
using namespace std;
typedef pair<int,string> pis;
int n,k;
string s;
unordered_map<string,int>h;
unordered_map<string,vector<pis>>g;
unordered_map<string,int>dis;
unordered_map<string,vector<string>>pre;
void dijkstra(){
    priority_queue<pis,vector<pis>,greater<pis>>q;
    q.push({0,s});
    unordered_map<string,bool>vis;
    dis[s]=0;
    while(!q.empty()){
        pis cur=q.top();
        q.pop();
        if(vis[cur.second])
            continue;
        vis[cur.second]=1;
        for(auto &i:g[cur.second]){
            if(vis[i.second])
                continue;
            if(!dis.count(i.second))
                dis[i.second]=1e9;
            if(dis[i.second]>i.first+cur.first){
                pre[i.second].clear();
                pre[i.second].push_back(cur.second);
                dis[i.second]=i.first+cur.first;
                q.push({dis[i.second],i.second});
            }else if(dis[i.second]==i.first+cur.first){
                pre[i.second].push_back(cur.second);
                q.push({dis[i.second],i.second});
            }
        }
    }
}
vector<string>path,tmp;
int happiness=0,cnt=0;
double avg=0.0;
void dfs(string x){
    if(x==s){
        cnt++;
        tmp.push_back(x);
        int sum=0;
        for(int i=tmp.size()-2;i>=0;i--){
            sum+=h[tmp[i]];
        }
        double tmpavg=sum*1.0/(tmp.size()-1);
        if(sum>happiness){
            happiness=sum;
            path=tmp;
            avg=tmpavg;
        }else if(sum==happiness&&tmpavg>avg){
            happiness=sum;
            path=tmp;
            avg=tmpavg;
        }
        tmp.pop_back();
        return;
    }
    tmp.push_back(x);
    for(auto &i:pre[x])
        dfs(i);
    tmp.pop_back();
        
}
int main(){
    cin>>n>>k>>s;
    string u,v;
    int val;
    for(int i=0;i<n-1;i++){
        cin>>u>>val;
        h[u]=val;
    }
    for(int i=0;i<k;i++){
        cin>>u>>v>>val;
        g[u].push_back({val,v});
        g[v].push_back({val,u});
    }
    dijkstra();
    dfs("ROM"); 
    cout<<cnt<<" "<<dis["ROM"]<<" "<<happiness<<" "<<(int)avg<<'\n';
    for(int i=path.size()-1;i>=0;i--){
        cout<<path[i];
        if(i!=0)
            cout<<"->";
    }
    return 0;
}
