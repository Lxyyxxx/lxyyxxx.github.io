# L2-006 树的遍历 (25分)

传送门
2020/07/04 树
给定一棵二叉树的后序遍历和中序遍历，请你输出其层序遍历的序列。

#include<bits/stdc++.h>
using namespace std;
const int N=36;
int post[N],in[N],n;
struct TNode{
    int val;
    TNode *left,*right;
    TNode(int v):val(v),left(NULL),right(NULL){}
};
typedef TNode* Tree;
Tree build(Tree T,int root,int l,int r){
    if(l>r)
        return NULL;
    int mid=l;
    while(mid<=r&&in[mid]!=post[root])
        mid++;
    T=new TNode(post[root]);
    T->left=build(T->left,root-r+mid-1,l,mid-1);
    T->right=build(T->right,root-1,mid+1,r);
    return T;
}
void levelTra(Tree T){
    queue<Tree>q;
    q.push(T);
    bool print=false;
    Tree tmp;
    while(!q.empty()){
        tmp=q.front();
        q.pop();
        if(print)
            cout<<" ";
        cout<<tmp->val;
        print=true;
        if(tmp->left)
            q.push(tmp->left);
        if(tmp->right)
            q.push(tmp->right);
    }
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>post[i];
    for(int i=0;i<n;i++)
        cin>>in[i];
    Tree root=NULL;
    root=build(root,n-1,0,n-1);
    levelTra(root);
    return 0;
}

# L2-004 这是二叉搜索树吗？ (25分)

传送门
2020/07/07 树
如果输入序列是对一棵二叉搜索树或其镜像进行前序遍历的结果，则首先在一行中输出 YES ，然后在下一行输出该树后序遍历的结果。
判断：先根据给定的前序遍历建BST，再前序遍历BST看是否与原数组相同，不同再判断一次镜像

#include<bits/stdc++.h>
using namespace std;
const int N=1010;
int pre[N],pos=0;
bool print=false;
struct TNode{
    int val;
    TNode *left,*right;
    TNode(int v):val(v),left(NULL),right(NULL){}
};
typedef TNode* Tree;
Tree insert(Tree T,int X){
    if(!T)
        T=new TNode(X);
    else if(X<T->val)
        T->left=insert(T->left,X);
    else
        T->right=insert(T->right,X);
    return T;
}
bool judge1(Tree T){
    if(!T)
        return true;
    if(T->val!=pre[pos++])
        return false;
    return judge1(T->left)&&judge1(T->right);
}
bool judge2(Tree T){
    if(!T)
        return true;
    if(T->val!=pre[pos++])
        return false;
    return judge2(T->right)&&judge2(T->left);
}
void post1(Tree T){
    if(!T)
        return;
    post1(T->left);
    post1(T->right);
    if(print)
        cout<<" ";
    cout<<T->val;
    print=true;
}
void post2(Tree T){
    if(!T)
        return;
    post2(T->right);
    post2(T->left);
    if(print)
        cout<<" ";
    cout<<T->val;
    print=true;
}
int main(){
    int n;
    cin>>n;
    Tree T=NULL;
    for(int i=0;i<n;i++){
        cin>>pre[i];
        T=insert(T,pre[i]);
    }
    if(judge1(T)){
        cout<<"YES\n";
        post1(T);
    }else{
        pos=0;
        if(judge2(T)){
            cout<<"YES\n";
            post2(T);
        }else{
            cout<<"NO\n";
        }
    }
    return 0;
}

        T=new TNode(X);
    else if(X>T->val)
        T->left=insert(T->left,X);

        T->right=insert(T->right,X);

    q.push(T);

# L1-059 敲笨钟 (20分)

对每一行诗句，判断其是否压“ong”韵。即上下两句末尾的字都是“ong”结尾。如果是压此韵的，就按题面方法糟改之后输出，输出格式同输入；否则输出 Skipped，即跳过此句。

# L1-064 估值一亿的AI核心代码 (20分)

传送门
2020/07/16 字符串
本题要求你实现一个稍微更值钱一点的 AI 英文问答程序，规则是：

  1. 无论用户说什么，首先把对方说的话在一行中原样打印出来；
  2. 消除原文中多余空格：把相邻单词间的多个空格换成 1 个空格，把行首尾的空格全部删掉，把标点符号前面的空格删掉；
  3. 把原文中所有大写英文字母变成小写，除了 I；
  4. 把原文中所有独立的 can you、could you 对应地换成 I can、I could—— 这里“独立”是指被空格或标点符号分隔开的单词；
  5. 把原文中所有独立的 I 和 me 换成 you；
  6. 把原文中所有的问号 ? 换成惊叹号 !；
  7. 在一行中输出替换后的句子作为 AI 的回答。
#include<bits/stdc++.h>
using namespace std;
void lower(string &s){
    for(auto &i:s){
        if(isupper(i)&&i!='I')
            i=tolower(i);
        else if(i=='?')
            i='!';
    }
}
int main(){
    string s,t,tmp;
    int n;
    cin>>n;
    getchar();
    bool f;
    while(n--){
        getline(cin,s);
        // 1. 
        cout<<s<<'\n';
        // 2.
        stringstream ss(s);
        tmp="";
        f=false;
        while(ss>>t){
            // 3. 6.
            lower(t);
            // 开头为标点，代表前面有空格，合并到前一个单词，不输出空格
            if(!isalnum(t[0])||!f)
               tmp+=t;
            else
               tmp+=" "+t;
            f=true;
        }
        // 7.
        cout<<"AI: ";
        // 4. 5.
        for(int i=0;i<tmp.size();i++){
            if(!isalnum(tmp[i]))
                cout<<tmp[i];
            else if(i+9<=tmp.size()&&tmp.substr(i,9)=="could you"&&(i==0||!isalnum(tmp[i-1]))&&(i+9==tmp.size()||!isalnum(tmp[i+9]))){
                cout<<"I could";
                i+=8;
            }else if(i+7<=tmp.size()&&tmp.substr(i,7)=="can you"&&(i==0||!isalnum(tmp[i-1]))&&(i+7==tmp.size()||!isalnum(tmp[i+7]))){
                cout<<"I can";
                i+=6;
            }else if(i+2<=tmp.size()&&tmp.substr(i,2)=="me"&&(i==0||!isalnum(tmp[i-1]))&&(i+2==tmp.size()||!isalnum(tmp[i+2]))){
                cout<<"you";
                i+=1;
            }else if(i+1<=tmp.size()&&tmp[i]=='I'&&(i==0||!isalnum(tmp[i-1]))&&(i+1==tmp.size()||!isalnum(tmp[i+1]))){
                cout<<"you";
            }else
                cout<<tmp[i];
        }
        cout<<'\n';
    }
    return 0;
}

注意要把原话输出一遍，debug的时候是输入一个句子就输出一个答案，所以会把输入的当作输出的，对样例的时候会发现都一样啊为什么还是零分QAQ

# L2-032 彩虹瓶 (25分)

传送门
2020/07/21 栈
题意：固定栈的容量，问给的入栈顺序能不能生成1,2,…,N1,2,\dots,N1,2,…,N这样的出栈顺序

#include<bits/stdc++.h>
using namespace std;
int n,m,k;
int main(){
    cin>>n>>m>>k;
    int num;
    while(k--){
        int cur=1;
        stack<int>s;
        bool f=true;
        for(int i=0;i<n;i++){
            cin>>num;
            if(num==cur)
                cur++;
            else
                s.push(num);
            if(s.size()>m)
                f=false;
            while(!s.empty()){
                if(s.top()==cur){
                    cur++;
                    s.pop();
                }else
                	break;
            }
        }
        if(!f||cur!=n+1)
        	cout<<"NO\n";
        else
			cout<<"YES\n";
    }
    return 0;
}

# L2-031 深入虎穴 (25分)

传送门
2020/07/23 图
找出距离入口最远的那扇门。
注意：题目中没给起点，需要计算入度得到起点

#include<bits/stdc++.h>
using namespace std;
const int N=1e5+7;
vector<int>g[N];
bool vis[N];
int n,ans=0;
void bfs(int x){
    fill(vis,vis+N,0);
    queue<int>q;
    q.push(x);
    vector<int>dis(N,0);
    while(!q.empty()){
        int cur=q.front();
        q.pop();
        if(vis[cur])
            continue;
        vis[cur]=1;
        ans=cur;
        for(int i:g[cur]){
            if(vis[i])
                continue;
            q.push(i);
        }
    }
}
int main(){
    cin>>n;
    int k,u;
    for(int i=1;i<=n;i++){
        cin>>k;
        while(k--){
            cin>>u;
            g[i].push_back(u);
            vis[u]=1;
        }
    }
    int s=1;
    while(s<=n&&vis[s])
        s++;
    bfs(s);
    cout<<ans;
    return 0;
}

# L2-005 集合相似度 (25分)

传送门
2020/07/28 集合
给定两个整数集合，它们的相似度定义为：Nc/Nt×100N_c/N_t \times 100%Nc​/Nt​×100。其中NcN_cNc​是两个集合都有的不相等整数的个数，NtN_tNt​是两个集合一共有的不相等整数的个数。
用set存可以直接去重，只要统计共有的正数个数NcN_cNc​，就可以根据集合1大小+集合2大小−Nc集合1大小+集合2大小-N_c集合1大小+集合2大小−Nc​得到NtN_tNt​

#include<bits/stdc++.h>
using namespace std;
const int N=56;
int n;
unordered_set<int>st[N];
int main(){
    cin>>n;
    int m,k,num,u,v,nc,nt;
    for(int i=1;i<=n;i++){
        cin>>m;
        while(m--){
            cin>>num;
            st[i].insert(num);
        }
    }
    cin>>k;
    while(k--){
        cin>>u>>v;
        nc=nt=0;
        for(auto i:st[u]){
        	if(st[v].find(i)!=st[v].end())
        		nc++;
		}
		nt=st[u].size()+st[v].size()-nc;
        printf("%.02lf%%\n",nc*1.0/nt*100);
    }
    return 0;
}

algorithmn库有计算集合交并差的函数，可以通过得到交集，得到NcN_cNc​，再根据上述方法得到NtN_tNt​
注意：使用的时候容器内的数据必须是有序的，所以不能用unordered_set，直接用set

#include<bits/stdc++.h>
using namespace std;
const int N=56;
int n,nt;
set<int>st[N],nc;
int main(){
    cin>>n;
    int m,k,num,u,v;
    for(int i=1;i<=n;i++){
        cin>>m;
        while(m--){
            cin>>num;
            st[i].insert(num);
        }
    }
    cin>>k;
    while(k--){
        cin>>u>>v;
        nc.clear();
        set_intersection(st[u].begin(),st[u].end(),st[v].begin(),st[v].end(),inserter(nc,nc.begin()));
        nt=st[u].size()+st[v].size()-nc.size();
        printf("%.02lf%%\n",nc.size()*1.0/nt*100);
    }
    return 0;
}

# L2-019 悄悄关注 (25分)

传送门
2020/07/30 集合
我们认为被该用户点赞次数大于其点赞平均数、且不在其关注列表上的人，很可能是其悄悄关注的人。根据这个假设，请你按用户ID字母序的升序输出可能是其悄悄关注的人，每行1个ID。如果其实并没有这样的人，则输出“Bing Mei You”。
使用一个set存关注的人，这样也方便查找。然后用map存每个用户对应的点赞数，因为map已经按key进行升序排序了，所以直接遍历map判断点赞次数就可以找到悄悄关注的人了

#include<bits/stdc++.h>
using namespace std;
const int N=5010;
int n,m;
unordered_set<string>st;
map<string,int>mp;
int main(){
    cin>>n;
    string s;
    for(int i=0;i<n;i++){
        cin>>s;
        st.insert(s);
    }
    int sum=0,num,cnt;
    cin>>m;
    for(int i=0;i<m;i++){
        cin>>s>>num;
        mp[s]+=num;
        sum+=num;
    }
    cnt=mp.size();
    double avg=sum*1.0/cnt;
    bool op=0;
    for(auto &i:mp){
        if(i.second>avg&&st.count(i.first)==0){
            cout<<i.first<<'\n';
            op=1;
        }
    }
    if(!op)
        cout<<"Bing Mei You\n";
    return 0;
}

# L2-021 点赞狂魔 (25分)

传送门
2020/08/03 排序
统计每个人点赞的不同标签的数量，找出数量最大的前3名，在一行中顺序输出他们的用户名,其间以1个空格分隔,且行末不得有多余空格。如果有并列，则输出标签出现次数平均值最小的那个，题目保证这样的用户没有并列。若不足3人，则用-补齐缺失，例如mike jenny -就表示只有2人。
因为标签范围在[1,107][1,10^7][1,107]，所以用unordered_set存比较好，数组可能会爆掉，然后就可以统计不同标签的点赞数和全部点赞数，计算出标签出现次数平均值，进行结构体排序，输出答案
输出最大的前3名就是第K最大值，通常部分排序就可以搞定，比如：partial_sort、priority_queue、make_heap/push_heap + pop_heap都可以搞定，但是这题sort和sort_heap这些全部排序的方法耗时和前面那些部分排序的都差不多

#include<bits/stdc++.h>
using namespace std;
const int N=111;
struct data{
    string name;
    int sum;
    double avg;
}d[N];
bool cmp(const data &a,const data &b){
    if(a.sum==b.sum)
        return a.avg<b.avg;
    return a.sum>b.sum;
}
int main(){
    int n,k,f;
    cin>>n;
    string name;
    for(int i=0;i<n;i++){
        cin>>name>>k;
        d[i].name=name;
        unordered_set<int> like;
        int cnt=0;
        while(k--){
            cin>>f;
            like.insert(f);
            cnt++;
        }
        d[i].sum=like.size();
        d[i].avg=cnt*1.0/d[i].sum;
    }
    sort(d,d+n,cmp);
    bool p=false;
    for(int i=0;i<min(n,3);i++){
        if(p)
            cout<<' ';
        cout<<d[i].name;
        p=true;
    }
    for(int i=0;i<max(0,3-n);i++){
        if(p)
            cout<<' ';
        cout<<'-';
        p=true;
    }
    return 0;
}
