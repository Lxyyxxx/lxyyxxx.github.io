# string

字符串

  • string str="abcd";
  • string(cnt,char)
  • str[0]等价于*str.begin()，str.front()
  • 允许str.begin()+i，访问str[i]
  • 可以使用比较运算符
  • 可通过>>，<<输入输出
函数 时间复杂度及说明
str+="lalala"
str.size()/str.length() O(1)O(1)O(1)
str.insert(pos,string) O(N)O(N)O(N)
str.insert(str_it,insert_it_left,insert_it_right) O(N)O(N)O(N)
str.erase(str.begin()+pos) O(N)O(N)O(N)
str.erase(str.begin()+left,str.begin()+right) O(N)O(N)O(N)
str.clear() O(1)O(1)O(1)
str.substr(pos,len) O(len)O(len)O(len)
string::npos = -1 = UINT_MAX find失配的返回值
str.find(str2) 返回str2第一次在str中出现的下标，O(NM)O(NM)O(NM)
str.find(str,pos) 从pos开始找str2
str.replace(pos,len,str2) 从pos开始将长度为len的子串替换为str2，O(str.size())O(str.size())O(str.size())
str.replace(it1,it2,str2) [it1,it2)[it_1,it_2)[it1​,it2​)替换为str2
str.c_str() string to char *
str.back()

# vector

变长数组

  • vector<int>v(N,val);
  • vector<int>v(v1.begin(),v1.end())等价于vector<int>(v1)
  • vector<int>v{1,2,3};
  • v[0]等价于*v.begin()，v.front()
  • 允许v.begin()+i，访问v[i]
  • 可以使用比较运算符
函数 时间复杂度及说明
v.push_back(X) O(1)O(1)O(1)
v.pop_back() O(1)O(1)O(1)
v.size() O(N)O(N)O(N)
v.clear() O(N)O(N)O(N)
v.insert(v.begin()+pos,X) O(N)O(N)O(N)
v.erase(v.begin()+pos) O(N)O(N)O(N)
v.erase(v.begin()+left,v.begin()+right) O(N)O(N)O(N)
v.emplace_back(X) 没有push_back中的移动操作
v.empty() O(1)O(1)O(1)
v.resize(N,val) O(N)O(N)O(N)
v.back()

# priority_queue

优先队列，堆

  • priority_queue<int> q;
  • priority_queue<int,vector<int>,greater<int>> q;
struct node{
	int val;
	...
};
struct cmp{
	bool operator () (const node &a,const node &b){
		return a.val>b.val;
	}
};
priority_queue<node,vector<node>,cmp> q;

# ios::sync_with_stdio

设置标准 C++ 流是否与标准 C 流在每次输入/输出操作后同步
标准 C++ 流为下列者：std::cin 、std::cout、std::cerr、std::clog、std::wcin、std::wcout、std::wcerr和std::wclog
标准 C 流为下列者：stdin、stdout和stderr

std::ios::sync_with_stdio(false);
std::cout << "a\n";
std::printf("b\n");
std::cout << "c\n";
