# 5177. 转变日期格式

这模拟就是说怎么长成这个样子啊……

class Solution {
public:
    string reformatDate(string date) {
        unordered_map<string,string>mp;
        mp["Jan"]="01";
        mp["Feb"]="02";
        mp["Mar"]="03";
        mp["Apr"]="04";
        mp["May"]="05";
        mp["Jun"]="06";
        mp["Jul"]="07";
        mp["Aug"]="08";
        mp["Sep"]="09";
        mp["Oct"]="10";
        mp["Nov"]="11";
        mp["Dec"]="12";
        stringstream ss(date);
        string a,b,c,ans;
        ss>>a>>b>>c;
        string day=a.substr(0,a.size()-2);
        if(day.size()==1)
            day="0"+day;
        string month=mp[b];
        ans=c+"-"+month+"-"+day;
        return ans;
    }
};

# 5445. 子数组和排序后的区间和

暴力写了25分钟……没看到要连续子数组QAQ

class Solution {
public:
    int rangeSum(vector<int>& nums, int n, int left, int right) {
        vector<int>v;
        for(int i=0;i<n;i++){
            v.push_back(nums[i]);
            int sum=nums[i];
            for(int j=i+1;j<n;j++){
                sum+=nums[j];
                v.push_back(sum);
            }
        }
        sort(v.begin(),v.end());
        int ans=0;
        const int mod=1e9+7;
        left--,right--;
        for(int i=left;i<=right;i++){
            ans+=v[i];
            ans%=mod;
        }
        return ans;
    }
};

# 5446. 三次操作后最大值与最小值的最小差

只要确定三次操作后可能的最大最小值然后枚举最小差就完事了

class Solution {
public:
    int minDifference(vector<int>& nums) {
        if(nums.size()<=4)
            return 0;
        sort(nums.begin(),nums.end());
        const int N=nums.size();
        int l1=nums[0],l2=nums[1],l3=nums[2],l4=nums[3];
        int r1=nums[N-1],r2=nums[N-2],r3=nums[N-3],r4=nums[N-4];
        return min(r1-l4,min(r2-l3,min(r3-l2,r4-l1)));
    }
};

# 5447. 石子游戏 IV

dp[i]表示剩iii个石子，先手是否能赢，赢为true

  1. i=0i=0i=0，无石子可操作，先手必输，即dp[0]=false;
  2. 枚举i∈[1,n],j∗j⩽ii\in [1,n],j*j \leqslant ii∈[1,n],j∗j⩽i，若dp[i-j*j]=false，表示先手拿完后，后手拿必输，所以存在一种可能使先手必胜，即dp[i]=true
class Solution {
public:
    bool winnerSquareGame(int n) {
        vector<bool>dp(n+1,false);
        vector<int>v;
        for(int i=1;i*i<=n;i++)
            v.push_back(i*i);
        for(int i=1;i<=n;i++){
            for(int j:v){
                if(j>i)
                    break;
                if(!dp[i-j]){
                    dp[i]=true;
                    break;
                }
            }
        }
        return dp[n];
    }
};
