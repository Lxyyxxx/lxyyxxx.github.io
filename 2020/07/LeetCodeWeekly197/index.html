# 5460. 好数对的数目

签到都长成这样不好吗？

class Solution {
public:
    int numIdenticalPairs(vector<int>& nums) {
        int ans=0;
        const int N=nums.size();
        for(int i=0;i<N;i++){
            for(int j=i+1;j<N;j++){
                if(nums[i]==nums[j])
                    ans++;
            }
        }
        return ans;
    }
};

# 5461. 仅含 1 的子串数

计算连续的111的个数nnn，可组成n(n+1)2\dfrac{n(n+1)}{2}2n(n+1)​个符合条件的子串，累加，记得取模

class Solution {
public:
    typedef long long ll;
    int cal(int x,int mod){
        return 1ll*x*(x+1)/2%mod;
    }
    int numSub(string s) {
        int ans=0;
        const int N=s.size();
        const int mod=1e9+7;
        int cnt=0;
        for(int i=0;i<N;i++){
            if(s[i]=='1')
                cnt++;
            else{
                ans+=cal(cnt,mod);
                ans%=mod;
                cnt=0;
            }
        }
        ans+=cal(cnt,mod);
        ans%=mod;
        return ans;
    }
};

# 5211. 概率最大的路径

DFS TLE一发
Dijkstra + priority_queue，把最短路改成最长路做就ok了

class Solution {
public:
    vector<vector<pair<int,double>>>g;
    vector<bool>vis;
    vector<double>prob;
    double ans;
    void dij(int s,int e){
        priority_queue<pair<double,int>>q;
        prob[s]=1.0;
        q.push({1.0,s});
        while(!q.empty()){
            auto cur=q.top();
            q.pop();
            if(cur.second==e){
                ans=max(ans,cur.first);
                continue;
            }
            if(cur.first<prob[cur.second])
                continue;
            for(auto i:g[cur.second]){
                if(i.second*cur.first>prob[i.first]){
                    prob[i.first]=i.second*cur.first;
                    q.push({prob[i.first],i.first});
                }
            }
        }
    }
    double maxProbability(int n, vector<vector<int>>& edges, vector<double>& succProb, int start, int end) {
        g.resize(n);
        const int N=edges.size();
        for(int i=0;i<N;i++){
            g[edges[i][0]].push_back({edges[i][1],succProb[i]});
            g[edges[i][1]].push_back({edges[i][0],succProb[i]});
        }
        vis.resize(n,false);
        ans=0.0;
        prob.resize(n,0.0);
        dij(start,end);
        return ans;
    }
};
