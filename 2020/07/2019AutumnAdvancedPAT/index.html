bool isprime(int x){
    if(x==1||x==2)
        return false;
    for(int i=2;i*i<=x;i++)
        if(x%i==0)
            return false;
    return true;
}
bool check(int x,int y){
    return isprime(__gcd(x,y));
}
map<int,set<string>>mp; 
// 总个数，当前数字，总和，现在的个数，现在的和，数字 
void dfs(int n,int cur,int m,int cnt,int sum,string num){
//    cout<<n<<" "<<cur<<" "<<m<<" "<<cnt<<" "<<sum<<" "<<num<<endl;
    if(cnt>n||sum>m)
        return;
    if(sum==m&&cnt==n){
//        cout<<num<<endl;
        string tn=num;
        sort(tn.begin(),tn.end());
        unordered_set<string>st;
        do{
//            cout<<tn<<endl;
            if(tn[0]=='0'||st.count(tn))
                continue;
            string tt=to_string(stoi(tn)+1);
            int nn=0;
            for(auto &i:tt)
                nn+=i-'0';
            if(check(nn,m))
                mp[nn].insert(tn);
            st.insert(tn);
        }while(next_permutation(tn.begin(),tn.end()));
//        cout<<"-----\n";
        return;
    }
    for(int i=cur;i>=0;i--){
        if(sum+i>m)
            continue;
        dfs(n,i,m,cnt+1,sum+i,num+to_string(i));
    }
}
int main(){
    int n,k,m;
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>k>>m;
        cout<<"Case "<<i<<'\n';
        if(k*9<m)
            cout<<"No Solution\n";
        else{
            mp.clear();
            dfs(k,9,m,0,0,"");
            if(mp.empty())
                cout<<"No Solution\n";
            else{
                for(auto &i:mp){
                    for(auto &j:i.second){
                        cout<<i.first<<" "<<j<<endl;
                    }
                }
            }
        }
    }
    return 0;
}

# 7-2 Merging Linked Lists (25分)

传送门

# Idea

模拟链表，利用栈存短的那个，每输出两个再输出栈里的一个，最后把剩下的输出

# Code

#include<bits/stdc++.h>
using namespace std;
int l1,l2,n;
struct node{
    int data,nxt;
};
unordered_map<int,node>mp;
int length(int l){
    int tmp=l,cnt=0;
    while(tmp!=-1){
        cnt++;
        tmp=mp[tmp].nxt;
    }
    return cnt;
}
void solve(){
    stack<int>s;
    while(l2!=-1){
        s.push(l2);
        l2=mp[l2].nxt;
    }
    int cnt=0,tmp=l1,t2;
    while(tmp!=-1&&!s.empty()){
        if(tmp!=l1)
            printf(" %05d\n",tmp);
        printf("%05d %d",tmp,mp[tmp].data);
        cnt++;
        if(cnt==2){
            t2=s.top();
            s.pop();
            printf(" %05d\n%05d %d",t2,t2,mp[t2].data);
            cnt=0; 
        }
        tmp=mp[tmp].nxt;
    }
    while(tmp!=-1){
        if(tmp!=-1)
            printf(" %05d\n",tmp);
        printf("%05d %d",tmp,mp[tmp].data);
        tmp=mp[tmp].nxt;
    }
    printf(" -1\n");
}
int main(){
    cin>>l1>>l2>>n;
    int addr,data,nxt;
    for(int i=0;i<n;i++){
        cin>>addr>>data>>nxt;
        mp[addr]={data,nxt};
    }
    if(length(l1)<length(l2))
        swap(l1,l2);
    solve();
    return 0;
}

# 7-3 Postfix Expression (25分)

传送门

# Idea

后序遍历输出语法树，但是在前面的正负号需要放在前面输出，这里在输出的时候需要特判一下
好像mooc的习题有一个跟这个长得差不多的hhh

WA的原因是 主要是看不懂with parentheses reflecting the precedences of the operators这句话…… 只在前面输出了负号，后来就直接把所有在前面的单个符号都判断了，然后过了，最后想想应该是只要正负号就ok了，其他情况也出现不了

# Code

#include<bits/stdc++.h>
using namespace std;
const int N=25;
struct node{
    string val;
    int left,right;
}t[N];
int root=-1;
void build(){
    int n;
    cin>>n;
    bool vis[N]={false};
    for(int i=1;i<=n;i++){
        cin>>t[i].val>>t[i].left>>t[i].right;
        if(t[i].left!=-1)
            vis[t[i].left]=1;
        if(t[i].right!=-1)
            vis[t[i].right]=1;
    }
    for(int i=1;i<=n;i++)
        if(!vis[i]){
            root=i;
            break;
        } 
}
bool isop(string s){
    return (s=="+")||(s=="-");
}
void post(int cur){
    if(cur==-1)
        return;
    bool p=false;
    cout<<'(';
    if(isop(t[cur].val)&&t[cur].left==-1){
        p=true;
        cout<<t[cur].val;
    } 
    post(t[cur].left);
    post(t[cur].right);
    if(!p)
        cout<<t[cur].val;
    cout<<')';
}
int main(){
    build();
    post(root);
    cout<<'\n';
    return 0;
}

# 7-4 Dijkstra Sequence (30分)

传送门

# Idea

害，题目也看不懂，以下是我猜的
判断是否为Dijkstra的顶点访问顺序集合
所以只要看集合中第一个源到其他顶点的最短路是不是非降序排列的就ok了

# Code

#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> pii;
const int N=1e3+7;
int g[N][N],nv,ne,k;
vector<int> dijkstra(int begin){
    priority_queue<pii,vector<pii>,greater<pii>>q;
    q.push({0,begin});
    vector<bool>vis(nv+1,false);
    vector<int>dis(nv+1,1e9);
    dis[begin]=0;
    while(!q.empty()){
        pii cur=q.top();
        q.pop();
        if(vis[cur.second])
            continue;
        vis[cur.second]=true;
        for(int i=1;i<=nv;i++){
            if(vis[i]||g[cur.second][i]==1e9)
                continue;
//            cout<<cur.second<<" "<<i<<endl;
//            cout<<dis[i]<<" "<<dis[cur.second]<<" "<<g[cur.second][i]<<endl;
            if(dis[i]>dis[cur.second]+g[cur.second][i]){
                dis[i]=dis[cur.second]+g[cur.second][i];
                q.push({dis[i],i});
            }
        }
    }
    return dis;
}
int main(){
    fill(g[0],g[0]+N*N,1e9);
    cin>>nv>>ne;
    int u,v,d,pre;
    for(int i=0;i<ne;i++){
        cin>>u>>v>>d;
        g[u][v]=g[v][u]=d;
    }
    cin>>k;
    bool f;
    for(int i=0;i<k;i++){
        cin>>u>>v;
        vector<int>dis=dijkstra(u);
        f=true;
        pre=dis[v];
//        cout<<u<<"\n"<<v<<" "<<dis[v]<<endl;
        for(int j=2;j<nv;j++){
            cin>>v;
            if(dis[v]<pre)
                f=false;
            pre=dis[v];
//            cout<<v<<" "<<dis[v]<<endl;
        }
        if(f)
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}
