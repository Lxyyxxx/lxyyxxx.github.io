# 7-1 A Greeting from Qinhuangdao

rrr个红球，bbb个蓝球，抽两次都是红球的概率

ans=rr+b⋅r−1r+b−1ans=\dfrac{r}{r+b}\cdot \dfrac{r-1}{r+b-1} ans=r+br​⋅r+b−1r−1​

#include <bits/stdc++.h>
using namespace std;
int T, r, b;
int main() {
    scanf("%d", &T);
    for (int i = 1; i <= T; i++) {
        scanf("%d%d", &r, &b);
        int up = r * r - r;
        int down = r + b;
        down *= (down - 1);
        printf("Case #%d: ", i);
        if (up == 0)
            printf("0/1\n");
        else {
            int gcd = __gcd(up, down);
            printf("%d/%d\n", up / gcd, down / gcd);
        }
    }
    return 0;
}

# 7-7 Good Number

满足⌊xk⌋∣x⌊\sqrt[k]{x}⌋|x⌊kx​⌋∣x且x⩽nx \leqslant nx⩽n的xxx个数
枚举i2i^2i2，计算(i−1)2(i-1)^2(i−1)2到i2−1i^2-1i2−1之间被iii整除的个数，累加

#include <bits/stdc++.h>
using namespace std;
int T, n, k;
long long cal(int x, int k) {
    long long ans = 1;
    while (k--) {
        ans *= x;
        if (ans > n) return ans;
    }
    return ans;
}
int main() {
    scanf("%d", &T);
    for (int i = 1; i <= T; i++) {
        scanf("%d%d", &n, &k);
        printf("Case #%d: ", i);
        if (k == 1) {
            printf("%d\n", n);
            continue;
        }
        long long ans = 0;
        int j = 1, part;
        long long pre = 1, cur = cal(j + 1, k);
        for (; cur <= n; pre = cur, j++, cur = cal(j + 1, k)) {
            part = ((cur - 1) / j - pre / j) + 1;
            ans += part;
        }
        part = (n / j - pre / j) + 1;
        ans += part;
        printf("%lld\n", ans);
    }
    return 0;
}
