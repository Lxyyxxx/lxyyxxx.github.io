d=(k−1)+∑i=1n−12∣ai−k∣+∣an−k∣+(an−1)d=(k-1)+\sum_{i=1}^{n-1}2|a_i-k|+|a_n-k|+(a_n-1)d=(k−1)+∑i=1n−1​2∣ai​−k∣+∣an​−k∣+(an​−1)，使ddd最小的就是a−1a-1a−1最小的，即aaa最小的
注意数据要开long long

#include <bits/stdc++.h>
using namespace std;
const int N = 2e6 + 7;
int T, n, a[N], m, k;
int main() {
    scanf("%d", &T);
    while (T--) {
        scanf("%d%d%d", &n, &m, &k);
        for (int i = 0; i < m; i++)
            scanf("%d", &a[i]);
        sort(a, a + m);
        long long ans = (k - 1) + (a[0] - 1) + abs(a[0] - k);
        for (int i = 1; i < m; i++)
            ans += 2 * abs(a[i] - k);
        printf("%lld\n", ans);
    }
    return 0;
}
