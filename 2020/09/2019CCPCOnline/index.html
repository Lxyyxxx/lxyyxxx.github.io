# 6702 ^&^

求使得(A^C)&(B^C)(A\text{\textasciicircum}C)\&(B\text{\textasciicircum}C)(A^C)&(B^C)最小的CCC，若当(A^C)&(B^C)=0(A\text{\textasciicircum}C)\&(B\text{\textasciicircum}C)=0(A^C)&(B^C)=0时，C=0C=0C=0，那么输出111
即求(A&B)^C=0(A\&B)\text{\textasciicircum}C=0(A&B)^C=0的最小CCC，可得C=(A&B)C=(A\&B)C=(A&B)，当C=0C=0C=0时，输出111

#include<bits/stdc++.h>
using namespace std;
unsigned long long a,b;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int _;cin>>_;
    while(_--){
        cin>>a>>b;
        unsigned long long ans=a&b;
        if(ans==0)
            cout<<a<<endl;
        else
            cout<<ans<<endl; 
    }
    return 0;
}

# 6707 Shuffle Card

将相应牌面的牌移到第一个

#include<bits/stdc++.h>
using namespace std;
int n,m,a,num;
list<int>v;
list<int>::iterator it;
unordered_map<int,list<int>::iterator>mp;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin>>n>>m;
    for(int i=0;i<n;i++){
        cin>>a;
        v.push_back(a);
        it=v.end();
        it--;
        mp[a]=it;
    }
    while(m--){
        cin>>num;
        v.erase(mp[num]);
        v.push_front(num);
        mp[num]=v.begin();
    }
    for(auto &i:v)
        cout<<i<<" ";
    return 0;
}
